1 /*!
2 Linear Technology DC1009A-A Demonstration Board.
3 LTC2492: 24-Bit, 4-Channel Delta Sigma ADC with SPI interface
4 
5 @verbatim
6 
7 NOTES
8  Setup:
9  Set the terminal baud rate to 115200 and select the newline terminator. Equipment
10  required is a precision voltage source and a precision voltmeter. Ensure the COM
11  and REF- pins are connected to ground. The REF+ pin should be connected to +5V.
12 
13  How to test Single-Ended mode:
14  The voltage source should be connected to the ADC such that the negative lead is
15  connected to the COM(common) pin. The positive lead may be connected to any
16  channel input. Ensure voltage is within analog input voltage range -0.3 to 2.5V.
17 
18  How to test Differential Mode:
19  The voltage source should be connected with positive and negative leads to paired
20  channels. The voltage source negative output must also be connected to the COM
21  pin in order to provide a ground-referenced voltage. Ensure voltage is within
22  analog input voltage range -0.3V to +2.5V. Swapping input voltages results in a
23  reversed polarity reading.
24 
25 USER INPUT DATA FORMAT:
26  decimal : 1024
27  hex : 0x400
28  octal : 02000 (leading 0 "zero")
29  binary : B10000000000
30  float : 1024.0
31 
32 @endverbatim

147 //! Repeats Linduino loop
148 void loop()
149 {
150  int16_t user_command; // The user input command
151  uint8_t ack_EOC = 0; // Keeps track of the EOC timeout
152  if (demo_board_connected)
153  {
154  if (Serial.available()) // Check for user input
155  {
156  user_command = read_int(); // Read the user command
157  if (user_command != 'm')
158  Serial.println(user_command); // Prints the user command to com port
159  Serial.flush();
160  switch (user_command)
161  {
162  case 1:
163  ack_EOC |= menu_1_read_single_ended();
164  break;
165  case 2:
166  ack_EOC |= menu_2_read_differential();
167  break;
168  case 3:
169  menu_3_set_1X2X();
170  break;
171  case 4:
172  menu_4_select_rejection_freq();
173  break;
174  default:
175  Serial.println(F("Incorrect Option"));
176  }
177  if (ack_EOC)
178  Serial.println(F("\n******SPI ERROR******\n"));
179  Serial.print(F("\n*************************\n"));
180  print_prompt();
181  }
182  }
183 }

234 //! Read channels in single-ended mode
235 //! @return Returns 0=successful, 1=unsuccessful (exceeded timeout)
236 uint8_t menu_1_read_single_ended()
237 {
238  uint8_t adc_command_high; // The LTC2492 command high byte
239  uint8_t adc_command_low; // The LTC2492 command low byte
240  int16_t user_command; // The user input command
241  int32_t adc_code = 0; // The LTC2492 code
242  float adc_voltage = 0; // The LTC2492 voltage
243 
244  while (1)
245  {
246  Serial.print(F("*************************\n\n"));
247  Serial.print(F("0-CH0\n"));
248  Serial.print(F("1-CH1\n"));
249  Serial.print(F("2-CH2\n"));
250  Serial.print(F("3-CH3\n"));
251  Serial.print(F("4-ALL\n"));
252  Serial.print(F("m-Main Menu\n"));
253  Serial.print(F("Enter a Command: "));
254 
255  user_command = read_int(); // Read the single command
256  if (user_command == 'm')
257  break;
258  Serial.println(user_command);
259  if (user_command == 4)
260  {
261  Serial.print(F("ALL\n"));
262  adc_command_high = BUILD_COMMAND_SINGLE_ENDED[0]; // Build ADC command for channel 0
263  adc_command_low = rejection_mode | two_x_mode;
264  if (LTC2492_EOC_timeout(LTC2492_CS, eoc_timeout)) // Checks for EOC with a timeout
265  return 1;
266  LTC2492_read(LTC2492_CS, adc_command_high, adc_command_low, &adc_code); // Throws out last reading
267 
268  for (int8_t x = 0; x <= 3; x++) // Read all channels in single-ended mode
269  {
270  if (two_x_mode)
271  {
272  if (LTC2492_EOC_timeout(LTC2492_CS, eoc_timeout))
273  return 1;
274  LTC2492_read(LTC2492_CS, adc_command_high, adc_command_low, &adc_code); // Throws out an extra reading in 2x mode
275  }
276  adc_command_high = BUILD_COMMAND_SINGLE_ENDED[(x + 1) % 4];
277  if (LTC2492_EOC_timeout(LTC2492_CS, eoc_timeout))
278  return 1;
279  LTC2492_read(LTC2492_CS, adc_command_high, adc_command_low, &adc_code);
280  adc_voltage = LTC2492_code_to_voltage(adc_code, LTC2492_vref);
281 
282  Serial.print(F(" ****"));
283  Serial.print(F("CH"));
284  Serial.print(x);
285  Serial.print(F(": "));
286  Serial.print(adc_voltage, 4);
287  Serial.print(F("V\n\n"));
288  }
289  }
290  else
291  {
292  adc_command_high = BUILD_COMMAND_SINGLE_ENDED[user_command]; // Build ADC command
293  adc_command_low = rejection_mode | two_x_mode;
294  Serial.print(F("ADC Command: 0x"));
295  Serial.println((adc_command_high<<8) | adc_command_low, HEX);
296  if (LTC2492_EOC_timeout(LTC2492_CS, eoc_timeout)) // Checks for EOC with a timeout
297  return 1;
298  LTC2492_read(LTC2492_CS, adc_command_high, adc_command_low, &adc_code); // Throws out last reading
299  if (two_x_mode)
300  {
301  if (LTC2492_EOC_timeout(LTC2492_CS, eoc_timeout))
302  return 1;
303  LTC2492_read(LTC2492_CS, adc_command_high, adc_command_low, &adc_code); // Throws out an extra reading in 2x mode
304  }
305  if (LTC2492_EOC_timeout(LTC2492_CS, eoc_timeout))
306  return 1;
307  LTC2492_read(LTC2492_CS, adc_command_high, adc_command_low, &adc_code); // Now we're ready to read the desired data
308 
309  Serial.print(F("Received Code: 0x"));
310  Serial.println(adc_code, HEX);
311  adc_voltage = LTC2492_code_to_voltage(adc_code, LTC2492_vref);
312  Serial.print(F(" ****"));
313  Serial.print(F("CH"));
314  Serial.print(user_command);
315  Serial.print(F(": "));
316  Serial.print(adc_voltage, 4);
317  Serial.print(F("V\n\n"));
318  }
319  }
320  return 0;
321 }
