1 /*!
2 Linear Technology DC1011A-A Demonstration Board.
3 LTC2498: 24-Bit, 16-Channel Delta Sigma ADC with SPI interface
4 
5 @verbatim
6 
7 NOTES
8  Setup:
9  Set the terminal baud rate to 115200 and select the newline terminator. Equipment
10  required is a precision voltage source and a precision voltmeter. Additionally,
11  an external power supply is required to provide a negative voltage for Amp V-.
12  Set it to anywhere from -1V to -5V. Set Amp V+ to Vcc. Ensure the COM and REF-
13  pins are connected to ground. The REF+ pin should be connected to +5V.
14 
15  How to test Single-Ended mode:
16  The voltage source should be connected to the ADC such that the negative lead is
17  connected to the COM(common) pin. The positive lead may be connected to any
18  channel input. Ensure voltage is within analog input voltage range -0.3 to 2.5V.
19 
20  How to test Differential Mode:
21  The voltage source should be connected with positive and negative leads to paired
22  channels. The voltage source negative output must also be connected to the COM
23  pin in order to provide a ground-referenced voltage. Ensure voltage is within
24  analog input voltage range -0.3V to +2.5V. Swapping input voltages results in a
25  reversed polarity reading.
26 
27  How to calibrate:
28  Apply 100mV CH0 with respect to COM. Next, measure this voltage with a precise
29  voltmeter and enter this value. (This takes the reading.) Now apply approximately
30  2.40 volts to CH0. Measure this voltage with a precise voltmeter and enter this
31  value. Calibration is now stored in EEPROM. Upon start-up the calibration values
32  will be restored.
33 
34 USER INPUT DATA FORMAT:
35  decimal : 1024
36  hex : 0x400
37  octal : 02000 (leading 0 "zero")
38  binary : B10000000000
39  float : 1024.0
40 
41 @endverbatim

160 //! Repeats Linduino loop
161 void loop()
162 {
163  int16_t user_command; // The user input command
164  uint8_t ack_EOC = 0; // Keeps track of the EOC timeout
165  if (demo_board_connected)
166  {
167  if (Serial.available()) // Check for user input
168  {
169  user_command = read_int(); // Read the user command
170  if (user_command != 'm')
171  Serial.println(user_command); // Prints the user command to com port
172  Serial.flush();
173  switch (user_command)
174  {
175  case 1:
176  ack_EOC |= menu_1_read_single_ended();
177  break;
178  case 2:
179  ack_EOC |= menu_2_read_differential();
180  break;
181  case 3:
182  ack_EOC |= menu_3_read_temperature();
183  break;
184  case 4:
185  menu_4_set_1X2X();
186  break;
187  default:
188  Serial.println(F("Incorrect Option"));
189  }
190  if (ack_EOC)
191  Serial.println(F("\n******SPI ERROR******\n"));
192  Serial.print(F("\n*************************\n"));
193  print_prompt();
194  }
195  }
196 }

283 //! Read channels in single-ended mode
284 //! @return Returns 0=successful, 1=unsuccessful (exceeded timeout)
285 uint8_t menu_1_read_single_ended()
286 {
287  uint8_t adc_command_high; // The LTC2498 command high byte
288  uint8_t adc_command_low; // The LTC2498 command low byte
289  int16_t user_command; // The user input command
290  int32_t adc_code = 0; // The LTC2498 code
291  float adc_voltage = 0; // The LTC2498 voltage
292 
293  while (1)
294  {
295  Serial.print(F("*************************\n\n"));
296  Serial.print(F("0-CH0 8-CH8\n"));
297  Serial.print(F("1-CH1 9-CH9\n"));
298  Serial.print(F("2-CH2 10-CH10\n"));
299  Serial.print(F("3-CH3 11-CH11\n"));
300  Serial.print(F("4-CH4 12-CH12\n"));
301  Serial.print(F("5-CH5 13-CH13\n"));
302  Serial.print(F("6-CH6 14-CH14\n"));
303  Serial.print(F("7-CH7 15-CH15\n"));
304  Serial.print(F("16-ALL\n"));
305  Serial.print(F("m-Main Menu\n"));
306  Serial.print(F("Enter a Command: "));
307 
308  user_command = read_int(); // Read the single command
309  if (user_command == 'm')
310  break;
311  Serial.println(user_command);
312 
313  if (user_command == 16)
314  {
315  Serial.print(F("ALL\n"));
316  adc_command_high = BUILD_COMMAND_SINGLE_ENDED[0]; // Build ADC command for channel 0
317  adc_command_low = rejection_mode | two_x_mode;
318  if (LTC2498_EOC_timeout(LTC2498_CS, eoc_timeout))
319  return 1;
320  LTC2498_read(LTC2498_CS, adc_command_high, adc_command_low, &adc_code); // Throws out last reading
321 
322  for (int8_t x = 0; x <= 15; x++) // Read all channels in single-ended mode
323  {
324 
325 
326  adc_command_high = BUILD_COMMAND_SINGLE_ENDED[(x + 1) % 16];
327  if (LTC2498_EOC_timeout(LTC2498_CS, eoc_timeout))
328  return 1;
329  LTC2498_read(LTC2498_CS, adc_command_high, adc_command_low, &adc_code);
330  adc_voltage = LTC2498_code_to_voltage(adc_code, LTC2498_vref);
331 
332  Serial.print(F(" ****"));
333  Serial.print(F("CH"));
334  Serial.print(x);
335  Serial.print(F(": "));
336  Serial.print(adc_voltage, 4);
337  Serial.print(F("V\n\n"));
338  }
339  }
340  else
341  {
342  adc_command_high = BUILD_COMMAND_SINGLE_ENDED[user_command];
343  adc_command_low = rejection_mode | two_x_mode;
344  Serial.print(F("ADC Command: 0x"));
345  Serial.println((adc_command_high<<8) | adc_command_low, HEX);
346 
347  quikeval_SPI_connect();
348  if (LTC2498_EOC_timeout(LTC2498_CS, eoc_timeout))
349  return 1;
350  LTC2498_read(LTC2498_CS, adc_command_high, adc_command_low, &adc_code); // Throws out last reading
351  if (LTC2498_EOC_timeout(LTC2498_CS, eoc_timeout))
352  return 1;
353  LTC2498_read(LTC2498_CS, adc_command_high, adc_command_low, &adc_code); // Now we're ready to read the desired data
354 
355  Serial.print(F("Received Code: 0x"));
356  Serial.println(adc_code, HEX);
357  adc_voltage = LTC2498_code_to_voltage(adc_code, LTC2498_vref);
358  Serial.print(F(" ****"));
359  Serial.print(F("CH"));
360  Serial.print(user_command);
361  Serial.print(F(": "));
362  Serial.print(adc_voltage, 4);
363  Serial.print(F("V\n\n"));
364  }
365  }
366  return 0;
367 }

440  }
441  }
