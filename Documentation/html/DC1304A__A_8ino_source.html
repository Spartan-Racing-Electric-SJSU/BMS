144 //! Set filter settings
145 //! @return void
146 void menu_1_filter_settings()
147 {
148  int16_t user_command; // The user input command
149  while (1)
150  {
151  Serial.print(F("*************************\n\n"));
152  Serial.println (F("0-Set Gain"));
153  Serial.println (F("1-Set Lowpass"));
154  Serial.println (F("2-Set Highpass"));
155  Serial.println (F("3-Upload Filter Settings"));
156  Serial.print (F("m-Main Menu\n"));
157  Serial.print (F("\nEnter a Command: "));
158  user_command = read_int(); // Read the single command
159  if (user_command == 'm')
160  return;
161  else
162  Serial.println(user_command);
163  Serial.println();
164 
165  switch (user_command)
166  {
167  case 0:
168  Serial.println (F("Gain: 0-0dB, 1-12dB, 2-24dB, 3-30dB"));
169  user_command = read_int();
170  switch (user_command)
171  {
172  case 0:
173  control_byte = (control_byte & LTC6602_GAIN_MASK) | LTC6602_GAIN_0dB;
174  break;
175  case 1:
176  control_byte = (control_byte & LTC6602_GAIN_MASK) | LTC6602_GAIN_12dB;
177  break;
178  case 2:
179  control_byte = (control_byte & LTC6602_GAIN_MASK) | LTC6602_GAIN_24dB;
180  break;
181  case 3:
182  control_byte = (control_byte & LTC6602_GAIN_MASK) | LTC6602_GAIN_30dB;
183  break;
184  default:
185  Serial.println (F("incorrect option"));
186  }
187  break;
188  case 1:
189  Serial.println (F("Lowpass Divider: 0-100, 1-100, 2-300, 3-600"));
190  user_command = read_int();
191  switch (user_command)
192  {
193  case 0:
194  control_byte = (control_byte & LTC6602_LPF_MASK) | LTC6602_LPF3;
195  break;
196  case 1:
197  control_byte = (control_byte & LTC6602_LPF_MASK) | LTC6602_LPF2;
198  break;
199  case 2:
200  control_byte = (control_byte & LTC6602_LPF_MASK) | LTC6602_LPF1;
201  break;
202  case 3:
203  control_byte = (control_byte & LTC6602_LPF_MASK) | LTC6602_LPF0;
204  break;
205  default:
206  Serial.println (F("incorrect option"));
207  }
208  break;
209  case 2:
210  Serial.println (F("Highpass Divider: 0-1000, 1-2000, 2-6000, 3-Bypass HPF"));
211  user_command = read_int();
212  switch (user_command)
213  {
214  case 0:
215  control_byte = (control_byte & LTC6602_HPF_MASK) | LTC6602_HPF2;
216  break;
217  case 1:
218  control_byte = (control_byte & LTC6602_HPF_MASK) | LTC6602_HPF1;
219  break;
220  case 2:
221  control_byte = (control_byte & LTC6602_HPF_MASK) | LTC6602_HPF0;
222  break;
223  case 3:
224  control_byte = (control_byte & LTC6602_HPF_MASK) | LTC6602_HPF3;
225  break;
226  default:
227  Serial.println (F("incorrect option"));
228  }
229  break;
230  case 3:
231  spi_out = control_byte;
232  LTC6602_write(LTC6602_CS, &spi_out, (uint8_t)1);
233  break;
234  default:
235  Serial.println("Incorrect Option");
236  break;
237  }
238  Serial.print ("\Control byte 0x" + String(control_byte, HEX) + "\n");
239  }

242 void menu_2_set_gpo()
243 {
244  int16_t user_command; // The user input command
245 
246  while (1)
247  {
248  Serial.print(F("*************************\n\n"));
249  Serial.println(F("0-Set GPIO Low"));
250  Serial.println (F("1-Set GPIO High"));
251  Serial.print(F("m-Main Menu\n"));
252  Serial.print(F("\nEnter a Command: "));
253  user_command = read_int(); // Read the single command
254  if (user_command == 'm')
255  return;
256  else
257  Serial.println(user_command);
258  Serial.println();
259  switch (user_command)
260  {
261  case 0:
262  control_byte = (control_byte & LTC6602_GPO_MASK) | LTC6602_GPO_LOW;
263  spi_out = control_byte;
264  LTC6602_write(LTC6602_CS, &spi_out, (uint8_t)1);
265  break;
266  case 1:
267  control_byte = (control_byte & LTC6602_GPO_MASK) | LTC6602_GPO_HIGH;
268  spi_out = control_byte;
269  LTC6602_write(LTC6602_CS, &spi_out, (uint8_t)1);
270  break;
271  default:
272  Serial.println (F("incorrect option"));
273  break;
274  }
