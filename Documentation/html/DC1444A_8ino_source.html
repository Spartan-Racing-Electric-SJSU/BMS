1 /*!
2 Linear Technology DC1337A Demonstration Board.
3 LTC2301: 1-Channel, 12-Bit SAR ADC with I2C Interface.
4 LTC2305: 2-Channel, 12-Bit SAR ADC with I2C Interface.
5 
6 @verbatim
7 
8 NOTES
9  Setup:
10  Set the terminal baud rate to 115200 and select the newline terminator. Equipment
11  required is a voltage source (preferably low-noise) and a precision voltmeter.
12  Ensure all jumpers on the demo board are installed in their default positions
13  from the factory. Refer to Demo Manual DC1444A.
14 
15  LTC2301
16  How to test Differential Mode:
17  Bipolar Mode:
18  The voltage source should be connected between inputs VIN+ and VIN-. Ensure both
19  inputs are within their specified absolute input voltage range. (It is easiest
20  to tie the voltage source negative terminal to COM.) Ensure the voltage
21  source is set within the range of 0V to +4.096V (differential voltage range).
22  (Swapping input voltages results in a reversed polarity reading.)
23 
24  Unipolar Mode:
25  The voltage source should be connected between inputs VIN+ and VIN-. Ensure both
26  inputs are within their specified absolute input voltage range. (It is easiest
27  to tie the voltage source negative terminal to COM.) Ensure the voltage
28  source is set within the range of 0V to +4.096V (differential voltage range).
29 
30 LTC2305
31  How to test Single-Ended mode:
32  Bipolar Mode:
33  The voltage source should be connected to input VCH0 or VCH1. Ensure the input is
34  within its specified absolute input voltage range. (It is easiest
35  to tie the voltage source negative terminal to COM.) Ensure the voltage
36  source is set within the range of 0V to +4.096V.
37  (Swapping input voltages results in a reversed polarity reading.)
38 
39  Unipolar Mode:
40  The voltage source should be connected to input VCH0 or VCH1. Ensure the input is
41  within its specified absolute input voltage range. (It is easiest
42  to tie the voltage source negative terminal to COM.) Ensure the voltage
43  source is set within the range of 0V to +4.096V.

150 //! Repeats Linduino loop
151 void loop()
152 {
153  int8_t user_command; // The user input command
154  uint8_t acknowledge = 0; // I2C acknowledge bit
155 
156  if (Serial.available()) // Check for user input
157  {
158  user_command = read_int(); // Read the user command
159  if (user_command != 'm')
160  Serial.println(user_command); // Prints the user command to com port
161  Serial.flush();
162  switch (user_command)
163  {
164  case 1:
165  menu_1_read_input();
166  break;
167  case 2:
168  menu_2_select_part();
169  break;
170  case 3:
171  menu_3_select_uni_bipolar();
172  break;
173  case 4:
174  menu_4_select_single_ended_differential();
175  break;
176  case 5:
177  menu_5_select_polarity();
178  break;
179  case 6:
180  menu_6_sleep();
181  break;
182  case 7:
183  menu_7_set_address();
184  break;
185  default:
186  Serial.println(F("Invalid Option"));
187  break;
188  }
189  Serial.println();
190  if (acknowledge != 0) Serial.println(F("Error: No Acknowledge. Check I2C Address."));
191  Serial.println(F("*************************"));
192  print_prompt();
193  }
194 }

198 //! Read channels in single-ended mode
199 //! @return 1 if successful, 0 if not
200 int8_t menu_1_read_input()
201 {
202  uint8_t user_command;
203  uint16_t adc_command; // The LTC2305 command byte
204  uint16_t adc_code = 0; // The LTC2305 code
205  uint8_t x, y, startcount, endcount;
206  uint16_t display_code;
207  float adc_voltage;
208  uint8_t acknowledge = 0; // I2C acknowledge bit
209 
210  if (part == LTC2305)
211  {
212  if (single_ended_differential == LTC2305_SINGLE_ENDED_MODE)
213  {
214  if (uni_bipolar == LTC2305_UNIPOLAR_MODE)
215  {
216  Serial.println(F(" Read LTC2305 Input in Single-Ended Unipolar mode:"));
217  Serial.println(F(" Note that in Unipolar mode, input voltages less than zero are reported as 0.0V"));
218  }
219  else
220  Serial.println(F(" Read LTC2305 Input in Single-Ended Bipolar mode:"));
221  user_command = print_user_command_single_ended();
222  }
223  else
224  {
225  if (uni_bipolar == LTC2305_UNIPOLAR_MODE)
226  {
227  Serial.println(F(" Read LTC2305 Input in Differential Unipolar mode:"));
228  Serial.println(F(" Note that in Unipolar mode, input voltages less than zero are reported as 0.0V"));
229  }
230  else
231  Serial.println(F(" Read LTC2305 Input in Differential Bipolar mode:"));
232  }
233  }
234 
235  else
236  {
237  if (uni_bipolar == LTC2305_UNIPOLAR_MODE)
238  {
239  Serial.println(F(" Read LTC2301 Input in Differential Unipolar mode:"));
240  Serial.println(F(" Note that in Unipolar mode, input voltages less than zero are reported as 0.0V"));
241  }
242  else
243  Serial.println(F(" Read LTC2301 Input in Differential Bipolar mode:"));
244  }
245 
246  if (user_command == 8) //read both channels
247  {
248  startcount = 0;
249  endcount = 1;
250  }
251  else
252  {
253  startcount = user_command;
254  endcount = user_command;
255  }
256 
257  if (part == LTC2305) //read LTC2305
258  {
259  if (single_ended_differential == LTC2305_SINGLE_ENDED_MODE)
260  {
261  for (int8_t x = startcount; x <= endcount; x++)
262  {
263  adc_command = BUILD_COMMAND_SINGLE_ENDED[(x+1) % 2] | polarity | uni_bipolar; // Send channel config for the NEXT conversion to take place
264  acknowledge |= LTC2305_read(LTC2305_I2C_ADDRESS, adc_command, &adc_code); // Throws out last reading and starts CH0 conversion
265  acknowledge |= LTC2305_read(LTC2305_I2C_ADDRESS, adc_command, &adc_code); // Read previous channel conversion (x-1) and start next one (x)
266 
267  display_code = adc_code >> (16 - LTC2305_bits);
268  display_code = display_code & 0xFFF;
269 
270  Serial.print(F(" Received Code: b"));
271  Serial.println(display_code, BIN);
272 
273  adc_voltage = LTC2305_code_to_voltage(adc_code, LTC2305_vref, uni_bipolar);
274  Serial.print(F(" Voltage read on "));
275  Serial.print(F("Ch"));
276  Serial.print(x);
277  Serial.print(F(": "));
278  Serial.print(adc_voltage, 4);
279  Serial.println(F("V"));
280  Serial.println();
281  }
282  }
283 
284  else //differential reading
285  {
286  adc_command = polarity | uni_bipolar; // Send channel config for the NEXT conversion to take place
287  acknowledge |= LTC2305_read(LTC2305_I2C_ADDRESS, adc_command, &adc_code); // Throws out last reading and starts CH0 conversion
288  acknowledge |= LTC2305_read(LTC2305_I2C_ADDRESS, adc_command, &adc_code); // Read previous channel conversion (x-1) and start next one (x)
289 
290  display_code = adc_code >> (16 - LTC2305_bits);
291  display_code = display_code & 0xFFF;
292 
293  Serial.print(F(" Received Code: b"));
294  Serial.println(display_code, BIN);
295 
296  adc_voltage = LTC2305_code_to_voltage(adc_code, LTC2305_vref, uni_bipolar);
297  Serial.print(F(" Voltage read on "));
298  Serial.print(F("Ch"));
299  Serial.print(x);
300  Serial.print(F(": "));
301  Serial.print(adc_voltage, 4);
302  Serial.println(F("V"));
303  Serial.println();
304  }
305  }
306 
307  else //read LTC2301
308  {
309  adc_command = polarity | uni_bipolar;
310  acknowledge |= LTC2305_read(LTC2305_I2C_ADDRESS, adc_command, &adc_code); // Throws out last reading and starts new conversion
311  acknowledge |= LTC2305_read(LTC2305_I2C_ADDRESS, adc_command, &adc_code);
312 
313  display_code = adc_code >> (16 - LTC2305_bits);
314  display_code = display_code & 0xFFF;
315 
316  Serial.print(F(" Received Code: b"));
317  Serial.println(display_code, BIN);
318 
319  adc_voltage = LTC2305_code_to_voltage(adc_code, LTC2305_vref, uni_bipolar);
320  Serial.print(F(" Voltage read on "));
321  Serial.print(F("Ch"));
322  Serial.print(user_command);
323  Serial.print(F(": "));
324  Serial.print(adc_voltage, 4);
325  Serial.println(F("V"));
326  Serial.println();
327  }
328  return(acknowledge);
329 }
