11 @verbatim
12 
13 NOTES
14  Setup:
15  Set the terminal baud rate to 115200 and select the newline terminator. A voltage
16  source (preferably low-noise) and a precision voltmeter are required. Ensure all
17  jumpers on the demo board are installed in their default positions from the
18  factory. Refer to Demo Manual DC1491A.
19  * Run menu entry 4 if the voltage readings are not accurate. The DC1491A is not
20  calibrated in the factory. Default values will be used until first customer
21  calibration.
22  If the voltage readings are in error, run menu entry 4 again. The previous
23  calibration by the user may have been performed improperly.
24 
25  Menu Entry 1: Read ADC Voltage
26  Ensure the applied voltage is within the analog input voltage range of 0V to
27  +1.25V. Connect the voltage source between the DC1491A's IN terminal and GND
28  terminal. This menu entry configures the LTC2461 to operate in 30 Hz mode.
29 
30  Menu Entry 2: Sleep Mode
31  Enables sleep mode. Note that the REFOUT voltage will go to zero. Any
32  subsequent read command will wake up the LTC2461.
33 
34  Menu Entry 3: 60 Hz Speed Mode
35  Setup is the same as for Menu Entry 1, except the LTC2461 operates in 60 Hz mode,
36  and continuous background offset calibration is not performed.
37 
38  Menu Entry 4: Calibration
39  Follow the command cues to alternately enter voltages with VIN at GND and with
40  VIN near full scale voltage, approximately 1.20V. Upon startup, the calibration
41  values will be restored.
42 
43 USER INPUT DATA FORMAT:
44  decimal : 1024
45  hex : 0x400
46  octal : 02000 (leading 0 "zero")
47  binary : B10000000000
48  float : 1024.0

148 //! Repeats Linduino loop
149 void loop() // ***** BEGIN MAIN CONTROL LOOP *****
150 {
151  int8_t ack = 0; // Set I2C acknowledge bit to zero at start of every loop.
152  if (demo_board_connected)
153  {
154  if (Serial.available()) // Check for user input
155  {
156  uint16_t user_command;
157  user_command = read_int(); // Read the user command
158  Serial.println(user_command);
159 
160  switch (user_command)
161  {
162  // 30 Hz Standard Read Mode
163  case 1:
164  ack |= menu_1_read_30Hz_mode();
165  break;
166 
167  case 2:
168  ack |= menu_2_sleep_mode();
169  break;
170 
171  case 3:
172  // 60 Hz Read Mode
173  ack |= menu_3_read_60Hz_mode();
174  break;
175 
176  case 4:
177  // Calibration
178  ack |= menu_4_calibrate();
179  break;
180 
181  default:
182  Serial.println("***** INCORRECT ENTRY *****"); // Input validation
183  }
184  Serial.println();
185  if (ack != 0)
186  {
187  Serial.println("Error: No Acknowledge. Check I2C Address.");
188  Serial.println("***** ACK ERROR *****");
189  }
190 
191  Serial.println("\n*****************************************************************");
192  print_prompt();
193  }
194  }
195 }

227 //! Store measured calibration parameters to nonvolatile EEPROM on demo board
228 void store_calibration()
229 // Store the ADC calibration to the EEPROM
230 {
231  eeprom_write_int16(EEPROM_I2C_ADDRESS, EEPROM_CAL_KEY, EEPROM_CAL_STATUS_ADDRESS); // Cal key
232  eeprom_write_int32(EEPROM_I2C_ADDRESS, LTC2461_offset_code, EEPROM_CAL_STATUS_ADDRESS+2); // Store Offset

237 //! Read stored calibration parameters from nonvolatile EEPROM on demo board
238 //! @return Return 1 if successful, 0 if not
239 int8_t restore_calibration()
240 {
241 // Read the calibration from EEPROM
242 // Return 1 if successful, 0 if not
243 
244  int16_t cal_key;
245  // Read the cal key from the EEPROM
246  eeprom_read_int16(EEPROM_I2C_ADDRESS, &cal_key, EEPROM_CAL_STATUS_ADDRESS);
247  if (cal_key == EEPROM_CAL_KEY)
248  {
249  // Calibration has been stored, read offset and lsb
250  eeprom_read_int32(EEPROM_I2C_ADDRESS, &LTC2461_offset_code, EEPROM_CAL_STATUS_ADDRESS+2); // Offset
251  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2461_lsb, EEPROM_CAL_STATUS_ADDRESS+6); // LSB
252  Serial.println("Calibration Restored");
253  return(1);
254  }
255  else
256  {
257  Serial.println("Calibration not found");
258  return(0);
259  }
260 }

262 //! Read ADC in 30 Hz mode
263 //! @return ACK bit (0=acknowledge, 1=no acknowledge)
264 int8_t menu_1_read_30Hz_mode()
265 {
266  uint16_t adc_code_30Hz;
267  int8_t ack = 0;
268  Serial.println();
269  Serial.print("ADC Command: b");
270  Serial.println(LTC2461_60HZ_SPEED_MODE, BIN);
271  ack |= LTC2461_read(LTC2461_I2C_ADDRESS, LTC2461_30HZ_SPEED_MODE, &adc_code_30Hz); // Throw out this reading
272  delay(100);
273  ack |= LTC2461_read(LTC2461_I2C_ADDRESS, LTC2461_30HZ_SPEED_MODE, &adc_code_30Hz);
274  delay(100);
275  Serial.print("Received Code: 0x");
276  Serial.println(adc_code_30Hz, HEX);
277 
278  // Use the calibration values stored on the DC1491A EEPROM to convert the raw_adc_reading to a (float) voltage value which is stored in adc_voltage_30Hz.
279  // Run menu entry 4 if the voltage readings are not accurate. The DC1491A is not calibrated in the factory. Default values will be used until first customer calibration.
280  // If the voltage readings are in error, run menu 4 again. The previous calibration by the user may have been performed improperly.
281 
282  float adc_voltage_30Hz;
283  adc_voltage_30Hz = LTC2461_code_to_voltage(adc_code_30Hz, LTC2461_lsb, LTC2461_offset_code);
284  Serial.println();
285  Serial.print("Voltage Reading: ");
286  Serial.print(adc_voltage_30Hz, 4);
287  Serial.println("V");
288  return(ack);
289 }

339 //! Calibrate the LTC2461
340 //! @return ACK bit (0=acknowledge, 1=no acknowledge)
341 int8_t menu_4_calibrate()
342 {
343  // Calibration
344  uint16_t zero_code; // Cal zero code
345  float zero_voltage;
346  int8_t ack = 0;
347 
348  Serial.println("Apply 100mV to the input.");
349  Serial.println("or apply a voltage for the lower point in two point calibration");
350  Serial.print("Enter the measured input voltage:");
351  zero_voltage = read_float();
352  Serial.println(zero_voltage, 6);
353 
354  ack |= LTC2461_read(LTC2461_I2C_ADDRESS, LTC2461_30HZ_SPEED_MODE, &zero_code); // Throw out this reading
355  delay(100);
356  ack |= LTC2461_read(LTC2461_I2C_ADDRESS, LTC2461_30HZ_SPEED_MODE, &zero_code); // Measure zero
357 
358  Serial.println("Apply ~1.20V input voltage to input.");
359  Serial.println("Enter the measured input voltage:");
360  float fs_voltage; // Measured cal voltage
361  fs_voltage = read_float(); // read function in UserInterface.h: reads float from serial input
362  Serial.println(fs_voltage, 6);
363 
364  uint16_t fs_code; // Cal full scale code
365  ack |= LTC2461_read(LTC2461_I2C_ADDRESS, LTC2461_30HZ_SPEED_MODE, &fs_code); // Throw out this reading
366  delay(100);
367  ack |= LTC2461_read(LTC2461_I2C_ADDRESS, LTC2461_30HZ_SPEED_MODE, &fs_code); // Measure full scale
368  LTC2461_cal_voltage(zero_code, fs_code, zero_voltage, fs_voltage, &LTC2461_lsb, &LTC2461_offset_code); // Modifies offset code and lsb size using linear interpolation
369 
370  Serial.print("ADC offset code: ");
371  Serial.print(LTC2461_offset_code);
372  Serial.println();
373  Serial.print("ADC calibrated lsb size : ");
374  Serial.print(LTC2461_lsb * 1.0e6, 4);
375  Serial.println(" uV (16-bits)");
376  store_calibration(); // Store cal factors in EEPROM
377  return(ack);
378 }
