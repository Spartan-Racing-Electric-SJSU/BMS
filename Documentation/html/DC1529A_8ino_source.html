1 /*!
2 Linear Technology DC1703A Demonstration Board.
3 LTC2657: Octal I2C 16-/12-Bit Rail-to-Rail DACs with 10ppm/C Max Reference.
4 
5 @verbatim
6 NOTES
7  Setup:
8  Set the terminal baud rate to 115200 and select the newline terminator.
9  The program displays calculated voltages which are based on the voltage
10  of the reference used, be it internal or external. A precision voltmeter
11  is needed to verify the actual measured voltages against the calculated
12  voltage displayed. If an external reference is used, a precision voltage
13  source is required to apply the external reference voltage. A
14  precision voltmeter is also required to measure the external reference
15  voltage. No external power supply is required. Any assembly option may
16  be used: DC1529A-A, DC1529A-B
17 
18 
19  Explanation of Commands:
20  1- Select DAC: Select one of eight DACs to test : A, B, C, D, E, F, G, H.
21 
22  2- Write to DAC input register: Value is stored in the DAC for updating
23  later, allowing multiple channels to be updated at once, either
24  through a software "Update All" command or by asserting the LDAC# pin.
25  User will be prompted to enter either a code in hex or decimal, or a
26  voltage. If a voltage is entered, a code will be calculated based on
27  the active scaling and reference parameters - ideal values if no
28  calibration was ever stored.
29 
30  3- Write and Update: Similar to item 1, but DAC is updated immediately.
31 
32  4- Update DAC: Copies the value from the input register into the DAC
33  Register. Note that a "write and update" command writes the code to
34  BOTH the input register and DAC register, so subsequent "update"
35  commands will simply re-copy the same data (no change in output.)
36 
37  5- Power Down DAC: Disable DAC output. Power supply current is reduced.
38  DAC code present in DAC registers at time of shutdown are preserved.
39 
40  6- Set reference mode, either internal or external: Selecting external
41  mode prompts for the external reference voltage, which is used directly
42  if no individual DAC calibration is stored. The selection and entered
43  voltage are stored to EEPROM so it is persistent across reset / power cycles.
44 
45  7- Calibrate DAC: Use a precision voltmeter to obtain and enter VOUT
46  readings taken with different DAC codes. Set reference mode FIRST,
47  as values are stored separately for internal and external reference
48  mode. Entries are used to calculate the closest code to send to the
49  DAC to achieve an entered voltage.
50 
51  8- Enable / Disable calibration: Switch between stored calibration
52  values and defaults. Calibration parameters are stored separately
53  for internal and external reference modes. Ideal calibration will
54  be used if the calibration parameter valid key is not set.
55 
56 
57 USER INPUT DATA FORMAT:
58  decimal : 1024
59  hex : 0x400
60  octal : 02000 (leading 0 "zero")
61  binary : B10000000000
62  float : 1024.0
63 
64 @endverbatim

200 //! Repeats Linduino loop
201 void loop()
202 {
203  int8_t ack=0;
204  int16_t user_command;
205  static int16_t selected_dac = 0; // The selected DAC to be updated (0=A, 1=B ... 8=All). Initialized to "A".
206  // The main control loop
207  if (demo_board_connected) // Do nothing if the demo board is not connected
208  {
209  if (Serial.available()) // Check for user input
210  {
211  user_command = read_int(); // Read the user command
212  Serial.println(user_command);
213  Serial.flush();
214  ack = 0;
215  switch (user_command)
216  {
217  case 1:
218  ack |= menu_1_select_dac(&selected_dac);
219  break;
220  case 2:
221  ack |= menu_2_write_to_input_register(selected_dac);
222  break;
223  case 3:
224  ack |= menu_3_write_and_update_dac(selected_dac);
225  break;
226  case 4:
227  ack |= menu_4_update_power_up_dac(selected_dac);
228  break;
229  case 5:
230  ack |= menu_5_power_down_dac(selected_dac);
231  break;
232  case 6:
233  ack |= menu_6_set_reference_mode(); // int, ext, if ext, prompt for voltage
234  ack |= restore_calibration();
235  break;
236  case 7:
237  ack |= menu_7_calibrate_dacs();
238  restore_calibration();
239  break;
240  case 8:
241  menu_8_enable_calibration();
242  restore_calibration();
243  break;
244  default:
245  Serial.println("Incorrect Option");
246  break;
247  }
248  if (ack) Serial.println("I2C NACK received, check address\n");
249  Serial.println("\n*****************************************************************");
250  print_prompt(selected_dac);
251  }
252  }
253 }

393 //! Read stored calibration parameters from nonvolatile EEPROM on demo board
394 //! @return Return 1 if successful, 0 if not
395 int8_t restore_calibration()
396 // Read the DAC calibration from EEPROM
397 // Return 1 if successful, 0 if not
398 {
399  int8_t ack=0;
400  int16_t intvalid, extvalid;
401  uint8_t i, eeaddr;
402  float dac_count; // The number of codes, 4096 for 12 bits, 65536 for 16 bits
403 
404  Serial.println(F("\n\nReading Calibration parameters from EEPROM..."));
405  float full_scale; // To avoid confusion - in internal ref mode, FS=Vref, in ext mode, FS=2*Vref
406  // Read the cal keys from the EEPROM.
407  eeprom_read_int16(EEPROM_I2C_ADDRESS, &intvalid, INT_CAL_VALID_BASE);
408  eeprom_read_int16(EEPROM_I2C_ADDRESS, &extvalid, EXT_CAL_VALID_BASE);
409  // Read the stored reference state
410  eeprom_read_byte(EEPROM_I2C_ADDRESS, (char *) &reference_mode, STORED_REF_STATE_BASE);
411  // Read external ref V unconditionally, overwrite with defaults if no cal found
412  eeprom_read_float(EEPROM_I2C_ADDRESS, &reference_voltage, EXT_REF_V_BASE);
413 
414  if (reference_mode == REF_EXTERNAL)
415  {
416  Serial.println(F("Restored external ref. Voltage:"));
417  Serial.println(reference_voltage, 5);
418  }
419  else // EITHER reference is set to internal, OR not programmed in which case default to internal
420  {
421  reference_mode = REF_INTERNAL; // Redundant if already set
422  Serial.println("Internal reference mode set");
423  }
424 
425  // Write the reference mode to the DAC right away
426  ack |= LTC2657_write(LTC2657_I2C_ADDRESS, reference_mode, 0x0F, 0x0000);
427 
428  // Set up default values, shift count, DAC count
429  // Calibration parameters MAY be changed next, if match
430  // between reference mode and stored calibration
431  full_scale = reference_voltage * 2.0; // If external ref mode, this applies.
432 
433  // The following two IF statements are used to allow the program to run with
434  // a QuikEval string that does not contain the demo board option.
435  // If the demo board option is found then these values are overwritten.
436  if (strcmp(demo_board.product_name, "LTC2657-L") == 0)
437  {
438  // LTC2657CUF-L16, 16-bits, 2.5V full scale
439  shift_count = 0;
440  if (reference_mode == REF_INTERNAL) full_scale = 2.5;
441  dac_count = 65536;
442  }
443  if (strcmp(demo_board.product_name, "LTC2657-H") == 0)
444  {
445  // LTC2657CUF-H16, 16-bits, 4.096V full scale
446  shift_count = 0;
447  if (reference_mode == REF_INTERNAL) full_scale = 4.096;
448  dac_count = 65536;
449  }
450 
451  switch (demo_board.option)
452  {
453  case 'A':
454  // LTC2657CUF-L16, 16-bits, 2.5V full scale
455  shift_count = 0;
456  if (reference_mode == REF_INTERNAL) full_scale = 2.5;
457  dac_count = 65536;
458  break;
459  case 'B':
460  // LTC2657CUF-H16, 16-bits, 4.096V full scale
461  shift_count = 0;
462  if (reference_mode == REF_INTERNAL) full_scale = 4.096;
463  dac_count = 65536;
464  break;
465  case 'C':
466  // LTC2657CUF-L12, 12-bits, 2.5V full scale
467  // Note: There are no demo boards for the 12-bit version. These are shown to
468  // show how to implement the 12-bit versions.
469  shift_count = 4;
470  if (reference_mode == REF_INTERNAL) full_scale = 2.5;
471  dac_count = 4096;
472  break;
473  case 'D':
474  // LTC2657CUF-H12, 12-bits, 4.096V full scale
475  // Note: There are no demo boards for the 12-bit version. These are shown to
476  // show how to implement the 12-bit versions.
477  shift_count = 4;
478  if (reference_mode == REF_INTERNAL) full_scale = 4.096;
479  dac_count = 4096;
480  break;
481  }
482 
483  for (i = 0; i <= 8; i++)
484  {
485  LTC2657_offset[i] = 0;
486  LTC2657_lsb[i] = full_scale / dac_count;
487  }
488 
489  // Restore calibration IF reference mode matches stored calibration
490  eeaddr = 0; // Assume no calibration present or mismatch between cal and reference mode
491 
492  if ((intvalid == EEPROM_CAL_KEY) && (reference_mode == REF_INTERNAL))
493  {
494  eeaddr = INT_CAL_PARAMS_BASE;
495  Serial.println(F("Found internal calibration, restoring...)"));
496  }
497  else if ((extvalid == EEPROM_CAL_KEY) && (reference_mode == REF_EXTERNAL))
498  {
499  eeaddr = EXT_CAL_PARAMS_BASE;
500  Serial.println(F("Found external calibration, restoring...)"));
501  }
502  else Serial.println(F("Calibration not found for this\nreference mode, using ideal calibration"));
503 
504  if (eeaddr != 0) // If cal key was enabled and reference mode is correct, read offset and lsb
505  {
506  eeprom_read_int16(EEPROM_I2C_ADDRESS, &LTC2657_offset[0], eeaddr);
507  eeprom_read_int16(EEPROM_I2C_ADDRESS, &LTC2657_offset[1], eeaddr + 2);
508  eeprom_read_int16(EEPROM_I2C_ADDRESS, &LTC2657_offset[2], eeaddr + 4);
509  eeprom_read_int16(EEPROM_I2C_ADDRESS, &LTC2657_offset[3], eeaddr + 6);
510  eeprom_read_int16(EEPROM_I2C_ADDRESS, &LTC2657_offset[4], eeaddr + 8);
511  eeprom_read_int16(EEPROM_I2C_ADDRESS, &LTC2657_offset[5], eeaddr + 10);
512  eeprom_read_int16(EEPROM_I2C_ADDRESS, &LTC2657_offset[6], eeaddr + 12);
513  eeprom_read_int16(EEPROM_I2C_ADDRESS, &LTC2657_offset[7], eeaddr + 14);
514  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2657_lsb[0], eeaddr + 16);
515  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2657_lsb[1], eeaddr + 20);
516  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2657_lsb[2], eeaddr + 24);
517  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2657_lsb[3], eeaddr + 28);
518  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2657_lsb[4], eeaddr + 32);
519  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2657_lsb[5], eeaddr + 36);
520  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2657_lsb[6], eeaddr + 40);
521  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2657_lsb[7], eeaddr + 44);
522  LTC2657_offset[8] = LTC2657_offset[0]; // Copy cal value for DAC A to cal value for
523  LTC2657_lsb[8] = LTC2657_lsb[0]; // DAC ALL
524  Serial.println("Calibration Restored");
525  }
526  for (i=0; i<=8; ++i)
527  {
528  Serial.print("DAC ");
529  Serial.print((char) ('A' + i));
530  Serial.print(" offset: ");
531  Serial.print(LTC2657_offset[i]);
532  Serial.print(" , lsb: ");
533  Serial.print(LTC2657_lsb[i]*1000, 4);

667 //! Calibrate the selected DAC using a voltmeter. The routine
668 //! does a linear curve fit given two data points.
669 //! @return ACK bit (0=acknowledge, 1=no acknowledge)
670 int8_t calibrate_dac(uint8_t index)
671 {
672  int8_t ack=0;
673  uint16_t code1 = 0x0200; //! Calibration code 1
674  uint16_t code2 = 0xFFFF; //! Calibration code 2
675  float voltage1; //! Calibration voltage 1
676  float voltage2; //! Calibration voltage 2
677  Serial.println("");
678  Serial.print("Calibrating DAC ");
679  Serial.println((char) (0x41 + index));
680  // Left align 12-bit code1 to 16 bits & write to DAC
681  ack |= LTC2657_write(LTC2657_I2C_ADDRESS, LTC2657_CMD_WRITE_UPDATE, index, code1 << shift_count);
682  Serial.print("DAC code set to 0x");
683  Serial.println(code1, HEX);
684  Serial.print("Enter measured DAC voltage:");
685  voltage1 = read_float();
686  Serial.print(voltage1, 6);
687  Serial.println(" V");
688  // Left align 12-bit code2 to 16 bits & write to DAC
689  ack |= LTC2657_write(LTC2657_I2C_ADDRESS, LTC2657_CMD_WRITE_UPDATE, index, code2 << shift_count);

694  Serial.print(voltage2, 6);
695  Serial.println(" V");

697  return(ack);
