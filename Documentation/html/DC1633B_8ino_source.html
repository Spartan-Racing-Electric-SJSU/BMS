91 //! Repeats Linduino loop
92 //! @return void
93 void loop()
94 {
95  uint8_t user_command;
96  uint8_t *addresses = NULL;
97  uint16_t res;
98 
99  if (Serial.available()) //! Checks for user input
100  {
101  user_command = read_int(); //! Reads the user command
102  if (user_command != 'm')
103  Serial.println(user_command);
104 
105  switch (user_command) //! Prints the appropriate submenu
106  {
107  case 0:
108  break;
109  case 1:
110  //! write all LTC2933 registers with demo-board default settings
111  //! all channels except 1 expect 2.0v from the DAC
112  ltc2933_demo_board_defaults(ltc2933_i2c_address);
113  break;
114  case 2:
115  //! read the LTC2933 STATUS_WORD register
116  res = smbus->readWord(ltc2933_i2c_address, LTC2933_STATUS_WORD);
117  Serial.println(res, HEX);
118  break;
119  case 3:
120  //! read all LTC2933 registers
121  ltc2933_read_registers(ltc2933_i2c_address);
122  break;
123  case 4:
124  //! write to the LTC2637 DACs on the DC1633 board to interesting voltages
125  //! other than the 2.0v DAC defaults
126  dc1633_write_dac_voltage(dc1633_dac_address, 0, 3.3);
127  dc1633_write_dac_voltage(dc1633_dac_address, 1, 2.5);
128  dc1633_write_dac_voltage(dc1633_dac_address, 2, 1.8);
129  dc1633_write_dac_voltage(dc1633_dac_address, 3, 1.5);
130  dc1633_write_dac_voltage(dc1633_dac_address, 4, 1.2);
131  dc1633_write_dac_voltage(dc1633_dac_address, 5, 1.0);
132  break;
133  case 5:
134  //! write all LTC2933 registers with more interesting voltage settings
135  ltc2933_demo_board_demo_thresholds(ltc2933_i2c_address);
136  break;
137  case 6:
138  // margin the supplies to the 20% high side
139  // write to the LTC2637 DACs on the DC1633 board
140  dc1633_write_dac_voltage(dc1633_dac_address, 0, 3.96);
141  dc1633_write_dac_voltage(dc1633_dac_address, 1, 3.0);
142  dc1633_write_dac_voltage(dc1633_dac_address, 2, 2.16);
143  dc1633_write_dac_voltage(dc1633_dac_address, 3, 1.8);
144  dc1633_write_dac_voltage(dc1633_dac_address, 4, 1.44);
145  dc1633_write_dac_voltage(dc1633_dac_address, 5, 1.2);
146  break;
147  case 7:
148  //! write the LTC2933 CLEAR_ALERT
149  ltc2933_clear_alertb(ltc2933_i2c_address);
150  break;
151  case 8:
152  addresses = smbus->probe(0);
153  while (*addresses != 0)
154  {
155  Serial.print(F("ADDR 0x"));
156  Serial.println(*addresses++, HEX);
157  }
158  break;
159  default:
160  Serial.println(F("Incorrect Option"));
161  break;
162  }
163  print_prompt();
164  }
165 }

204 //! Return 1 if the LTC2933 is write-protected
205 // 0 otherwise
206 int ltc2933_is_write_protected(uint8_t ltc2933_i2c_address)
207 {
208  uint16_t res;
209 
210  res = smbus->readWord(ltc2933_i2c_address, LTC2933_STATUS_WORD);
211 
212  //b[0] is the write-protect bit
213  return ((res&0x0001) == 0x0001) ? 1 : 0;
214 }

364 //! program the DAC on the DC1633B demo board to a voltage
365 // refer to the LTC2637 datasheet
366 void dc1633_write_dac_voltage(uint8_t dac_address, int channel, float voltage)
367 {
368  uint8_t *data = (uint8_t *)malloc(3*sizeof(uint8_t));
369  uint8_t cmd, ch_addr;
370  uint16_t v_data;
371  float v;
372 
373  // pack the data bytes with the necessary bits
374  // channel numbers 0 - 7 correspond to letters A - H in the datasheet
375  if ((channel < 8) && (channel >= 0))
376  {
377  ch_addr = (uint8_t)channel;
378  }
379  else
380  {
381  //address all channels
382  ch_addr = 0x0F;
383  }
384  cmd = 0x30; // the write to and update command
385  data[0] = cmd | ch_addr;
386 
387  if ((voltage > 0) && (voltage < 4.096))
388  {
389  v = (voltage/4.096);
390  v_data = (uint16_t)(v*4096);
391  }
392  else
393  {
394  Serial.println(F("\nERROR: Voltage out of DAC range"));
395  }
396  data[1] = (uint8_t)(v_data >> 4); // most significant bits
397  data[2] = (uint8_t)(v_data << 4); // least significant bits
398 
399  //write the command and data to the DAC
400  Wire.beginTransmission(dac_address);
401  Wire.write(data[0]);
402  Wire.write(data[1]);
403  Wire.write(data[2]);
404  Wire.endTransmission(1);
405 
406  free(data);
407  return;
408 
409 }
