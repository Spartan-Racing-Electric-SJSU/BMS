13 @verbatim
14 
15 NOTES
16  Setup:
17  Set the terminal baud rate to 115200 and select the newline terminator. Equipment
18  required is a voltage source (preferably low-noise) and a precision voltmeter.
19  Ensure all jumpers on the demo board are installed in their default positions
20  from the factory. Refer to Demo Manual DC1796A.
21 
22  How to test:
23  The voltage source should be connected to input AIN+/AIN-. Ensure the
24  input is within its specified absolute input voltage range. (It is easiest
25  to tie the voltage source negative terminal to COM.) Ensure the voltage
26  source is set within the range of 0V to +4.096V.
27 
28 USER INPUT DATA FORMAT:
29  decimal : 1024
30  hex : 0x400
31  octal : 02000 (leading 0 "zero")
32  binary : B10000000000
33  float : 1024.0

164 //! Read channel
165 //! @return void
166 void menu_1_read_input()
167 {
168  uint8_t user_command;
169  uint32_t adc_code; // The LTC2370 code
170  int32_t display_code;
171  float adc_voltage; // The LTC2370 voltage
172 
173  // Read and display a selected channel
174  LTC2370_read(LTC2370_CS, &adc_code); //discard the first reading
175  delay(100);
176  LTC2370_read(LTC2370_CS, &adc_code);
177 
178  display_code = adc_code >> (32 - LTC2370_bits);
179  if (LTC2370_bits == 16)
180  display_code = display_code & 0xFFFF;
181  else
182  display_code = display_code & 0x3FFFF;
183 
184  Serial.print(F(" Received Code: b"));
185  Serial.println(display_code, BIN);
186 
187  // Convert the received code to voltage
188  adc_voltage = LTC2370_code_to_voltage(adc_code, LTC2370_vref);
189 
190  Serial.print(F(" Equivalent voltage: "));
191  Serial.print(adc_voltage, 4);
192  Serial.println(F("V"));
193 }
