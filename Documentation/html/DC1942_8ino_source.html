25

28

31

178 /*!*****************************************
179  \brief executes the user inputted command
180 
181  Menu Entry 1: Write Configuration \n
182  Writes the configuration register of the LTC6804. This command can be used to turn on the reference
183  and increase the speed of the ADC conversions.
184 
185  Menu Entry 2: Read Configuration \n
186  Reads the configuration register of the LTC6804, the read configuration can differ from the written configuration.
187  The GPIO pins will reflect the state of the pin
188 
189  Menu Entry 3: Start Cell voltage conversion \n
190  Starts a LTC6804 cell channel adc conversion.
191 
192  Menu Entry 4: Read cell voltages
193  Reads the LTC6804 cell voltage registers and prints the results to the serial port.
194 
195  Menu Entry 5: Start Auxiliary voltage conversion
196  Starts a LTC6804 GPIO channel adc conversion.
197 
198  Menu Entry 6: Read Auxiliary voltages
199  Reads the LTC6804 axiliary registers and prints the GPIO voltages to the serial port.
200 
201  Menu Entry 7: Start cell voltage measurement loop
202  The command will continuously measure the LTC6804 cell voltages and print the results to the serial port.
203  The loop can be exited by sending the MCU a 'm' character over the serial link.
204 
205 *******************************************/
206 void run_command(uint16_t cmd)
207 {
208  int8_t error = 0;
209 
210  char input = 0;
211  switch (cmd)
212  {
213 
214  case 1:
215  wakeup_sleep();
216  LTC6804_wrcfg(TOTAL_IC,tx_cfg);
217  print_config();
218  break;
219 
220  case 2:
221  wakeup_sleep();
222  error = LTC6804_rdcfg(TOTAL_IC,rx_cfg);
223  if (error == -1)
224  {
225  Serial.println("A PEC error was detected in the received data");
226  }
227  print_rxconfig();
228  break;
229 
230  case 3:
231  wakeup_sleep();
232  LTC6804_adcv();
233  delay(3);
234  Serial.println("cell conversion completed");
235  Serial.println();
236  break;
237 
238  case 4:
239  wakeup_sleep();
240  error = LTC6804_rdcv(0, TOTAL_IC,cell_codes); // Set to read back all cell voltage registers
241  if (error == -1)
242  {
243  Serial.println("A PEC error was detected in the received data");
244  }
245  print_cells();
246  break;
247 
248  case 5:
249  wakeup_sleep();
250  LTC6804_adax();
251  delay(3);
252  Serial.println("aux conversion completed");
253  Serial.println();
254  break;
255 
256  case 6:
257  wakeup_sleep();
258  error = LTC6804_rdaux(0,TOTAL_IC,aux_codes); // Set to read back all aux registers
259  if (error == -1)
260  {
261  Serial.println("A PEC error was detected in the received data");
262  }
263  print_aux();
264  break;
265 
266  case 7:
267  Serial.println("transmit 'm' to quit");
268  wakeup_sleep();
269  LTC6804_wrcfg(TOTAL_IC,tx_cfg);
270  while (input != 'm')
271  {
272  if (Serial.available() > 0)
273  {
274  input = read_char();
275  }
276  wakeup_idle();
277  LTC6804_adcv();
278  delay(10);
279  wakeup_idle();
280  error = LTC6804_rdcv(0, TOTAL_IC,cell_codes);
281  if (error == -1)
282  {
283  Serial.println("A PEC error was detected in the received data");
284  }
285  print_cells();
286  delay(500);
287  }
288  print_menu();
289  break;
290 
291  default:
292  Serial.println("Incorrect Option");
293  break;
294  }
295 }
