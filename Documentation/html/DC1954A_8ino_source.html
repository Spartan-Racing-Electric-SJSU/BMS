193 //! Menu 1: Load Default SPI Register Settings
194 //! This function identifies which of the 4 LTC6954 frequency versions are connected.
195 //! Based on the version connected, this function loads the register settings referenced
196 //! in the DC1954A demo manual's quick start section.
197 //! The register settings loaded are the same as the LTC6954_GUI 6954set files
198 //! LTC6954.6954set.
199 //! The setting loaded with this function assume the DC1954A's loop filter has
200 //! not been modified.
201 //! @return void
202 void menu_1_load_default_settings()
203 {
204 
205 // select which default register setting to load based on part number
206  if (demo_board.product_name[8]=='1') // if this is a LTC6954-1
207  set_LTC6954_ALLREGS(LTC6954_CS,0x00,0x80,0x04,0x80,0x04,0x80,0x04);
208  else if (demo_board.product_name[8]=='2') // if this is a LTC6954-2
209  set_LTC6954_ALLREGS(LTC6954_CS,0x00,0x80,0x04,0x80,0x04,0x80,0x04);
210  else if (demo_board.product_name[8]=='3') // if this is a LTC6954-3
211  set_LTC6954_ALLREGS(LTC6954_CS,0x00,0x80,0x04,0x80,0x04,0x80,0x04);
212  else if (demo_board.product_name[8]=='4') // if this is a LTC6954-4
213  set_LTC6954_ALLREGS(LTC6954_CS,0x00,0x80,0x04,0x80,0x04,0x80,0x04);
214  else
215  {
216  Serial.print("No default file for this board: ");
217  Serial.println(demo_board.product_name);
218  } // end if-then-else statement
219 
220  Serial.println(F("Registers Have Been Written"));
221 } // end menu_1_load_default_settings function

224 /* ------------------------------------------------------------------------- */
225 //! Menu 2: Reads and/or Writes the SPI register address
226 //! This function reads and displays all SPI register address settings in HEX format.
227 //! It then provides an option to modify(write to) individual registers one at time
228 //!
229 //! EXAMPLE:
230 //! - 0- ADDR00 = 0x00
231 //! - 1- ADDR01 = 0x08
232 //! - ....
233 //! - 6- ADDR06 = 0x04
234 //! - 7- ADDR07 = 0x21 (read only)
235 //! - -1 - Return to Main Menu
236 //! - Enter a command (0-6 to modify register, or '-1' to return to Main Menu):
237 //! @return void
238 void menu_2_RW_to_reg_addresss()
239 {
240  uint8_t i, regval, user_regval, num_regs;
241  uint16_t user_address; // User input command
242 
243  num_regs = get_LTC6954_REGSIZE();
244 // Read/Write loop, can exit loop by choosing '0'
245  user_address=99;
246  while (user_address != -1)
247  {
248  Serial.println(F("\n*****************************************************************"));
249  // Read All Registers and display results
250  for (i=0; i<num_regs; i++)
251  {
252  regval = LTC6954_read(LTC6954_CS,i);
253  Serial.print(i);
254  if (i<16)
255  Serial.print(F("- ADDR0"));
256  else
257  Serial.print(F("- ADDR"));
258  Serial.print(i, HEX);
259  Serial.print(F(" = 0x"));
260  if (regval<16) Serial.print(F("0"));
261  Serial.print(regval, HEX);
262  if (i==(num_regs-1)) Serial.print(" (read only) ");
263  Serial.println("");
264  } // end for loop
265  Serial.print("-1 - Return to Main Menu\n\n");
266  // User input: Select which register to modify, or return to main menu
267  Serial.print("Enter a command (0-6 to modify register, or '-1' to return to Main Menu): ");
268  user_address = read_int(); //! Reads the user command
269  Serial.println(user_address);
270 
271  // User input: enter new setting for selected register
272  if (user_address >=0 && user_address<(num_regs-1))
273  {
274  Serial.print("What value should ADDR");
275  Serial.print(user_address);
276  Serial.print(" be set to (ex: HEX format 0xff): ");
277  user_regval = read_int(); //! Reads the user command
278  Serial.println(user_regval);
279 
280  // writes new setting to part
281  LTC6954_write(LTC6954_CS, (uint8_t)user_address, user_regval);
282  } // end if statement
283  } // end while loop
284 } // end menu_2_RW_to_reg_addresss
