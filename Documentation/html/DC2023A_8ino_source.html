88 //! Repeats Linduino loop
89 //! @return void
90 void loop()
91 {
92  uint8_t user_command;
93  uint8_t res;
94  uint8_t model[7];
95  uint8_t *addresses = NULL;
96 
97  if (Serial.available()) //! Checks for user input
98  {
99  user_command = read_int(); //! Reads the user command
100  if (user_command != 'm')
101  Serial.println(user_command);
102 
103  switch (user_command) //! Prints the appropriate submenu
104  {
105  case 1:
106  menu_1_basic_commands(); // Print single-ended voltage menu
107  break;
108  case 2:
109  pmbus->enablePec(ltm2987_i2c_address_1);
110  pmbus->enablePec(ltm2987_i2c_address_2);
111  delete smbus;
112  delete pmbus;
113  smbus = new LT_SMBusPec();
114  pmbus = new LT_PMBus(smbus);
115  break;
116  case 3:
117  pmbus->disablePec(ltm2987_i2c_address_1);
118  pmbus->disablePec(ltm2987_i2c_address_2);
119  delete smbus;
120  delete pmbus;
121  smbus = new LT_SMBusNoPec();
122  pmbus = new LT_PMBus(smbus);
123  break;
124  case 4:
125  addresses = smbus->probe(0);
126  while (*addresses != 0)
127  {
128  Serial.print(F("ADDR 0x"));
129  Serial.println(*addresses++, HEX);
130  }
131  break;
132  case 5 :
133  pmbus->clearAllFaults(ltm2987_i2c_address_1);
134  pmbus->clearAllFaults(ltm2987_i2c_address_2);
135  break;
136  default:
137  Serial.println(F("Incorrect Option"));
138  break;
139  }
140  print_prompt();
141  }
