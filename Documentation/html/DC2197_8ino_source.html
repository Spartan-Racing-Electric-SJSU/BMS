204 //! Initializes Pin 9 and 10 for 16-bit PWM
205 void init_16_bit_PWM()
206 {
207  pinMode(DAC_A, OUTPUT); // Set Pin 9 as output
208  pinMode(DAC_B, OUTPUT); // Set Pin 10 as output
209 
210 
211  // Enable timer1 for PWM, Phase Correct mode for pin 9 and 10.
212  // Use IRC1 as top, no prescalar (~122Hz),
213  // Negative Edge Triggered
214  TCCR1A = _BV(COM1A1) | _BV(COM1B1) | _BV(WGM11);
215  TCCR1B = _BV(WGM13) | _BV(CS10);
216  TCCR1C = 0x0;
217  ICR1 = 0xFFFF; // Set Timer for 16-bit
218 
219  // The following code should be uncommented if Fast PWM Mode
220  // is desired
221  /*
222  // Enable timer1 for Fast PWM mode for pin 9 and 10.
223  // Use IRC1 as top, no prescalar (~244Hz),
224  // Negative Edge Triggered
225  TCCR1A = _BV(COM1A1) | _BV(COM1B1) | _BV(WGM11);
226  TCCR1B = _BV(WGM13)| _BV(WGM12) | _BV(CS10);
227  TCCR1C = 0x0;
228  ICR1 = 0xFFFF; // Set Timer for 16-bit
229  */
230 
231  // Set to 0% duty cycle
232  OCR1A = 0x0;
233  OCR1B = 0x0;
234 }

328 //! Calculate a LTC2645 duty code given the desired output voltage
329 //! @return PWM Code
330 uint16_t LTC2645_voltage_to_pwm_code(float dac_voltage, float fullScaleVoltage, int8_t pwmBitResolution)
331 {
332  float float_code, max_code;
333  max_code = pow(2, pwmBitResolution) - 1; // Calculate max code
334  max_code = (max_code > (floor(max_code) + 0.5)) ? ceil(max_code) : floor(max_code); // Round max code
335  float_code = (dac_voltage / fullScaleVoltage) * max_code; // Calculate the DAC code
336  float_code = (float_code > (floor(float_code) + 0.5)) ? ceil(float_code) : floor(float_code); // Round
337  if (float_code < 0) // If DAC code < 0, Then DAC code = 0
338  float_code = 0;
339  if (float_code > max_code) // If DAC code > max code, then DAC code = max code
340  float_code = max_code;
341  return ((uint16_t)float_code); // Cast DAC code as uint16_t
342 }
