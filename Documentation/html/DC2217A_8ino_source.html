186 //! Scan I2C Bus for addresses
187 void menu_1_scan_addresses()
188 {
189 
190  int maxAddr = 127; //! Max Number for possible 7-bit address values
191  int addrCount = 0; //! Initialization of address count.
192 
193  for (int addr = 0x00; addr <= maxAddr; addr++) //! Scan possible address values till max address number
194  {
195  int8_t ret= 0 ; //! Initialize return bit variable
196  if (i2c_start()!=0) //! I2C START
197  {
198  Serial.print(F("Bus is Busy. Aborting...")); //! Stop and abort scanning if bus is busy.
199  break;
200  }
201 
202  ret |= i2c_write((addr<<1)|I2C_WRITE_BIT); //! Write the I2C 7 bit address with W bit
203  i2c_stop(); //! I2C STOP
204  if (ret==0) //! If acknolwege recieved on bus, print address
205  {
206  addrCount++; //! Increment address count
207  Serial.print(F("\n Address Found - #"));
208  Serial.print(addrCount); //! Print address count
209  Serial.print(F(": "));
210  Serial.print(addr, HEX); //! Print address
211 
212  switch (addr) //! Switch statement to identify factory set possible address values. Note: If external address has the same address as any of the jumper settings
213  //! it might register as one of the switch statements
214  {
215  case EEPROM_ADDRESS:
216  Serial.print(F(" - Onboard EEPROM Address."));
217  break;
218  case (LTC2631_GLOBAL_ADDRESS):
219  Serial.print(F(" - Through Global Address of LTC2631"));
220  break;
221  case (LTC2631_GLOBAL_ADDRESS ^ LTC4316_OFFSET):
222  Serial.print(F(" - Remapped Global Address of LTC2631"));
223  break;
224  case (LTC2631_JP_ADDRESS_1):
225  Serial.print(F(" - Through Jumper Setting of Onboard LTC2631"));
226  break;
227  case (LTC2631_JP_ADDRESS_2):
228  Serial.print(F(" - Through Jumper Setting of Onboard LTC2631"));
229  break;
230  case (LTC2631_JP_ADDRESS_3):
231  Serial.print(F(" - Through Jumper Setting of Onboard LTC2631"));
232  break;
233  case (LTC2631_JP_ADDRESS_1 ^ LTC4316_OFFSET):
234  Serial.print(F(" - Remapped Address of Onboard LTC2631 Based On Jumper Setting 12h"));
235  break;
236  case (LTC2631_JP_ADDRESS_2 ^ LTC4316_OFFSET):
237  Serial.print(F(" - Remapped Address of Onboard LTC2631 Based On Jumper Setting 11h"));
238  break;
239  case (LTC2631_JP_ADDRESS_3 ^ LTC4316_OFFSET):
240  Serial.print(F(" - Remapped Address of Onboard LTC2631 Based On Jumper Setting 11h"));
241  break;
242  default:
243  Serial.print(F(" - Unknown Address Found. Possible external device connected and/or Non-Factory address selected."));
244  break;
245  }
246  Serial.print("\n\n");
247  }
248  }
249 
250  if (addrCount == 0)
251  Serial.println(F("Nothing Found on Bus"));
252 }

254 //! DAC Control Menu
255 uint8_t menu_2_DAC()
256 {
257 //! @return Returns the state of the acknowledge bit after the I2C address write. 0=acknowledge, 1=no acknowledge.
258 
259  int ack = 0; //! I2C acknowledge indicator
260  static uint8_t user_command; //! The user input command
261  int maxAddr = 127; //! Max Number for possible 7-bit address values
262  int addrCount = 0; //! Initialization of address count.
263  uint8_t LTC2631_CMD_WRITE_UPDATE = 0x30; //! LTC2631 Write and Update Command Byte
264  uint16_t LTC2631_FULLSCALE_OUTPUT = 0xFFC0; //! LTC2631 10 bit fullscale input value
265  uint8_t through_address = 0; //! Through address value
266  uint8_t remapped_address = 0; //! Remapped address value
267 
268  for (int addr = 0x00; addr <= maxAddr; addr++)
269  {
270  int8_t ret= 0 ;
271 
272  if (i2c_start()!=0) //I2C START
273  {
274  Serial.print(F("Bus is Busy. Aborting...")); //Stop and abort scanning if bus is busy.
275  break;
276  }
277 
278  ret |= i2c_write((addr<<1)|I2C_WRITE_BIT); //Write the I2C 7 bit address with W bit
279  i2c_stop(); //I2C STOP
280  if (ret==0) // Returns 1 if failed
281  {
282  switch (addr) //Switch statement to assign addresses based on default demoboard configuration. External addresses are not supported and could cause conflict.
283  {
284  case LTC2631_JP_ADDRESS_1:
285  case LTC2631_JP_ADDRESS_2:
286  case LTC2631_JP_ADDRESS_3:
287  through_address = addr;
288  break;
289  case LTC2631_JP_ADDRESS_1 ^ LTC4316_OFFSET:
290  case LTC2631_JP_ADDRESS_2 ^ LTC4316_OFFSET:
291  case LTC2631_JP_ADDRESS_3 ^ LTC4316_OFFSET:
292  remapped_address = addr;
293  break;
294  case EEPROM_ADDRESS:
295  case LTC2631_GLOBAL_ADDRESS:
296  case LTC2631_GLOBAL_ADDRESS ^ LTC4316_OFFSET:
297  break;
298  default:
299  Serial.print(F(" - Unknown Address Found. Possible external device connected and/or Non-Factory address selected."));
300  break;
301  }
302  Serial.print("\n\n");
303  }
304  }
305 
306  do
307  {
308  Serial.print(F("*************************\n\n"));
309  Serial.print(F("Turn on DACs\n"));
310  Serial.print(F("1-Turn on Through DAC\n"));
311  Serial.print(F("2-Turn on Remapped DAC\n"));
312  Serial.print(F("3-Turn off Through DAC\n"));
313  Serial.print(F("4-Turn off Through DAC\n"));
314  Serial.print(F("m-Main Menu\n\n"));
315  Serial.print(F("Enter a command: "));
316 
317  user_command = read_int(); //! Read user input command
318  if (user_command != 'm')
319  Serial.println(user_command);
320  if (user_command == 'm')
321  Serial.println("m");
322  Serial.println();
323  ack = 0;
324  switch (user_command) //! Switch statement for DAC menu
325  {
326  case 1:
327  ack |= i2c_write_word_data(through_address,LTC2631_CMD_WRITE_UPDATE, LTC2631_FULLSCALE_OUTPUT); //! I2C write command to turn Through DAC on
328  break;
329  case 2:
330  ack |= i2c_write_word_data(remapped_address,LTC2631_CMD_WRITE_UPDATE, LTC2631_FULLSCALE_OUTPUT); //! I2C write command to turn Remapped DAC on
331  break;
332  case 3:
333  ack |= i2c_write_word_data(through_address, LTC2631_CMD_WRITE_UPDATE, 0x0000); //! I2c write command to turn through DAC off
334  break;
335  case 4:
336  ack |= i2c_write_word_data(remapped_address, LTC2631_CMD_WRITE_UPDATE, 0x0000); //! I2C write command to turn remapped DAC off
337  default:
338  if (user_command != 'm')
339  Serial.println("Incorrect Option");
340  }
341  }
342  while (!(user_command == 'm' || (ack)));
343  if (ack != 0)
344  Serial.println(ack_error);
345  return ack;
346 }
