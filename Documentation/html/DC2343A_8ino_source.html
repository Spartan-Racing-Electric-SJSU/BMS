1 /*!
2 Linear Technology DC934 Demonstration Board.
3 LTC3335: Nanopower Buck-Boost DC/DC with Integrated Coulomb Counter
4 
5 @verbatim
6 
7 This sketch demonstrates the LTC3335 using the DC2343A demo board. The data read from
8 the data board is displayed on a serial terminal using 115200 baud, and newline terminations
9 for data input.
10 
11 Code will build for configuration specified in LTC3335_Config.h.
12 
13  1. The selected output voltage is specified by LTC3335_OUTPUT_VOLTAGE.
14  2. The LTC3335 prescaler will be chosen appropriately for the battery capacity specified
15  by LTC3335_CAPACITY.
16  3. The LTC3335 alarm will be chosen appropriately for the value specified by
17  LTC3335_ALARM_CAPACITY.
18  4. The coulomb count and battery current measurement will be scaled accordingly to the
19  LTC3335 prescaler selected by LTC3335_CAPACITY and the LTC3335_IPEAK_CONFIGURATION,
20  setting.
21  5. Set LTC3335_USE_CURRENT_MEASUREMENT if you wish to use the LTC3335 battery current measurement.
22  6. Set LTC3335_USE_SOFTWARE_CORRECTION if you wish to use correction of the LTC3335 coulomb
23  count and battery current measurement. A value of LTC3335_VBAT_TYP must be specified
24  for this sketch as it can not measure the actual battery voltage, and must assume a constant
25  value. Applications reusing this code would improve performance if the actual battery voltage
26  was measured and used for the correction functions. See the DC2343A schematic for a circuit
27  which measures the battery voltage with >10nA of average current drawn from the battery.
28 
29 @endverbatim
30

32 the DC2343-ASSY1 PCB from JP5:
33 \image html "../../Part Number/3000/3300/3335/DC2343A/Assy_PCB_Removed.PNG"
34 
35 A cable must then be constructed to connect the Linduino to the DC2343A as shown:
36 \image html "../../Part Number/3000/3300/3335/DC2343A/Linduino_Cable.png"

139 //! Initialize Linduino
140 void setup()
141 {
142 
143  Serial.begin(115200); //! Initialize the serial port to the PC.
144  while (!Serial); //! Wait for serial port to be opened in the case of Leonardo USB Serial.
145 
146  //! Wait for LTC3335 to initialize successfully.
147  while ((LTC3335_Init() != 0))
148  {
149  print_error(); //! Show reasons why LTC3335 might not have initialized.
150  delay(5000);
151  }
152 
153  //! Display configuration set in LTC3335_Config.h.
154  print_config();
155  print_data_header();
156 
157  //! Set up timing for tasks in loop.
158  data_point_counter = 0;
159  refresh_time_last = millis();
160  current_refresh_time_last = refresh_time_last;
161 
162  return;
163 }

165 //! Repeats Linduino Loop
166 void loop()
167 {
168  uint16_t time_new = millis(); // ! Get new time
169 
170  //! Check if it's time to poll LTC3335 and update the terminal.
171  if (time_new - refresh_time_last > REFRESH_INTERVAL)
172  {
173  //! Read the discharged capacity.
174 #if LTC3335_USE_SOFTWARE_CORRECTION == false
175  LTC3335_Get_Discharged_Capacity(&discharged_capacity);
176 #else
177  LTC3335_Get_Discharged_Capacity(&discharged_capacity, vbat);
178 #endif // LTC3335_USE_SOFTWARE_CORRECTION == false
179 
180  //! If current measurement is being used, read it. If software correction is used, pass the battery voltage
181  //! to the function so that the proper correction factor will be used.
182 #if LTC3335_USE_CURRENT_MEASUREMENT == true
183 #if LTC3335_USE_SOFTWARE_CORRECTION == false
184  LTC3335_Get_Counter_Test_Current(&current);
185 #else
186  LTC3335_Get_Counter_Test_Current(&current, vbat);
187 #endif // LTC3335_USE_SOFTWARE_CORRECTION == false
188 #endif // #if LTC3335_USE_CURRENT_MEASUREMENT == true
189 
190  //! Only need to read alarms unless nIRQ is active (although not when nIRQ is used for current measurement).
191  if ((LTC3335_NIRQ_READ() == 0) || (LTC3335_USE_CURRENT_MEASUREMENT == true))
192  {
193  LTC3335_Get_Alarms(&alarms);
194  }
195 
196  //! Update terminal.
197  print_data();
198 
199  // Schedule next loop.
200  refresh_time_last += REFRESH_INTERVAL;
201  data_point_counter++;
202  }
203 
204 #if LTC3335_USE_CURRENT_MEASUREMENT == true
205  //! If using the current measurement, clear it periodically. The current measurement will be averaged over time until it is reset.
206  if (time_new - current_refresh_time_last > CURRENT_REFRESH_INTERVAL)
207  {
208  LTC3335_Reset_Counter_Test_Current();
209  current_refresh_time_last += CURRENT_REFRESH_INTERVAL;
210  }
211 
212  //! Update the Counter Test current in the background. Must be called more often than LTC3335_MIN_CURRENT_TASK_RATE.
213  LTC3335_Counter_Test_Current_Task();
214 #endif // #if LTC3335_USE_CURRENT_MEASUREMENT == true
215 
216  //! If any character is entered, send command to clear interrupt.
217  if (Serial.available())
218  {
219  LTC3335_Clear_Int(&alarms);
220  Serial.read(); // Clear out character that caused Clear Int command to be sent.
221  }
222 
223 }

377 {

380 #if LTC3335_USE_CURRENT_MEASUREMENT == true

384  printf(F("%31b,"), alarms.coulomb_counter_overflow);
