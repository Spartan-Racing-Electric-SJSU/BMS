79 //! Initialize Linduino
80 void setup()
81 {
82  //Port initialization
83  DDRD = DDRD | B00111100; // Sets data direction of Digital Pins 0-7 (Logical OR to not interfere with Rx and Tx)
84  PORTD = B00000000; // Initializes Digital Pins 0-7 to LOW
85  DDRB = B00011111; // Sets data direction of Digital Pins 8-13
86  PORTB = B00000001; // Initializes Digital Pin 8 HIGH and 9-13 to LOW, Pin 8 is I2C Linduino select bit
87  i2c_enable(); // Initializes Linduino I2C port.
88  Serial.begin(115200); // Initialize the serial port to the PC
89  print_title();
90  print_prompt();
91 }

93 //! Repeats Linduino loop
94 void loop()
95 {
96  int8_t ack=0;
97  uint8_t user_command;
98  uint8_t data;
99  uint8_t user_register;
100 
101  if (Serial.available())
102  {
103  user_command = read_int(); //! Reads the user command
104  Serial.println(user_command); // Print user command
105 
106  //! Reads or writes to a LTM9100 and prints result.
107  switch (user_command)
108  {
109  case 1:
110  // Read every register and print it's data.
111  ack |= LTM9100_print_all_registers(i2c_address);
112  break;
113  case 2:
114  // Read a single register and print it's data.
115  Serial.print(F("\nAddress (in hex with '0x' prefix) of register to read: "));
116  user_register = read_int();
117  Serial.print("0x");
118  Serial.println(user_register, HEX);
119  if (!valid_register(user_register, reg_read_list, sizeof(reg_read_list)))
120  {
121  Serial.println(F(" Invalid input."));
122  break;
123  }
124  ack |= LTM9100_register_read(i2c_address, user_register, &data);
125  if (!ack)
126  {
127  Serial.print("Register data: 0x");
128  Serial.println(data, HEX);
129  }
130  break;
131  case 3:
132  // Write a byte to a chosen register.
133  Serial.print(F("\nAddress (in hex with '0x' prefix) of register to write: "));
134  user_register = read_int();
135  Serial.print("0x");
136  Serial.println(user_register, HEX);
137  if (!valid_register(user_register, reg_write_list, sizeof(reg_write_list)))
138  {
139  Serial.println(F(" Invalid input."));
140  break;
141  }
142  Serial.print(F("Data (in hex with '0x' prefix) to write: "));
143  data = read_int();
144  Serial.println(data, HEX);
145  ack |= LTM9100_register_write(i2c_address, user_register, data);
146  if (!ack)
147  {
148  Serial.print("0x");
149  Serial.print(data, HEX);
150  Serial.print(" written to register 0x");
151  Serial.println(user_register, HEX);
152  }
153  break;
154  case 4:
155  // Set a single bit within a chosen register.
156  Serial.print(F("\nAddress (in hex with '0x' prefix) of register: "));
157  user_register = read_int();
158  Serial.print("0x");
159  Serial.println(user_register, HEX);
160  if (!valid_register(user_register, reg_write_list, sizeof(reg_write_list)))
161  {
162  Serial.println(F(" Invalid input."));
163  break;
164  }
165  Serial.print(F("Bit position (0-7) to set: "));
166  data = read_int();
167  if (data < 0 || data > 7)
168  {
169  Serial.println(F(" Invalid input."));
170  break;
171  }
172  Serial.println(data, DEC);
173  ack |= LTM9100_bit_set(i2c_address, user_register, data);
174  if (!ack)
175  {
176  Serial.print("Bit set. Register data is now 0x");
177  ack |= LTM9100_register_read(i2c_address, user_register, &data);
178  Serial.println(data, HEX);
179  }
180  break;
181  case 5:
182  // Clear a single bit within a chosen register.
183  Serial.print(F("\nAddress (in hex with '0x' prefix) of register: "));
184  user_register = read_int();
185  Serial.print("0x");
186  Serial.println(user_register, HEX);
187  if (!valid_register(user_register, reg_write_list, sizeof(reg_write_list)))
188  {
189  Serial.println(F(" Invalid input."));
190  break;
191  }
192  Serial.print(F("Bit position (0-7) to clear: "));
193  data = read_int();
194  if (data < 0 || data > 7)
195  {
196  Serial.println(F("Invalid input."));
197  break;
198  }
199  Serial.println(data, DEC);
200  ack |= LTM9100_bit_clear(i2c_address, user_register, data);
201  if (!ack)
202  {
203  Serial.print("Bit cleared. Register data is now 0x");
204  ack |= LTM9100_register_read(i2c_address, user_register, &data);
205  Serial.println(data, HEX);
206  }
207  break;
208  case 6:
209  // Write a byte to a chosen register via the Mass Write I2C address.
210  Serial.print(F("\nAddress (in hex with '0x' prefix) of register to write: "));
211  user_register = read_int();
212  Serial.print("0x");
213  Serial.println(user_register, HEX);
214  if (!valid_register(user_register, reg_write_list, sizeof(reg_write_list)))
215  {
216  Serial.println(F(" Invalid input."));
217  break;
218  }
219  Serial.print(F("Data (in hex with '0x' prefix) to write: "));
220  data = read_int();
221  Serial.println(data, HEX);
222  ack |= LTM9100_register_write(0x1F, user_register, data);
223  if (!ack)
224  {
225  Serial.print("0x");
226  Serial.print(data, HEX);
227  Serial.print(" written to register 0x");
228  Serial.println(user_register, HEX);
229  }
230  break;
231  case 7:
232  ack |= menu_read_status(); // Print status menu
233  break;
234  case 8:
235  ack |= menu_read_write_faults_alerts(false); // Print Faults menu
236  break;
237  case 9:
238  ack |= menu_read_write_faults_alerts(true); // Print Alert setting menu
239  break;
240  case 10:
241  ack |= menu_read_write_controls(); // Print Controls menu
242  break;
243  case 11:
244  ack |= menu_read_write_adc_params(); // Print ADC menu
245  break;
246  case 12:
247  ack |= LTM9100_continuous_read_all_registers(i2c_address);
248  break;
249  }
250  if (ack != 0)
251  {
252  Serial.print(F("Error: No Acknowledge. \n"));
253  }
254  print_prompt();
255  }
256 }

300 int8_t menu_read_status()
301 {
302  int8_t ack=0;
303  uint8_t user_command;
304  uint8_t data;
305  uint8_t user_register = LTM_9100_STATUS_A_REG;
306  uint8_t user_bit;
307  do
308  {
309  //! Displays the Read Status menu
310  Serial.print(F("\nRead Status\n\n"));
311  Serial.print(F(" 1-Gate Status\n"));
312  Serial.print(F(" 2-PG Input Pin Status\n"));
313  Serial.print(F(" 3-FET Status\n"));
314  Serial.print(F(" 4-Overcurrent Condition\n"));
315  Serial.print(F(" 5-Undervoltage Condition\n"));
316  Serial.print(F(" 6-Overvoltage Condition\n"));
317  Serial.print(F(" m-Main Menu\n"));
318  Serial.print(F("\nEnter a command: "));
319 
320  user_command = read_int(); //! Reads the user command
321  if (user_command == 'm') // Print m if it is entered
322  Serial.print(F("m\n"));
323  else
324  Serial.println(user_command); // Print user command
325 
326  //! Reads LTM9100 and prints result.
327  switch (user_command)
328  {
329  case 1:
330  ack |= LTM9100_bit_read(i2c_address, user_register, LTM_9100_STATUS_GATE, &data);
331  Serial.print(F("Gate State: "));
332  (data)?Serial.println(F("On")):Serial.println(F("Off"));
333  break;
334  case 2:
335  ack |= LTM9100_bit_read(i2c_address, user_register, LTM_9100_STATUS_PGI, &data);
336  Serial.print(F("PG Input State: "));
337  (data)?Serial.println(F("High")):Serial.println(F("Low"));
338  break;
339  case 3:
340  ack |= LTM9100_bit_read(i2c_address, user_register, LTM_9100_STATUS_FET, &data);
341  Serial.print(F("FET State: "));
342  (data)?Serial.println(F("Short")):Serial.println(F("No Short"));
343  break;
344  case 4:
345  ack |= LTM9100_bit_read(i2c_address, user_register, LTM_9100_STATUS_OC, &data);
346  Serial.print(F("Overcurrent: "));
347  (data)?Serial.println(F("Yes")):Serial.println(F("No"));
348  break;
349  case 5:
350  ack |= LTM9100_bit_read(i2c_address, user_register, LTM_9100_STATUS_UV, &data);
351  Serial.print(F("Undervoltage: "));
352  (data)?Serial.println(F("Yes")):Serial.println(F("No"));
353  break;
354  case 6:
355  ack |= LTM9100_bit_read(i2c_address, user_register, LTM_9100_STATUS_OV, &data);
356  Serial.print(F("Overvoltage: "));
357  (data)?Serial.println(F("Yes")):Serial.println(F("No"));
358  break;
359  default:
360  if (user_command != 'm')
361  Serial.println("Invalid Selection");
362  break;
363  }
364 
365  }
366  while ((user_command != 'm') && (ack != 1));
367  return(ack);
368 }

537  Serial.print(F("Gate State: "));
538  (data)?Serial.println(F("Enabled")):Serial.println(F("Disabled"));
539  break;
540  break;
541  case 4:

543  Serial.print(F("Overcurrent: "));

545  break;

555  break;

559  if (user_bit > 3)

564  data &= ~LTM_9100_CTRL_PGIO_CFG_MASK;
565  data |= user_bit<<LTM_9100_CTRL_PGIO_CFG;
566  LTM9100_register_write(i2c_address, user_register, data);
567  }
568  break;
569  case 8:
570  ack |= ask_clear_set_bit(i2c_address, user_register, LTM_9100_CTRL_ADC_WRITE );
571  break;
572  case 9:
573  ack |= ask_clear_set_bit(i2c_address, user_register, LTM_9100_CTRL_GATE_CTRL);
