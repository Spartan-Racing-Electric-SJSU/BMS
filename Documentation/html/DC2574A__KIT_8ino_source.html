303 {
304  // init serial port
305  Serial.begin(115200);
306  // connect to LCD
307  lcd = new LCDKeypad(10);
308 #ifdef LTC2947_SPI_MODE_ENABLE

312  quikeval_SPI_connect();
313 #else
314  lcd->searchForLCD();
315  // set LTC2947 library to I2C mode
316  LTC2947_InitI2C(LTC2947_I2C_ADDR_LL);
317  // Initializes Linduino I2C port.
318  quikeval_I2C_init();

320  quikeval_I2C_connect();
321 #endif
322  // make a reset of LTC2947
323  LTC2947_WR_BYTE(LTC2947_REG_OPCTL, LTC2947_BM_OPCTL_RST);
324  // give device some time to restart
325  DelayAndCheckSerialData(120);
326  // write PRE and DIV setting
327  LTC2947_WR_BYTE(LTC2947_REG_TBCTL,
328  LTC2947_PRE << BM2BITPOS(LTC2947_BM_TBCTL_PRE_0) |
329  LTC2947_DIV << BM2BITPOS(LTC2947_BM_TBCTL_DIV_0));
330  // enable continuous measurement on LTC2947
331  LTC2947_WR_BYTE(LTC2947_REG_OPCTL, LTC2947_BM_OPCTL_CONT);
332  CLEAR_SCREEN();
333  lcd->EnableSerialOutput = true;
334  PRINTSCREEN(F("Hello LTC2947"));
335  NEWLINE();
336  PRINTSCREEN(F("30A PowerMonitor"));
337  lcd->EnableSerialOutput = false;
338  Serial.println();
339  if (!lcd->foundLCD)
340  Serial.print(F("No "));
341  Serial.println(F("LCD found! Enter l or m for options."));
342  Serial.flush();
343  DelayAndProcessInputs(SPLASH_SCREEN_DELAY); //DelayAndCheckSerialData(SPLASH_SCREEN_DELAY);
344 }

346 /** \fn void update()
347 \brief update LTC2947 measurement results */
348 void update()
349 {
350  // general buffer to hold data from 2947
351  byte data[8];
352  // stores the last value or PGCTL
353  boolean lastPageSelect;
354  // 1st check if we will actually read values
355  // from the device or just show labels of
356  // measured values on the LCD
357  if (!flagClear && userGroupSelect)
358  {
359  // labels might only be shown if no clear is requested and measurement values will be shown
360  boolean showHeader = false;
361  if (userGroupCycleTimer == 0)
362  {
363  if (userGroup2Cycle)
364  userSelGroup2 = (userSelGroup2 + 1) % USER_SEL_GRP2_CYCLE_COUNT;
365  userGroupCycleTimer = USERGROUPCYCLETIMERRELOAD;
366  showHeader = true;
367  }
368  else
369  userGroupCycleTimer--;
370  if (showHeader)
371  {
372  CLEAR_SCREEN();
373  switch (userSelGroup2)
374  {
375  case 0:
376  PRINTSCREEN(MEAS_LABEL00);
377  NEWLINE();
378  PRINTSCREEN(MEAS_LABEL01);
379  DelayAndProcessInputs(LABEL_DELAY);
380  return;
381  case 1:
382  PRINTSCREEN(MEAS_LABEL10);
383  NEWLINE();
384  PRINTSCREEN(MEAS_LABEL11);
385  DelayAndProcessInputs(LABEL_DELAY);
386  return;
387  case 2:
388  PRINTSCREEN(MEAS_LABEL20);
389  NEWLINE();
390  PRINTSCREEN(MEAS_LABEL21);
391  DelayAndProcessInputs(LABEL_DELAY);
392  return;
393  case 3:
394  PRINTSCREEN(MEAS_LABEL30);
395  NEWLINE();
396  PRINTSCREEN(MEAS_LABEL31);
397  DelayAndProcessInputs(LABEL_DELAY);
398  return;
399  case 4:
400  PRINTSCREEN(MEAS_LABEL40);
401  NEWLINE();
402  PRINTSCREEN(MEAS_LABEL41);
403  DelayAndProcessInputs(LABEL_DELAY);
404  return;
405  default:
406  PRINTSCREEN(MEAS_LABEL50);
407  NEWLINE();
408  PRINTSCREEN(MEAS_LABEL51);
409  DelayAndProcessInputs(LABEL_DELAY);
410  return;
411  }
412  }
413  }
414  // check if we are in cont mode
415  LTC2947_RD_BYTE(LTC2947_REG_OPCTL, data);
416  if (bitMaskClrChk(data[0], LTC2947_BM_OPCTL_CONT))
417  {
418  // enable continuous measurement on LTC2947
419  LTC2947_WR_BYTE(LTC2947_REG_OPCTL, LTC2947_BM_OPCTL_CONT);
420  return; // we were in IDLE, so nothing todo here
421  }
422  // store page select
423  // this is needed to restore last selected page to avoid page missmatch
424  // if the LTC2947 GUI is also communicating with Linduino
425  lastPageSelect = LTC2947_GetCurrentPageSelect();
426  if (lastPageSelect)
427  LTC2947_SetPageSelect(false); // go to page 0 where the measuremnt results are stored
428 #ifdef LTC2947_AUTO_READ_STATUS
429  // reading the STATUS register would also affect the GUI as status register bits are CoR!!!
430  // for this reason the following code is not enabled!
431  // still KEYPAD_RIGHT allows read of STATUS on demand and print result on LCD
432  // read all status and alert (if alerts are used it is important to read all in a single burst of 8 bytes!)
433  LTC2947_RD_BYTES(LTC2947_REG_STATUS, 8, data);
434  // if UPDATE bit is set we have a new conversion result
435  if (bitMaskSetChk(data[0], LTC2947_BM_STATUS_UPDATE))
436  return; // UPDATE bit not set, so no new data
437 #endif
438  if (flagClear)
439  {
440  // clear everything
441  flagClear = false;
442  NEWLINE();
443  PRINTSCREEN(F("****CLEARED*****"));
444  LTC2947_WR_BYTE(LTC2947_REG_OPCTL, 0); // leave continuous mode and go to IDLE
445  DelayAndCheckSerialData(100);
446  LTC2947_WR_BYTE(LTC2947_REG_OPCTL, LTC2947_BM_OPCTL_CLR); // clear tracking registers and accumulators
447  DelayAndCheckSerialData(50);
448  LTC2947_WR_BYTE(LTC2947_REG_OPCTL, LTC2947_BM_OPCTL_CONT); // enable continuous
449  }
450  else
451  {
452  if (userGroupSelect)
453  {
454  switch (userSelGroup2)
455  {
456  case 0://PRINTSCREEN(F("I V P TEMP"));
457  {
458  // show RAW quantities like I,P,V,TEMP
459  float I, P, V, TEMP, VCC;
460  LTC2947_Read_I_P_V_TEMP_VCC(&I, &P, &V, &TEMP, &VCC);
461  CheckSerialData();// we execute CheckSerialData() from time to time here to make sure the uart rx buffer never overflows
462  // print current and limit decimal points for values -30.00 ... -10.00 (but -9.999 ... 30.000)
463  CLEAR_SCREEN();
464  PRINTSCREEN2(I, I > -10.0 ? 3 : 2);
465  PRINTSCREEN('A');
466  SCREENSETCURSOR(8, 0);
467  PRINTSCREEN2(V, 3);
468  PRINTSCREEN('V');
469  CheckSerialData();
470  NEWLINE();
471  PRINTSCREEN2(P, 3); // max characters: -450.000W = 9: 0..8
472  PRINTSCREEN('W');
473  boolean limitTempCharacters = P <= -100.0;
474  SCREENSETCURSOR(10, 1);
475  PRINTSCREEN2(TEMP, limitTempCharacters ? 0 : 1);
476  WRITE_DEGREE_C();
477  }
478  break;
479  case 1://PRINTSCREEN(F("C1 VDVCC E1 TB1"));
480  case 2://PRINTSCREEN(F("C2 VDVCC E2 TB2"));
481  {
482  double C, E, TB;
483  boolean signC = true, signE = true;
484 #ifndef LTC2947_READ_SIGNED_ACCUS
485  LTC2947_Read_Abs_C_E_TB(userSelGroup2 == 1, &C, &signC, &E, &signE, &TB); //Read either C1/2,E1/2,TB1/2
486 #else
487  // instead of using LTC2947_Read_Abs_C_E_TB we could also call LTC2947_Read_C_E_TB to
488  // get signed charge and energy values. But as the following code makes usage of
489  // LTC2947_Read_Abs_C_E_TB's separation in absolute values and sign, we won't use
490  // LTC2947_Read_C_E_TB here!
491  LTC2947_Read_C_E_TB(userSelGroup2 == 1, &C, &E, &TB); //Read either C1/2,E1/2,TB1/2
492  if (C < 0) C = -C;
493  else signC = false;
494  if (E < 0) E = -E;
495  else signE = false;
496 #endif
497  C = C * LTC2947_LSB_FACTOR_MILLI_HOURS; //! convert As to mAh
498  E = E * LTC2947_LSB_FACTOR_MILLI_HOURS; //! convert Ws to mWh
499  CLEAR_SCREEN();
500  // convert to reasonable units
501  if (C < 1e3)
502  {
503  // restore sign and print value in mAh
504  PRINTSCREEN2(signC ? -C : C, 2); // e.g. -999.99mAh: 10 chars
505  PRINTSCREEN(F("mAh"));
506  }
507  else if (C < 1e6)
508  {
509  // restore sign and print value in Ah
510  C = C * 1e-3;
511  PRINTSCREEN2(signC ? -C : C, 3); // e.g. -999.999Ah: 10 chars
512  PRINTSCREEN(F("Ah"));
513  }
514  else if (C < 1e9)
515  {
516  // restore sign and print value in kAh
517  C = C * 1e-6;
518  PRINTSCREEN2(signC ? -C : C, 2); // e.g. -999.99kAh: 10 chars
519  PRINTSCREEN(F("kAh"));
520  }
521  CheckSerialData();
522  SCREENSETCURSOR(11, 0);
523  LTC2947_RD_BYTES(LTC2947_VAL_VDVCC, 2, data); // read VDVCC[15:0]
524  double vcc = LTC2947_BytesToDouble(data, 2, true, LTC2947_LSB_VDVCC * 1e-3); // convert to real volts
525  PRINTSCREEN2(vcc, 2);
526  PRINTSCREEN('V'); // e.g. 4.65V: 5 chars
527  CheckSerialData();
528  NEWLINE();
529  // convert to reasonable units
530  if (E < 1e3)
531  {
532  // restore sign and print value in mWh
533  PRINTSCREEN2(signE ? -E : E, signE && (E >= 1e2) ? 1 : 2);
534  PRINTSCREEN(F("mWh"));
535  }
536  else if (E < 1e6)
537  {
538  // restore sign and print value in Wh
539  E = E * 1e-3;
540  PRINTSCREEN2(signE ? -E : E, signE && (E >= 1e5) ? 2 : 3);
541  PRINTSCREEN(F("Wh"));
542  }
543  else if (E < 1e9)
544  {
545  // restore sign and print value in kWh
546  E = E * 1e-6;
547  PRINTSCREEN2(signE ? -E : E, signE && (E >= 1e8) ? 1 : 2);
548  PRINTSCREEN(F("kWh"));
549  }
550  CheckSerialData();
551  SCREENSETCURSOR(10, 1);
552  // convert to reasonable units
553  if (TB < 60.0) // less than one minute
554  {
555  //print time in seconds
556  PRINTSCREEN2(TB, 2);
557  PRINTSCREEN('s');//e.g. 59.9s
558  }
559  else if (TB < 60.0 * 60.0) // less than one hour
560  {
561  //print time in minutes and seconds
562  uint16_t intSec = TB;
563  PRINTSCREEN(intSec / 60);
564  PRINTSCREEN('m');
565  PRINTSCREEN(intSec % 60);
566  PRINTSCREEN('s');//e.g. 59m59s
567  }
568  else if (TB < 60.0 * 60.0 * 24.0)
569  {
570  //print time in hours and minutes
