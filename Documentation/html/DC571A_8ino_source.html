1 /*!
2 Linear Technology DC571A Demonstration Board.
3 LTC2418: 24-bit No Latency Delta Sigma 16-channel ADC
4 
5 @verbatim
6 
7 NOTES
8  Setup:
9  Set the terminal baud rate to 115200 and select the newline terminator.
10  Equipment required is a precision voltage source and a precision
11  voltmeter. No external power supply is required. Ensure all jumpers on
12  the demo board are installed in their default positions from the
13  factory.
14 
15  How to test Single-Ended mode:
16  The voltage source should be connected to the ADC such that the negative
17  lead is connected to ground. The positive lead may be connected to any
18  channel input. Ensure voltage is within analog input voltage range -0.3V to
19  +2.5V.
20 
21  How to test Differential Mode:
22  The voltage source should be connected with positive and negative leads to
23  paired channels. The voltage source negative output must also be connected to
24  the COM pin in order to provide a ground-referenced voltage. Ensure voltage is
25  within analog input voltage range -0.3V to +2.5V. Swapping input voltages
26  results in a reversed polarity reading.
27 
28  How to calibrate:
29  Apply 100mV to CH0 with respect to COM pin. Next, Measure this voltage with a
30  precise voltmeter and enter this value. (This takes the reading.) Apply a higher
31  voltage than the first voltage or approximately 2.40 volts to CH0. Anything
32  above 2.40 is not reccomened. It is very likely to reach the ADC's full-scale
33  and poduce invalide lsb and offset. Measure this voltage with a precise
34  voltmeter and enter this value. Calibration is now stored in EEPROM. Upon
35  startup the calibration values will be restored.
36 
37 USER INPUT DATA FORMAT:
38  decimal : 1024
39  hex : 0x400
40  octal : 02000 (leading 0 "zero")
41  binary : B10000000000
42  float : 1024.0
43 
44 @endverbatim

166 void loop()
167 {
168  int16_t user_command; // The user input command
169  uint8_t acknowledge = 0;
170  if (demo_board_connected)
171  {
172  if (Serial.available()) // Check for user input
173  {
174  user_command = read_int(); // Read the user command
175  if (user_command != 'm')
176  Serial.println(user_command); // Prints the user command to com port
177  switch (user_command)
178  {
179  case 1:
180  acknowledge |= menu_1_read_single_ended();
181  break;
182  case 2:
183  acknowledge |= menu_2_read_differential();
184  break;
185  case 3:
186  acknowledge |= menu_3_calibrate();
187  break;
188  default:
189  Serial.println("Incorrect Option");
190  break;
191  }
192  if (acknowledge)
193  Serial.println(F("***** SPI ERROR ******"));
194  Serial.print(F("\n*************************\n"));
195  print_prompt();
196  }
197  }

370 //! Calibrate ADC given two known inputs
371 //! @return 0 if successful, 1 if timeout for EOC
372 int8_t menu_3_calibrate()
373 {
374  // Calibration
375  float fs_voltage; // Measured full-scale voltage
376  float zero_voltage; // Measured zero voltage
377  uint32_t zero_code; // Cal zero code
378  uint32_t fs_code; // Cal full scale code
379  uint8_t adc_command; // The LTC2418 command byte
380  uint32_t adc_code; // The LTC2418 code
381 
382  Serial.println("Apply 100mV to CH0 with respect to COM");
383  Serial.println("or apply a voltage greater than GND for the");
384  Serial.println("lower point in two point calibration");
385  Serial.print("Enter the measured input voltage:");
386  zero_voltage = read_float();
387  Serial.println(zero_voltage,4);
388  adc_command = BUILD_COMMAND_SINGLE_ENDED[0]; // Build ADC command byte for voltage input
389  if (LTC2418_EOC_timeout(LTC2418_CS, MISO_TIMEOUT)) // Check for EOC
390  return(1);
391  LTC2418_read(LTC2418_CS, adc_command, &adc_code); // Throw away previous reading
392  delay(100);
393  if (LTC2418_EOC_timeout(LTC2418_CS, MISO_TIMEOUT)) // Check for EOC
394  return(1);
395  LTC2418_read(LTC2418_CS, adc_command, &zero_code); // Measure zero
396 
397  Serial.println("Apply ~2.40V input voltage to CH0");
398  Serial.println("or apply a voltage greater that first");
399  Serial.println("voltage and less than 2.40V.");
400  Serial.print("Enter the measured input voltage: ");
401 
402  fs_voltage = read_float();
403  Serial.println(fs_voltage, 4);
404  adc_command = BUILD_COMMAND_SINGLE_ENDED[0]; // Build ADC command byte for voltage input
405  if (LTC2418_EOC_timeout(LTC2418_CS, MISO_TIMEOUT)) // Check for EOC
406  return(1);
407  LTC2418_read(LTC2418_CS, adc_command, &adc_code); // Throw away previous reading
408  delay(100);
409  if (LTC2418_EOC_timeout(LTC2418_CS, MISO_TIMEOUT)) // Check for EOC
410  return(1);
411  LTC2418_read(LTC2418_CS, adc_command, &fs_code); // Measure full scale
412  LTC2418_cal_voltage(zero_code, fs_code, zero_voltage, fs_voltage, &LTC2418_lsb , &LTC2418_offset_code);
413 
414  Serial.print("ADC offset : ");
415  Serial.println(LTC2418_offset_code);
416  Serial.print("ADC lsb : ");
417  Serial.print(LTC2418_lsb*1.0e9, 4);
418  Serial.println("nV (32-bits)");
419  store_calibration();
420  return(0);
421 }

506 //! Store measured calibration parameters to nonvolatile EEPROM on demo board

511  eeprom_write_int32(EEPROM_I2C_ADDRESS, LTC2418_offset_code, EEPROM_CAL_STATUS_ADDRESS+2); // Offset
512  eeprom_write_float(EEPROM_I2C_ADDRESS, LTC2418_lsb, EEPROM_CAL_STATUS_ADDRESS+6); // LSB

516 //! Read stored calibration parameters from nonvolatile EEPROM on demo board
517 //! @return return 1 if successful, 0 if not
518 int8_t restore_calibration()
519 {
520  int16_t cal_key;
521  // read the cal key from the EEPROM
522  eeprom_read_int16(EEPROM_I2C_ADDRESS, &cal_key, EEPROM_CAL_STATUS_ADDRESS);
523  if (cal_key == EEPROM_CAL_KEY)
524  {
525  // calibration has been stored, read offset and lsb
526  eeprom_read_int32(EEPROM_I2C_ADDRESS, &LTC2418_offset_code, EEPROM_CAL_STATUS_ADDRESS+2); // Offset
527  eeprom_read_float(EEPROM_I2C_ADDRESS, &LTC2418_lsb, EEPROM_CAL_STATUS_ADDRESS+6); // LSB
528  Serial.println("Calibration Restored");
529  return(1);
530  }
531  else
532  {
533  Serial.println("Calibration not found");
534  return(0);
535  }
