1 /*!
2 Linear Technology DC682A Demonstration Board.
3 LTC1859: 16-Bit, 8-Channel 100ksps SAR ADC with SPI Interface.
4 
5 @verbatim
6 
7 NOTES
8  Setup:
9  Set the terminal baud rate to 115200 and select the newline terminator. Equipment
10  required is a voltage source (preferably low-noise) and a precision voltmeter. No
11  external power supply is required. Ensure all jumpers on the demo board are
12  installed in their default positions from the factory. Refer to Demo Manual
13  DC682A.
14 
15  How to test Single-Ended mode:
16  Bipolar Mode:
17  The voltage source should be connected to one of the inputs CH0,...CH7. Ensure the input is
18  within its specified absolute input voltage range. (It is easiest
19  to tie the voltage source negative terminal to COM.) Ensure the voltage
20  source is set within the range of 0V to +5V for low range, or 0V to 10V for high range.
21  (Swapping input voltages results in a reversed polarity reading.)
22 
23  Unipolar Mode:
24  The voltage source should be connected to one of the inputs CH0,...CH7. Ensure the input is
25  within its specified absolute input voltage range. (It is easiest
26  to tie the voltage source negative terminal to COM.) Ensure the voltage
27  source is set within the range of 0V to +5V for low range, or 0V to 10V for high range.
28 
29  How to test Differential Mode:
30  Bipolar Mode:
31  The voltage source should be connected between paired channels. Ensure both
32  inputs are within their specified absolute input voltage range. (It is easiest
33  to tie the voltage source negative terminal to COM or GND.) Ensure the voltage
34  source is set within the range of 0V to +5V for low range, or 0V to 10V for high range (differential voltage range).
35  (Swapping input voltages results in a reversed polarity reading.)
36 
37  Unipolar Mode:
38  The voltage source should be connected between paired channels. (It is easiest
39  to tie the voltage source negative terminal to COM or GND.) Ensure both inputs
40  are within their specified absolute input voltage range. (It is easiest to tie
41  the voltage source negative terminal to COM or GND.) Ensure the voltage source
42  is set within the range of 0V to +5V for low range, or 0V to 10V for high range (differential voltage range).
43 
44 USER INPUT DATA FORMAT:
45  decimal : 1024
46  hex : 0x400
47  octal : 02000 (leading 0 "zero")
48  binary : B10000000000
49  float : 1024.0
50 
51 @endverbatim

194 //! Read channels
195 //! @return 1 if successful, 0 if not
196 void menu_1_read_input()
197 {
198  uint8_t user_command;
199  uint16_t adc_command; // The LTC1859 command byte
200  uint16_t adc_code = 0; // The LTC1859 code
201  uint8_t x, y, startcount, endcount;
202  float adc_voltage;
203 
204  if (uni_bipolar == LTC1859_UNIPOLAR_MODE)
205  {
206  if (single_ended_differential == LTC1859_SINGLE_ENDED_MODE)
207  {
208  Serial.println(F(" Read Input in Single-Ended, Unipolar mode:"));
209  Serial.println(F(" Note that in Unipolar mode, input voltages less than zero are reported as 0.0V"));
210  user_command = print_user_command_single_ended();
211  }
212  else
213  {
214  Serial.println(F(" Read Input in Differential, Unipolar mode:"));
215  Serial.println(F(" Note that in Unipolar mode, input voltages less than zero are reported as 0.0V"));
216  user_command = print_user_command_differential();
217  }
218  }
219  else
220  {
221  if (single_ended_differential == LTC1859_SINGLE_ENDED_MODE)
222  {
223  Serial.println(F(" Read Input in Single-Ended, Bipolar mode:"));
224  user_command = print_user_command_single_ended();
225  }
226  else
227  {
228  Serial.println(F(" Read Input in Differential, Bipolar mode:"));
229  user_command = print_user_command_differential();
230  }
231  }
232 
233  if (user_command == 8) //read all channels
234  {
235  startcount = 0;
236  endcount = 7;
237  }
238  else
239  {
240  startcount = user_command;
241  endcount = user_command;
242  }
243 
244  if (single_ended_differential == LTC1859_SINGLE_ENDED_MODE) //read single-ended
245  {
246  adc_command = BUILD_COMMAND_SINGLE_ENDED[0] | uni_bipolar | range_low_high;
247  LTC1859_read(LTC1859_CS, adc_command, &adc_code); // Throws out last reading and starts CH0 conversion
248  for (x = startcount; x <= endcount; x++)
249  {
250  adc_command = BUILD_COMMAND_SINGLE_ENDED[x % 8] | uni_bipolar | range_low_high; // Send channel config for the NEXT conversion to take place
251  LTC1859_read(LTC1859_CS, adc_command, &adc_code); // Read previous channel conversion (x-1) and start next one (x)
252  LTC1859_read(LTC1859_CS, adc_command, &adc_code); // Read previous channel conversion (x-1) and start next one (x)
253  Serial.print(F(" Received Code: b"));
254  Serial.println(adc_code, BIN);
255 
256  adc_voltage = LTC1859_code_to_voltage(adc_code, LTC1859_vref, range_low_high, uni_bipolar);
257  Serial.print(F(" Voltage read on "));
258  Serial.print(F("Ch"));
259  Serial.print(x);
260  Serial.print(F(": "));
261  Serial.print(adc_voltage, 4);
262  Serial.println(F("V"));
263  Serial.println();
264  }
265  }
266 
267  else //read differential
268  {
269  adc_command = BUILD_COMMAND_DIFF[0] | uni_bipolar | range_low_high;
270  LTC1859_read(LTC1859_CS, adc_command, &adc_code); // Throws out last reading and starts CH0 conversion
271  for (x = startcount; x <= endcount; x++)
272  {
273  adc_command = BUILD_COMMAND_DIFF[x % 8] | uni_bipolar | range_low_high; // Send channel config for the NEXT conversion to take place
274  LTC1859_read(LTC1859_CS, adc_command, &adc_code); // Read previous channel conversion (x-1) and start next one (x)
275  LTC1859_read(LTC1859_CS, adc_command, &adc_code); // Read previous channel conversion (x-1) and start next one (x)
276  Serial.print(F(" Received Code: b"));
277  Serial.println(adc_code,BIN);
278 
279  adc_voltage = LTC1859_code_to_voltage(adc_code, LTC1859_vref, range_low_high, uni_bipolar);
280  Serial.print(F(" Voltage read between Chs "));
281  print_channel_selection(x + y);
282  Serial.print(adc_voltage, 4);
283  Serial.println(F("V"));
284  Serial.println();
285  }
286  }
287 }
