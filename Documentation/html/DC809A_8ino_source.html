4 @verbatim

6 The LTC2604/LTC2614/LTC2624 are quad 16-,14- and 12-bit 2.5V to 5.5V
7 rail-to-rail voltage output DACs in 16-lead narrow SSOP packages. These
8 parts have separate reference inputs for each DAC. They have built-in
9 high performance output buffers and are guaranteed monotonic.

127 //! Repeats Linduino loops
128 void loop()
129 {
130  int16_t user_command;
131  static int16_t selected_dac =0; // The selected DAC to be updated (0=A, 1=B ... 5=All). Initialized to "A".
132  // The main control loop
133  if (demo_board_connected) // Do nothing if the demo board is not connected
134  {
135  if (Serial.available()) // Check for user input
136  {
137  user_command = read_int(); // Read the user command
138  Serial.println(user_command);
139  Serial.flush();
140  switch (user_command)
141  {
142  case 1:
143  menu_1_select_dac(&selected_dac);
144  break;
145  case 2:
146  menu_2_write_to_input_register(selected_dac);
147  break;
148  case 3:
149  menu_3_write_and_update_dac(selected_dac);
150  break;
151  case 4:
152  menu_4_update_power_up_dac(selected_dac);
153  break;
154  case 5:
155  menu_5_power_down_dac(selected_dac);
156  break;
157  }
158  Serial.println("\n*****************************************************************");
159  print_prompt(selected_dac);
160  }
161  }
162 }

178 //! Write data to input register, but do not update DAC output
179 void menu_2_write_to_input_register(int16_t selected_dac) //!< what DAC to operate on
180 {
181  uint16_t dac_code;
182 
183  if (prompt_voltage_or_code() == PROMPT_VOLTAGE)
184  dac_code = get_voltage(LTC2604_lsb[selected_dac], LTC2604_offset);
185  else
186  dac_code = get_code();
187 
188  LTC2604_write(LTC2604_CS, LTC2604_CMD_WRITE, address_map[selected_dac], dac_code << shift_count);
189 }
