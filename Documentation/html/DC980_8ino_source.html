106 //! Main Linduino loop
107 //! @return void
108 void loop()
109 {
110  uint8_t user_command;
111  uint16_t return_val;
112 
113  int i = 0;
114 
115  if (Serial.available()) //! Checks for user input
116  {
117  user_command = read_int(); //! Reads the user command
118  if (user_command != 'm')
119  Serial.println(user_command);
120 
121  switch (user_command) //! Prints the appropriate submenu
122  {
123 
124  case 1 :
125  Serial.print(F("\n****INITIALIZING THE LTC2970****\n"));
126  ltc2970_configure();
127  break;
128 
129  case 2 :
130  Serial.print(F("\n****ENABLE LTC2970 CHANNEL 0 AND CHANNEL 1****\n"));
131  ltc2970_dac_disconnect(smbus, ltc2970_i2c_address, 0);
132  ltc2970_gpio_up(smbus, ltc2970_i2c_address, 0);
133 
134  ltc2970_dac_disconnect(smbus, ltc2970_i2c_address, 1);
135  ltc2970_gpio_up(smbus, ltc2970_i2c_address, 1);
136  break;
137 
138  case 3 :
139  Serial.print(F("\n****SOFT CONNECT LTC2970 DAC0 and DAC1****\n"));
140  ltc2970_soft_connect_dac(smbus, ltc2970_i2c_address, 0);
141  ltc2970_soft_connect_dac(smbus, ltc2970_i2c_address, 1);
142  break;
143 
144  case 4 :
145  Serial.print(F("\n****SERVO CHANNEL 0 and 1 VOLTAGES 10% LOW****\n"));
146  ltc2970_servo_to_adc_val(smbus, ltc2970_i2c_address, 0, servo0_value_marg);
147  ltc2970_servo_to_adc_val(smbus, ltc2970_i2c_address, 1, servo1_value_marg);
148  break;
149 
150  case 5 :
151  Serial.print(F("\n****SERVO CHANNEL 0 and 1 VOLTAGES TO NOMINAL****\n"));
152  ltc2970_servo_to_adc_val(smbus, ltc2970_i2c_address, 0, servo0_value_nom);
153  ltc2970_servo_to_adc_val(smbus, ltc2970_i2c_address, 1, servo1_value_nom);
154  break;
155 
156  case 6 :
157  Serial.print(F("\n****ADC CH_0 VOLTAGE = (HEX VALUE)\n"));
158  return_val = smbus->readWord(ltc2970_i2c_address, LTC2970_CH0_A_ADC);
159  Serial.println(((return_val & 0x7FFF)*500e-6), DEC);
160  Serial.println(return_val, HEX);
161 
162  Serial.print(F("\n****ADC CH_1 VOLTAGE = (HEX VALUE)\n"));
163  return_val = smbus->readWord(ltc2970_i2c_address, LTC2970_CH1_A_ADC);
164  Serial.println(((return_val & 0x7FFF)*500e-6), DEC);
165  Serial.println(return_val, HEX);
166  break;
167 
168  case 7 :
169  Serial.print(F("\n****ADC CH_0 CURRENT = (HEX VALUE)\n"));
170  return_val = smbus->readWord(ltc2970_i2c_address, LTC2970_CH0_B_ADC);
171  Serial.println((((return_val & 0x7FFF)*500e-6)/0.007), DEC);
172  Serial.println(return_val, HEX);
173 
174  Serial.print(F("\n****ADC CH_1 CURRENT = (HEX VALUE)\n"));
175  return_val = smbus->readWord(ltc2970_i2c_address, LTC2970_CH1_B_ADC);
176  Serial.println((((return_val & 0x7FFF)*500e-6)/0.008), DEC);
177  Serial.println(return_val, HEX);
178  break;
179 
180  case 8 :
181  Serial.print(F("\n****PRINT FAULTS, CLEAR LATCHED FAULTS \n"));
182  ltc2970_read_faults(smbus, ltc2970_i2c_address);
183  break;
184 
185  case 9 :
186  Serial.print(F("\n****PRINT DIE TEMPERATURE \n"));
187  ltc2970_print_die_temp (smbus, ltc2970_i2c_address);
188  break;
189 
190  default:
191  Serial.println(F("Incorrect Option"));
192  break;
193  }
194  print_prompt();
195  }

235 //! Writes configuration values to the LTC2970 registers
236 //! @return void
237 void ltc2970_configure()
238 {
239  uint16_t return_val;
240  //start the 2970 by configuring all of its registers for this application
241  // use SMbus commands
242  smbus->writeWord(ltc2970_i2c_address, LTC2970_FAULT_EN, 0x0168);
243  smbus->writeWord(ltc2970_i2c_address, LTC2970_IO, 0x000A);
244  smbus->writeWord(ltc2970_i2c_address, LTC2970_ADC_MON, 0x007F);
245  smbus->writeWord(ltc2970_i2c_address, LTC2970_VDD_OV, 0x2CEC);
246  smbus->writeWord(ltc2970_i2c_address, LTC2970_VDD_UV, 0x2328);
247  smbus->writeWord(ltc2970_i2c_address, LTC2970_V12_OV, 0x3FFF);
248  smbus->writeWord(ltc2970_i2c_address, LTC2970_V12_UV, 0x00000);
249 
250  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH0_A_OV, 0x2AF8);
251  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH0_A_UV, 0x2328);
252  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH0_A_SERVO, 0x0000);
253  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH0_A_IDAC, 0x0084);
254  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH0_B_OV, 0x3FFF);
255  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH0_B_UV, 0x0000);
256 
257  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH1_A_OV, 0x1C5D);
258  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH1_A_UV, 0x1770);
259  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH1_A_SERVO, 0x0000);
260  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH1_A_IDAC, 0x0084);
261  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH1_B_OV, 0x3FFF);
262  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH1_B_UV, 0x0000);
263 }
