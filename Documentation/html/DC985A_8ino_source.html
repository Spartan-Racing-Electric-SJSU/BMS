111 //! Repeats Linduino loop
112 void loop()
113 {
114  if (1)
115  {
116  int8_t ack = 0; // I2C acknowledge indicator
117  static uint8_t user_command; // The user input command
118  if (Serial.available()) //! Do nothing if serial is not available
119  {
120  user_command = read_int(); //! Read user input command
121  if (user_command != 'm')
122  Serial.println(user_command);
123  Serial.println();
124  ack = 0;
125 
126  switch (user_command) //! Prints the appropriate submenu
127  {
128  case 1:
129  ack |= main_menu_1_continuous_mode(); // continous mode
130  break;
131  case 2:
132  ack |= main_menu_2_read_and_clear_faults(); // read and clear faults
133  break;
134  case 3:
135  ack |= main_menu_3_send_ARA();// read address of alert
136  break;
137  case 4:
138  ack |= main_menu_4_manage_alerts(); // manage alerts
139  break;
140  case 5:
141  ack |= main_menu_5_settings(); // settings
142  break;
143  case 6:
144  ack |= main_menu_6_read_all_registers();
145  break;
146  default:
147  Serial.println("Incorrect Option");
148  break;
149  }
150 
151  print_prompt();
152  }
153  }
154 }

341 //! Function to read and clear fault register
342 int8_t main_menu_2_read_and_clear_faults()
343 //! @return Returns the state of the acknowledge bit after the I2C address write. 0=acknowledge, 1=no acknowledge.
344 {
345  int8_t ack = 0;
346 
347  uint8_t faults;
348  ack |= LTC4245_read(LTC4245_I2C_ADDRESS, LTC4245_FAULT1_REG, &faults);
349  if (faults != 0)
350  {
351  Serial.println(F("Faults Detected :"));
352  if (faults & LTC4245_NEG12V_OVERCURRENT_FAULT)
353  Serial.println(F(" Overcurrent Fault Occurred on –12V Supply"));
354  if (faults & LTC4245_3_3V_OVERCURRENT_FAULT)
355  Serial.println(F(" Overcurrent Fault Occurred on 3.3V Supply"));
356  if (faults & LTC4245_5V_OVERCURRENT_FAULT)
357  Serial.println(F(" Overcurrent Fault Occurred on 5V Supply"));
358  if (faults & LTC4245_12V_OVERCURRENT_FAULT)
359  Serial.println(F(" Overcurrent Fault Occurred on 12V Supply"));
360  if (faults & LTC4245_NEG12V_UNDERVOLTAGE_FAULT)
361  Serial.println(F(" Undervoltage Fault Occurred on –12V Supply"));
362  if (faults & LTC4245_3_3V_UNDERVOLTAGE_FAULT)
363  Serial.println(F(" Undervoltage Fault Occurred on 3.3V Supply"));
364  if (faults & LTC4245_5V_UNDERVOLTAGE_FAULT)
365  Serial.println(F(" Undervoltage Fault Occurred on 5V Supply"));
366  if (faults & LTC4245_12V_UNDERVOLTAGE_FAULT)
367  Serial.println(F(" Undervoltage Fault Occurred on 12V Supply"));
368  }
369  faults = 0;
370  ack |= LTC4245_read(LTC4245_I2C_ADDRESS, LTC4245_FAULT2_REG, &faults);
371  if (faults != 0)
372  {
373  if (faults & LTC4245_GPIO1_STATE_CHANGE)
374  Serial.println(F(" GPIO1 Pin Changed State"));
375  if (faults & LTC4245_BD_SEL_STATE_CHANGE)
376  Serial.println(F(" BD_SEL# Pin Changed State"));
377  if (faults & LTC4245_FET_SHORT_FAULT)
378  Serial.println(F(" FET Short Detected on at Least One Supply"));
379  if (faults & LTC4245_PGI_FAULT)
380  Serial.println(F(" PGI Fault Occurred"));
381  if (faults & LTC4245_NEG12V_POWER_BAD_FAULT)
382  Serial.println(F(" –12V Power Bad When VEEOUT Went High"));
383  if (faults & LTC4245_3_3V_POWER_BAD_FAULT)
384  Serial.println(F(" 3.3V Power Bad When 3VOUT Went Low"));
385  if (faults & LTC4245_5V_POWER_BAD_FAULT)
386  Serial.println(F(" 5V Power Bad When 5VOUT Went Low"));
387  if (faults & LTC4245_12V_POWER_BAD_FAULT)
388  Serial.println(F(" 12V Power Bad When 12VOUT Went Low"));
389  }
390  ack |= LTC4245_write(LTC4245_I2C_ADDRESS, LTC4245_FAULT1_REG, 0x00);
391  ack |= LTC4245_write(LTC4245_I2C_ADDRESS, LTC4245_FAULT2_REG, 0x00);
392  Serial.print(F("\nALL FAULTS CLEARED \n\n"));
393 
394  return (ack);
395 }
