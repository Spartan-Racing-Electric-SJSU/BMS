121 //! Repeats Linduino loop
122 void loop()
123 {
124  if (demo_board_connected)
125  {
126  int8_t ack = 0; // I2C acknowledge indicator
127  static uint8_t user_command; // The user input command
128  if (Serial.available()) //! Do nothing if serial is not available
129  {
130  user_command = read_int(); //! Read user input command
131  if (user_command != 'm')
132  Serial.println(user_command);
133  Serial.println();
134  ack = 0;
135  switch (user_command) //! Prints the appropriate submenu
136  {
137  case 1:
138  ack |= main_menu_1_continuous_mode(); // continous mode
139  break;
140  case 2:
141  ack |= main_menu_2_read_and_clear_faults(); // read and clear faults
142  break;
143  case 3:
144  ack |= main_menu_3_send_ARA();// read address of alert
145  break;
146  case 4:
147  ack |= main_menu_4_manage_alerts(); // manage alerts
148  break;
149  case 5:
150  ack |= main_menu_5_settings(); // settings
151  break;
152  case 6:
153  ack |= main_menu_6_read_all_registers();
154  break;
155  default:
156  Serial.println("Incorrect Option");
157  break;
158  }
159  if (ack != 0)
160  Serial.println(ack_error);
161  Serial.print(F("*****************************************"));
162  print_main_menu();
163  }
164  }

194 //! Function to read Rsense current, ADIN voltage and ADIN2 voltage in continous mode
195 int8_t main_menu_1_continuous_mode()
196 //! @return Returns the state of the acknowledge bit after the I2C address write. 0=acknowledge, 1=no acknowledge.
197 {
198  int8_t ack = 0;
199  float current, adin_voltage, adin2_voltage;
200  uint16_t current_code, adin_code, adin2_code;
201 
202  do
203  {
204  uint8_t faults;
205  Serial.print(F("********** Press Enter to Exit ***********\n\n"));
206  ack |= LTC4261_read(LTC4261_I2C_Address, LTC4261_FAULT_REG, &faults);
207  if (faults != 0)
208  {
209  Serial.println(F("Faults Detected :"));
210  if (faults & LTC4261_EXTERNAL_FAULT )
211  Serial.println(F(" EXTERNAL FAULT DETECTED"));
212  if (faults & LTC4261_PGIO_INPUT_HIGH )
213  Serial.println(F(" PGIO INPUT HIGH DETECTED"));
214  if (faults & LTC4261_FET_SHORT_FAULT )
215  Serial.println(F(" FET SHORT DETECTED"));
216  if (faults & LTC4261_EN_STATE_CHANGE)
217  Serial.println(F(" !EN STATE CHANGE DETECTED"));
218  if (faults & LTC4261_POWER_BAD_FAULT )
219  Serial.println(F(" BAD POWER DETECTED"));
220  if (faults & LTC4261_OVERCURRENT_FAULT )
221  Serial.println(F(" OVERCURRENT DETECTED"));
222  if (faults & LTC4261_UNDERVOLTAGE_FAULT )
223  Serial.println(F(" UNDERVOLTAGE DETECTED"));
224  if (faults & LTC4261_OVERVOLTAGE_FAULT )
225  Serial.println(F(" OVERVOLTAGE DETECTED"));
226  }
227  else
228  Serial.println(F(" NO FAULTS DETECTED"));
229 
230  ack |= LTC4261_read_10_bits(LTC4261_I2C_Address, LTC4261_SENSE_MSB_REG, &current_code);
231  ack |= LTC4261_read_10_bits(LTC4261_I2C_Address, LTC4261_ADIN_MSB_REG, &adin_code);
232  ack |= LTC4261_read_10_bits(LTC4261_I2C_Address, LTC4261_ADIN2_MSB_REG, &adin2_code);
233 
234  current = LTC4261_code_to_current(current_code, resistor, current_lsb);
235  adin_voltage = LTC4261_ADIN_code_to_voltage(adin_code, adin_lsb, resistive_ratio);
236  adin2_voltage = LTC4261_ADIN_code_to_voltage(adin2_code, adin_lsb, resistive_ratio);
237 
238  Serial.print(F("\nCURRENT CODE: "));
239  Serial.println(current_code);
240  Serial.print(F("CURRENT CALCULATED: "));
241  Serial.println(current,4);
242  Serial.print(F("\nADIN CODE: "));
243  Serial.println(adin_code, HEX);
244  Serial.print(F("VOLTAGE CALCULATED BY ADIN: "));
245  Serial.println(adin_voltage,4);
246  Serial.print(F("\nADIN2 CODE: "));
247  Serial.println(adin2_code, HEX);
248  Serial.print(F("VOLTAGE CALCULATED BY ADIN2: "));
249  Serial.println(adin2_voltage,4);
250  Serial.print("\n");
251  delay(2000);
252  }
253  while (Serial.available()==false && ack== 0);
254  if (Serial.available())
255  read_int(); // clears the Serial.available()
256  return (ack);
257 }
