7 The LTC1863/LTC1867 are pin-compatible, 8-channel 12-/16-bit A/D converters with
8 serial I/O, and an internal reference. The ADCs typically draw only 1.3mA from a
9 single 5V supply. The 8-channel input multiplexer can be configured for either
10 single-ended or differential inputs and unipolar or bipolar conversions (or
11 combinations thereof). The automatic nap and sleep modes benefit power sensitive
12 applications.
13 
14 The LTC1867's DC performance is outstanding with a +/-2LSB INL specification and
15 no missing codes over temperature. The signal-to-noise ratio (SNR) for the

94 // Calculates the LTC1867 input's bipolar voltage given the two's compliment data and lsb weight
95 float LTC1867_bipolar_code_to_voltage(uint16_t adc_code, float LTC1867_lsb, int32_t LTC1867_offset_bipolar_code)
96 {
97  float adc_voltage, sign = 1.0;
98  if (adc_code>>15)
99  {
100  adc_code = (adc_code ^ 0xFFFF)+1; //! 1) Convert ADC code from two's complement to binary
101  sign = -1;
102  }
103  adc_voltage=((float)(adc_code+LTC1867_offset_bipolar_code))*LTC1867_lsb*sign; //! 2) Calculate voltage from ADC code, lsb, offset.
104  return(adc_voltage);

107 // Calibrate the lsb
108 void LTC1867_cal_voltage(uint16_t zero_unipolar_code, uint16_t zero_bipolar_code, uint16_t fs_code, float zero_voltage, float fs_voltage, float *LTC1867_lsb, int32_t *LTC1867_offset_unipolar_code, int32_t *LTC1867_offset_bipolar_code)
109 {
110  float temp_offset;
111  *LTC1867_lsb = (fs_voltage-zero_voltage)/((float)(fs_code - zero_unipolar_code)); //! 1) Calculate the LSB
112 
113  temp_offset = (zero_voltage/ *LTC1867_lsb) - zero_unipolar_code; //! 2) Calculate Unipolar offset
114  temp_offset = (temp_offset > (floor(temp_offset) + 0.5)) ? ceil(temp_offset) : floor(temp_offset); //! 3) Round
115  *LTC1867_offset_unipolar_code = (int32_t)temp_offset; //! 4) Cast as int32_t
116 
117  temp_offset = (zero_voltage / *LTC1867_lsb) - zero_bipolar_code ; //! 5) Calculate Bipolar offset

120 }
