94 // Transmits 24 bits (3 bytes) of configuration information and
95 // reads back 24 bytes of data (3 bytes/ 24 bits for each channel)
96 // 24 bits: 18 bit data + 3 bit config + 3 bit channel number
97 // Read back is done in a new cycle
98 void LTC23XX_read(uint8_t cs_pin, uint32_t config_word, uint8_t data_array[24])
99 {
100  int i, j;
101  uint8_t tx_array[24];
102 
103  tx_array[23] = (uint8_t)(config_word >> 16);
104  tx_array[22] = (uint8_t)(config_word >> 8);
105  tx_array[21] = (uint8_t)(config_word);
106  for (i = 20; i >= 0; --i)
107  {
108  tx_array[i] = 0;
109  }
110  spi_transfer_block(cs_pin, tx_array, data_array, 24);
111 }

113 int32_t sign_extend_17(uint32_t data)
114 {

124 // Calculates the voltage from ADC output data depending on the channel configuration
125 float LTC23XX_voltage_calculator(uint32_t data, uint8_t channel_configuration)
126 {
127  float voltage;
128  int32_t data_signed;
129  switch (channel_configuration)
130  {
131  case 0:
132  voltage = 0;
133  break; // Disable Channel
134  case 1:
135  voltage = (float)data * (1.25 * VREF / 1.000) / POW2_18;
136  break;
137  case 2:
138  data_signed = sign_extend_17(data);
139  voltage = (float)data_signed * (1.25 * VREF / 1.024) / POW2_17;
140  break;
141  case 3:
142  data_signed = sign_extend_17(data);
143  voltage = (float)data_signed * (1.25 * VREF / 1.000) / POW2_17;
144  break;
145  case 4:
146  voltage = (float)data * (2.50 * VREF / 1.024) / POW2_18;
147  break;
148  case 5:
149  voltage = (float)data * (2.50 * VREF / 1.000) / POW2_18;
150  break;
151  case 6:
152  data_signed = sign_extend_17(data);
153  voltage = (float)data_signed * (2.50 * VREF / 1.024) / POW2_17;
154  break;
155  case 7:
156  data_signed = sign_extend_17(data);
157  voltage = (float)data_signed * (2.50 * VREF ) / POW2_17;
158  break;
159  }
160  return voltage;
161 }
