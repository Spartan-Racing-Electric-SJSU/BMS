6 The LTC2421/LTC2422 are 1- and 2-channel 2.7V to 5.5V micropower 20-bit analog-
7 to-digital converters with an integrated oscillator, 8ppm INL and 1.2ppm RMS
8 noise. These ultrasmall devices use delta-sigma technology and a new digital
9 filter architecture that settles in a single cycle. This eliminates the latency
10 found in conventional delta-sigma converters and simplifies multiplexed
11 applications. Through a single pin, the LTC2421/LTC2422 can be configured for
12 better than 110dB rejection at 50Hz or 60Hz +/-2%, or can be driven by an
13 external oscillator for a user defined rejection frequency in the range 1Hz to
14 120Hz. The internal oscillator requires no external frequency setting
15 components.

79 uint8_t LTC2422_EOC_timeout(uint8_t cs, uint16_t miso_timeout)
80 // Checks for EOC with a specified timeout (ms)
81 {
82  uint16_t timer_count = 0; // Timer count for MISO
83  output_low(cs); //! 1) Pull CS low
84  while (1) //! 2) Wait for SDO (MISO) to go low
85  {
86  if (input(MISO) == 0) break; //! 3) If SDO is low, break loop
87  if (timer_count++>miso_timeout) // If timeout, return 1 (failure)
88  {
89  output_high(cs); // Pull CS high
90  return(1);
91  }
92  else
93  delay(1);
94  }
95  output_high(cs); // Pull CS high
96  return(0);
97 }

99 // Read ADC code from the LTC2422. Does not wait for end-of-conversion.
100 // To automatically wait for conversion to complete, use the LTC2422_EOC_timeout before this function..
101 void LTC2422_adc_read(uint8_t cs, uint8_t *adc_channel, int32_t *code)
102 {
103  LT_union_int32_4bytes data, command; // LTC2422 data
104 
105  command.LT_int32 = 0x00000000;
106 
107  spi_transfer_block(LTC2422_CS, command.LT_byte , data.LT_byte, (uint8_t)3);
108  if (data.LT_byte[2] & 0x40) // Obtains Channel Number
109  {
110  *adc_channel = 1;
111  }
112  else
113  {
114  *adc_channel = 0;
115  }
116  data.LT_byte[2] &= 0x3F; // Clear channel bit here so code to voltage function doesn't have to.
117  data.LT_byte[3] = 0x00; // Make sure MS Byte is zero, data is right-justified.
118  *code = data.LT_int32; // Return data
119 }
