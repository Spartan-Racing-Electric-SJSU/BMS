89 int8_t LTC2449_EOC_timeout(uint8_t cs, uint16_t miso_timeout)
90 // Checks for EOC with a specified timeout (ms)
91 {
92  uint16_t timer_count = 0; // Timer count for MISO
93  output_low(cs); //! 1) Pull CS low
94  while (1) //! 2) Wait for SDO (MISO) to go low
95  {
96  if (input(MISO) == 0) break; //! 3) If SDO is low, break loop
97  if (timer_count++>miso_timeout) // If timeout, return 1 (failure)
98  {
99  output_high(cs); // Pull CS high
100  return(1);
101  }
102  else
103  delay(1);
104  }
105  output_high(cs); // Pull CS high
106  return(0);
107 }

111 {

116  command.LT_byte[2] = temp_comm.LT_byte[0];
117  command.LT_byte[1] = 0;
118  command.LT_byte[0] = 0;
119 
120  output_low(cs); //! 1) Pull CS low

122  spi_transfer_block(cs, command.LT_byte, data.LT_byte, (uint8_t)4); //! 2) Transfer arrays
123 
124  output_high(cs); //! 3) Pull CS high
125  *adc_code = data.LT_int32;

128 // Calculates the voltage corresponding to an adc code, given lsb weight (in volts) and the calibrated
129 // adc offset code (zero code that is subtracted from adc_code).
130 float LTC2449_code_to_voltage(int32_t adc_code, float LTC2449_lsb, int32_t LTC2449_offset_code)
131 {
132  float adc_voltage;
133  adc_code -= 536870912; //! 1) Converts offset binary to binary
134  adc_voltage=(float)(adc_code+LTC2449_offset_code)*LTC2449_lsb; //! 2) Calculate voltage from ADC code, lsb, offset.
135  return(adc_voltage);
136 }

138 // Calculate the lsb weight and offset code given a full-scale code and a measured zero-code.
139 void LTC2449_cal_voltage(int32_t zero_code, int32_t fs_code, float zero_voltage, float fs_voltage, float *LTC2449_lsb, int32_t *LTC2449_offset_code)
140 {
141  zero_code -= 536870912; //! 1) Converts zero code from offset binary to binary
142  fs_code -= 536870912; //! 2) Converts full scale code from offset binary to binary
143 
144  float temp_offset;
145  *LTC2449_lsb = (fs_voltage-zero_voltage)/((float)(fs_code - zero_code)); //! 3) Calculate the LSB
146 
147  temp_offset = (zero_voltage/ *LTC2449_lsb) - zero_code; //! 4) Calculate Unipolar offset
148  temp_offset = (temp_offset > (floor(temp_offset) + 0.5)) ? ceil(temp_offset) : floor(temp_offset); //! 5) Round
149  *LTC2449_offset_code = (int32_t)temp_offset; //! 6) Cast as int32_t
150 }
