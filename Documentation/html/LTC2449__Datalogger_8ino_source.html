1 /*!
2 LTC2449: 24-Bit High Speed 16-Channel Delta Sigma ADCs with Selectable Speed/Resolution
3 
4 @verbatim
5 
6 The LTC2449 is a 16-channel (8-differential) high speed 24-bit No Latency Delta Sigma ADCs.
7 They use a proprietary delta-sigma architecture enabling variable speed/resolution. Through
8 a simple 4-wire serial interface, ten speed/resolution combinations 6.9Hz/280nVRMS to
9 3.5kHz/25uVRMS (4kHz with external oscillator) can be selected with no latency between
10 conversion results or shift in DC accuracy (offset, full-scale, linearity, drift).
11 Additionally, a 2X speed mode can be selected enabling output rates up to 7kHz (8kHz if an
12 external oscillator is used) with one cycle latency.
13 
14 @endverbatim

123 //! Repeats Linduino loop
124 void loop()
125 {
126  static float LTC2499_vref = 5.0; // Reference voltage
127  static uint16_t EOC_timout = 250; // End of conversion timeout
128  static uint16_t loop_delay = 500; // the delay between loops
129  static uint16_t channel_read_delay = 20; // The delay between channel readings
130  static char user_command; // The user input command
131 
132  print_prompt(); // Prints the prompt
133 
134  user_command = read_char(); // Reads the users input
135  Serial.println(user_command); // Displays the user input
136 
137  switch (user_command)
138  {
139  case 'S':
140  case 's':
141  // Single loop
142  read_LTC2449(LTC2499_vref, EOC_timout, channel_read_delay); // Read all the channels
143  break;
144  case 'C':
145  case 'c':
146  // Continous loop
147  do
148  {
149  read_LTC2449(LTC2499_vref, EOC_timout, channel_read_delay); // Read all the channels
150  delay(loop_delay); // Delay before repeating the loop
151  }
152  while (Serial.available() == false); // Check to see is anything was entered
153  read_int(); // Clears the Serial.available
154  break;
155  default:
156  Serial.println(F("Incorrect Option"));
157  break;
158  }
159 }

172 //! Reads the LTC2449 and displays the results
173 void read_LTC2449(float vref, uint16_t eoc_timeout, uint16_t channel_delay)
174 {
175  float results[16];
176  int32_t adc_code = 0;
177  uint8_t i;
178 
179  LTC24XX_EOC_timeout(CS, eoc_timeout); // Wait for end of conversion
180  LTC24XX_SPI_16bit_command_32bit_data(CS, SINGLE_ENDED_CONFIG_ARRAY[0], OSR_CONFIG_ARRAY[7], &adc_code); // Prime with CH0
181 
182  for (i=0; i<=15; i++)
183  {
184  delay(channel_delay);
185  LTC24XX_EOC_timeout(CS, eoc_timeout); // Wait for end of conversion
186  LTC24XX_SPI_16bit_command_32bit_data(CS, SINGLE_ENDED_CONFIG_ARRAY[(i+1)%16], OSR_CONFIG_ARRAY[7], &adc_code); // Read the ADC
187  results[i%16] = LTC24XX_diff_code_to_voltage(adc_code, vref); // Convert ADC code to voltage
188  }
189 
190  print_all(results); // Display results
191 }

201  else Serial.print(F(" , "));
202  }
