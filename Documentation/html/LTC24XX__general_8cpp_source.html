81 int8_t LTC24XX_EOC_timeout(uint8_t cs, uint16_t miso_timeout)
82 // Checks for EOC with a specified timeout (ms)
83 {
84  uint16_t timer_count = 0; // Timer count for MISO
85  output_low(cs); //! 1) Pull CS low
86  while (1) //! 2) Wait for SDO (MISO) to go low
87  {
88  if (input(MISO) == 0) break; //! 3) If SDO is low, break loop
89  if (timer_count++>miso_timeout) // If timeout, return 1 (failure)
90  {
91  output_high(cs); // Pull CS high
92  return(1);
93  }
94  else
95  delay(1);
96  }
97  output_high(cs); // Pull CS high
98  return(0);
99 }

374 float LTC24XX_SE_code_to_voltage(int32_t adc_code, float vref)
375 {
376  float voltage;
377  adc_code -= 0x20000000; //! 1) Subtract offset

379  voltage = voltage / 268435456.0; //! 2) This calculates the input as a fraction of the reference voltage (dimensionless)

381  return(voltage);

385 // This function handles all differential input parts, including the "single-ended" mode on multichannel

425 // Calculate the lsb weight and offset code given a full-scale code and a measured zero-code.
426 void LTC24XX_calibrate_voltage(int32_t zero_code, int32_t fs_code, float zero_voltage, float fs_voltage, float *LTC24XX_lsb, int32_t *LTC24XX_offset_code)
427 {
428  zero_code -= 536870912; //! 1) Converts zero code from offset binary to binary
429  fs_code -= 536870912; //! 2) Converts full scale code from offset binary to binary
430 
431  float temp_offset;
432  *LTC24XX_lsb = (fs_voltage-zero_voltage)/((float)(fs_code - zero_code)); //! 3) Calculate the LSB
433 
434  temp_offset = (zero_voltage/ *LTC24XX_lsb) - zero_code; //! 4) Calculate Unipolar offset
435  temp_offset = (temp_offset > (floor(temp_offset) + 0.5)) ? ceil(temp_offset) : floor(temp_offset); //! 5) Round
436  *LTC24XX_offset_code = (int32_t)temp_offset; //! 6) Cast as int32_t
437 }
