159 // Used to set and clear bits in a control register. bits_to_set will be bitwise OR'd with the register.
160 // bits_to_clear will be inverted and bitwise AND'd with the register so that every location with a 1 will result in a 0 in the register.
161 int8_t LTC2942_register_set_clear_bits(uint8_t i2c_address, uint8_t register_address, uint8_t bits_to_set, uint8_t bits_to_clear)
162 {
163  uint8_t register_data;
164  int8_t ack = 0;
165 
166  ack |= LTC2942_read(i2c_address, register_address, &register_data);
167  register_data = register_data & (~bits_to_clear);
168  register_data = register_data | bits_to_set;
169  ack |= LTC2942_write(i2c_address, register_address, register_data);
170  return(ack);
171 }
