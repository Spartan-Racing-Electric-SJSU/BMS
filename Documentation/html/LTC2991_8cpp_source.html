4 @verbatim
5 
6 The LTC2991 is used to monitor system temperatures, voltages and currents.
7 Through the I2C serial interface, the eight monitors can individually measure
8 supply voltages and can be paired for differential measurements of current sense
9 resistors or temperature sensing transistors. Additional measurements include
10 internal temperature and internal VCC. The internal 10ppm reference minimizes

13 various systems needing temperature, voltage or current data. The LTC2991 fits

92 // Reads a 14-bit adc_code from the LTC2991 but enforces a maximum timeout.
93 // Similar to LTC2991_adc_read except it repeats until the data_valid bit is set, it fails to receive an I2C acknowledge, or the timeout (in milliseconds)
94 // expires. It keeps trying to read from the LTC2991 every millisecond until the data_valid bit is set (indicating new data since the previous
95 // time this register was read) or until it fails to receive an I2C acknowledge (indicating an error on the I2C bus).
96 int8_t LTC2991_adc_read_timeout(uint8_t i2c_address, uint8_t msb_register_address, int16_t *adc_code, int8_t *data_valid, uint16_t timeout, uint8_t status_bit)
97 {
98  int8_t ack = 0;
99  uint8_t reg_data;
100  uint16_t timer_count; // Timer count for data_valid
101 
102  for (timer_count = 0; timer_count < timeout; timer_count++)
103  {
104 
105  if (status_bit<8)
106  {
107  ack |= LTC2991_register_read(i2c_address, LTC2991_STATUS_LOW_REG, &reg_data); //! 1)Read status register until correct data valid bit is set
108  }
109  else
110  {
111  ack |= LTC2991_register_read(i2c_address, LTC2991_STATUS_HIGH_REG, &reg_data); //! 1)Read status register until correct data valid bit is set
112  if (status_bit==8)
113  {
114  status_bit =1;
115  }
116  else
117  {
118  status_bit = 0;
119  }
120  }
121 
122  if ((ack) || (((reg_data>>status_bit)&0x1)==1))
123  {
124  break;
125  }
126 
127  delay(1);
128  }
129 
130  ack |= LTC2991_adc_read(i2c_address, msb_register_address, &(*adc_code), &(*data_valid)); //! 2) It's either valid or it's timed out, we read anyways
131  if (*data_valid !=1)
132  {
133  Serial.println("Data not valid");
134  Serial.println(*data_valid);
135  return (1);
136  }
137  return(ack);
138 }

140 // Reads new data (even after a mode change) by flushing old data and waiting for the data_valid bit to be set.
141 // This function simplifies adc reads when modes are changing. For example, if V1-V2 changes from temperature mode
142 // to differential voltage mode, the data in the register may still correspond to the temperature reading immediately
143 // after the mode change. Flushing one reading and waiting for a new reading guarantees fresh data is received.
144 // If the timeout is reached without valid data (*data_valid=1) the function exits.
145 int8_t LTC2991_adc_read_new_data(uint8_t i2c_address, uint8_t msb_register_address, int16_t *adc_code, int8_t *data_valid, uint16_t timeout)
146 {
147  int8_t ack = 0;
148 
149  ack |= LTC2991_adc_read_timeout(i2c_address, msb_register_address, &(*adc_code), &(*data_valid), timeout, ((msb_register_address/2) - 0x05)); //! 1) Throw away old data
150  ack |= LTC2991_adc_read_timeout(i2c_address, msb_register_address, &(*adc_code), &(*data_valid), timeout, ((msb_register_address/2) - 0x05)); //! 2) Read new data
151 
152  return(ack);
153 }

174 // Used to set and clear bits in a control register. bits_to_set will be bitwise OR'd with the register.
175 // bits_to_clear will be inverted and bitwise AND'd with the register so that every location with a 1 will result in a 0 in the register.
176 int8_t LTC2991_register_set_clear_bits(uint8_t i2c_address, uint8_t register_address, uint8_t bits_to_set, uint8_t bits_to_clear)
177 {
178  uint8_t register_data;
179  int8_t ack = 0;
180 
181  ack |= LTC2991_register_read(i2c_address, register_address, &register_data); //! 1) Read register
182  register_data = register_data & (~bits_to_clear); //! 2) Clear bits that were set to be cleared
183  register_data = register_data | bits_to_set;
184  ack |= LTC2991_register_write(i2c_address, register_address, register_data); //! 3) Write to register with the cleared bits
185  return(ack);
186 }

189 float LTC2991_code_to_single_ended_voltage(int16_t adc_code, float LTC2991_single_ended_lsb)
190 {
191  float voltage;
192  int16_t sign = 1;
193  if (adc_code >> 14)
194  {
195  adc_code = (adc_code ^ 0x7FFF) + 1; //! 1) Converts two's complement to binary
196  sign = -1;
197  }
198  adc_code = (adc_code & 0x3FFF);
199  voltage = ((float) adc_code) * LTC2991_single_ended_lsb * sign; //! 2) Convert code to voltage from lsb
200  return (voltage);
