4 @verbatim
5 
6 The LTC3589 is a complete power management solution for ARM and ARM-based
7 processors and advanced portable microprocessor systems. The device contains
8 three step-down DC/DC converters for core, memory and SoC rails, a
9 buck-boost regulator for I/O at 1.8V to 5V and three 250mA LDO regulators
10 for low noise analog supplies. An I2C serial port is used to control enables,
11 output voltage levels, dynamic voltage scaling, operating modes and status
12 reporting.
13 Regulator start-up is sequenced by connecting outputs to enable pins in the
14 desired order or programmed via the I2C port. System power-on, power-off,
15 and reset functions are controlled by pushbutton interface, pin inputs, or
16 I2C interface.
17 The LTC3589 supports i.MX53/51, PXA and OMAP processors with eight
18 independent rails at appropriate power levels. Other features include
19 interface signals such as the VSTB pin that simultaneously toggle up to
20 four rails between programmed run and standby output voltages. The device
21 is available in a low profile 40-pin 6mm x 6mm exposed pad QFN package.

189 // Writes the Feedback Reference Voltage of LDO2.
190 float LTC3589_set_ldo2_fb_ref(uint8_t i2c_address, uint8_t register_address, float fb_ref_voltage)
191 {
192  uint8_t register_data;
193  uint8_t fb_ref_data = 0;
194  int8_t nack = 0;
195  fb_ref_data = (uint8_t) ((fb_ref_voltage + 6.25 - 362.5) / 12.5); //Generates 5-bit code for closest allowable value to user's chosen reference voltage.
196  nack |= LTC3589_register_read(i2c_address, register_address, &register_data); //Read current register value
197  register_data = register_data & (~LTC3589_FB_REF_MASK); //Clears existing feedback reference bits
198  register_data = register_data | fb_ref_data;
199  nack |= LTC3589_register_write(i2c_address, register_address, register_data); //Writes new register value
200  nack |= LTC3589_set_go_bits(i2c_address);
201  return (362.5 + (fb_ref_data * 12.5));
202 }

296 // Sets the switching mode for the specified buck regulator(s).
297 int8_t LTC3589_set_buck_mode(uint8_t i2c_address, uint8_t buck_number, uint8_t mode)
298 {
299  uint8_t register_data;
300  int8_t ack = 0;
301  if (buck_number == 0xFF)
302  {
303  ack |= LTC3589_set_buck_mode(i2c_address, 0x01, mode);
304  ack |= LTC3589_set_buck_mode(i2c_address, 0x02, mode);
305  ack |= LTC3589_set_buck_mode(i2c_address, 0x03, mode);
306  return(ack);
307  }
308  ack |= LTC3589_register_read(i2c_address, LTC3589_REG_SCR1, &register_data); //Read register
309  register_data = register_data & (~LTC3589_BUCK_MODE_MASK(buck_number)); //Clears the Mode bits in the Buck register
310  register_data = register_data | (mode << ((buck_number*2)-2));
311  ack |= LTC3589_register_write(i2c_address, LTC3589_REG_SCR1, register_data); //Writes new register value
312  return(ack);

373  return(ack);

375  ack |= LTC3589_register_read(i2c_address, LTC3589_REG_VRRCR, &register_data); //Read register
