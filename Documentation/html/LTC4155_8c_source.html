62 /*! @file
63  * @ingroup LTC4155
64  * @brief LTC4155 lightweight, hardware ambiguous, embeddable C Communication
65  * Library.
66  *
67  * Communication is bit-field based as well as whole-register based. This library
68  * automatically masks and right-justifies bit fields to ease programmer workload.
69  *
70  * This C library provides a header file with the complete definitions of registers
71  * and bit fields within those registers, and routines to read and write those
72  * registers and individual bit-fields.
73  *
74  * Outside of the Linduino environment, this library can be built with any
75  * standard C compiler and can be used with different I2C/SMBus communication
76  * hardware simply by swapping out the pointers to appropriate user-supplied
77  * functions *@ref smbus_write_register and *@ref smbus_read_register. An
78  * example is provided using the i2c-tools Linux kernel driver which has been
79  * tested on the BeagleBone Black Linux single board computer. It can also be
80  * readily adapted to common microcontrollers with minimal memory impact on
81  * embedded systems.
82  *
83  * A higher level hardware ambiguous Python communication library is also available.
84  *
85  * Please visit http://www.linear.com/product/LTC4155#code or contact the
86  * factory at 408-432-1900 or www.linear.com for further information.
87  */

139 static inline uint8_t get_mask(uint16_t registerinfo)
140 {
141  uint8_t mask = 1 << get_offset(registerinfo);
142  uint8_t size = get_size(registerinfo);
143  uint8_t i;
144  for (i=0; i<size-1; i++)
145  {
146  mask |= mask << 1;
147  }
148  return mask;
149 }

151 int LTC4155_write_register(LTC4155 chip_handle, uint16_t registerinfo, uint8_t data)
152 {
153  LTC4155_chip_cfg_t *chip = (LTC4155_chip_cfg_t *) chip_handle;
154  int failure;
155  uint8_t command_code = get_subaddr(registerinfo);
156  if (get_size(registerinfo) != 8)
157  {
158  uint8_t offset = get_offset(registerinfo);
159  uint16_t mask = get_mask(registerinfo);
160  uint8_t read_data;
161  failure = chip->read_register(chip->addr,command_code,&read_data,chip->port_configuration);
162  if (failure) return failure;
163  data = (read_data & ~mask) | (data << offset);
164  }
165  return chip->write_register(chip->addr,command_code,data,chip->port_configuration);
166 }

168 int LTC4155_read_register(LTC4155 chip_handle, uint16_t registerinfo, uint8_t *data)
169 {
170  LTC4155_chip_cfg_t *chip = (LTC4155_chip_cfg_t *) chip_handle;
171  int result;
172  uint8_t command_code = get_subaddr(registerinfo);
173  uint8_t offset = get_offset(registerinfo);
174  uint16_t mask = get_mask(registerinfo);
175  result = chip->read_register(chip->addr,command_code,data,chip->port_configuration);
176  *data &= mask;
177  *data = *data >> offset;
178  return result;
179 }
