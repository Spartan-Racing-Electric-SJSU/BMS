87 // Take inputs and apply same settings to both mixer channels
88 uint8_t LTC5566_dupl_settings()
89 {
90  uint8_t power_bit; // Bit for controlling the LTC5566's power mode
91  char *power_prompt = "Enter a command: "; // Prompt for getting power mode
92  uint8_t tune_bits; // Bit for controlling the LTC5566's RF input tune mode
93  char *tune_prompt = "\nDesired RF Input Tune setting: "; // Prompt for getting tune mode
94  uint8_t att_bits; // Bit for controlling the LTC5566's IF attenuation
95  char *att_prompt = "\nDesired IF Attenuation (dB): "; // Prompt for getting att level
96  uint8_t output_register;
97 
98  power_bit = LTC5566_get_power_mode(power_prompt); // Get power mode bit
99  tune_bits = LTC5566_get_tune_mode(tune_prompt); // Get RF input tune mode
100  att_bits = LTC5566_get_att(att_prompt); // Get IF attenuation level
101 
102  mixer_1_value = ((power_bit | tune_bits) | att_bits); // Build mixer command byte
103  whole_register = mixer_1_value; // Load the register command with the first command byte
104  whole_register = whole_register << 8; // Shift the first command byte up to the most significant byte
105  whole_register = whole_register | mixer_1_value; // Load the least significant byte with the mixer command byte
106 
107  LTC5566_write(LTC5566_CS, whole_register, &output); // Send the data to the LTC5566 and read back the previously loaded value
108 
109  output_register = output & 0xFF; // Truncate to 8 bits since the registers received the same settings
110  return output_register; // Return the register's value to the main program
111 }
