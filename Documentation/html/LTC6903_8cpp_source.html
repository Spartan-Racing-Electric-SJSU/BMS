82 // Calculates the code necessary to create the clock frequency
83 // Returns the Code for the LTC6903
84 uint16_t LTC6903_frequency_to_code(float frequency, uint8_t clk)
85 {
86  uint8_t oct;
87  double oct_double;
88  float float_dac;
89  uint16_t dac;
90 
91  // Calculate OCT
92  oct_double = log10((double)((frequency*1000000)/1039));
93  oct_double *= 3.322;
94 
95  // Keep OCT within range
96  if (oct_double>15)
97  oct = 15;
98  if (oct_double<0)
99  oct = 0;
100  oct = (uint8_t)oct_double; // Cast as uint8_t , round down
101 
102  // Calculate DAC code
103  float_dac = 2048-(2078*pow(2, (10+ oct)))/(frequency*1000000); // Calculate the dac code
104  float_dac = (float_dac > (floor(float_dac) + 0.5)) ? ceil(float_dac) : floor(float_dac); // Round
105 
106  // Keep DAC within range
107  if (float_dac>1023)
108  float_dac = 1023;
109  if (float_dac<0)
110  float_dac = 0;
111 
112  dac = (uint16_t)float_dac; // Cast as uint16_t
113 
114  return((uint32_t)((oct<<12)| (dac<<2) | clk));
115 }
