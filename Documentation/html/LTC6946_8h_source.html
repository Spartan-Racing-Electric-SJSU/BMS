277 /* ------------------------------------------------------------------------- */
278 //! FUNCTION: LTC6946_set_frf
279 //! Calculates the integer (N), fractional (NUM) and output divider (OD) SPI values
280 //! using self created 64bit math functions.
281 //!
282 //! Datasheet equations:
283 //! - fvco = fpfd*(N + F)
284 //! - frf = fvco/O
285 //! - fpfd = fref/R
286 //!
287 //! can be modified to the following equations:
288 //! - N = (int) (fvco/fpfd) = (int) frf*O*R/fref
289 //!
290 //! where:
291 //! - N = ndiv, O= odiv in the code below
292 //!
293 //! Linduino One (Arduino Uno) is limited to 32 bit floats/double/long.
294 //! 32 bit math functions will create rounding errors with the above equations,
295 //! that can result in frequency errors.
296 //! Therefore, the following code uses self created 64bit functions for 64bit integer math.
297 //!
298 //! - frf (33 bits) LTC6946-4 max frf/fvco = 6.4GHZ, which is 33 bit number (2^33 = 8.59G)
299 //! - fref (23 bits) LTC6946 min fref = 10MHz, which is a 23 bit number (2^23 = 8.3M)
300 //! - O (3 bits)
301 //! - R (10 bits)
302 //!
303 //! step 1: create 64 bit frf and fref numbers
304 //!
305 //! step 2: calculate O (output divider)
306 //!
307 //! step 3: get current R-divider setting
308 //!
309 //! step 4: calculate frf*O*R
310 //! max bit count/resolution: 33b+3b+10b= 46b
311 //!
312 //! step 5: calculate N(16b), using value from step 1
313 //! - N = (int) frf*O*R/fref
314 //! - max bit count/resolution: 46b-23b = 23b
315 //! @return void
316 void LTC6946_set_frf(char part_version[] //!< LTC6946 part number LTC6946-x
317  );
