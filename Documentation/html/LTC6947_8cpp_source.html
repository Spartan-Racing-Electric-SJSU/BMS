96 /* -------------------------------------------------------------------------
97  FUNCTION: LTC6947_read
98  - reads 8 bit Data field to LTC6947.
99  - has to shift data by one bit to account for RW bit
100  -------------------------------------------------------------------------- */
101 uint8_t LTC6947_read(uint8_t cs, int8_t address)
102 {
103  int8_t address_shift;
104  LT_union_int16_2bytes rx;
105 
106  address_shift =(address << 1) | 0x01; // shift to left to account for R/W bit, set bit high for read
107  spi_transfer_word(cs, address_shift<<8 , &rx.LT_uint16);
108 
109  LTC6947_reg[address]=rx.LT_byte[0];
110  return(rx.LT_byte[0]);
111 }

114 /* -------------------------------------------------------------------------
115  FUNCTION: LTC6947_read_field
116  For SPI FIELDS located in 1 or multiple address location
117  - reads specific address locations
118  - identifies and returns specific field in question
119  - can handle SPI fields in multiple addresses, if MSB bit is in the lower number address
120 --------------------------------------------------------------------------- */
121 long LTC6947_read_field(uint8_t cs, uint8_t address, uint8_t MSB_loc, uint8_t numbits)
122 {
123  int bit_shift, i, num_reg;
124  long field_val, maskbits, pow2;
125 
126  num_reg=0;
127  field_val=0;
128 // determines how many register are used
129  do
130  {
131  bit_shift = (MSB_loc+1)- (numbits-num_reg*8); // determines bit_shift for last register location
132  field_val=LTC6947_read(cs, (address+num_reg))+(field_val<<8); // reads current address locations, shifts previous address location 8 bits
133  num_reg++;
134  }
135  while ((bit_shift<0) && (num_reg<4));
136 
137 // creates a bit mask for complete word,
138  maskbits = 1;
139  pow2=1;
140  for (i=1, maskbits=1; i<numbits; i++)
141  {
142  pow2=pow2*2;
143  maskbits = maskbits+pow2;
144  }
145 
146  field_val=(field_val >>bit_shift) &maskbits;
147  return field_val;
148 }

177 /* -------------------------------------------------------------------------
178  FUNCTION: LTC6947_write_field
179  For SPI FIELDS
180  - reads specific address location
181  - identifies and returns specific field in question
182  - can handle SPI fields in multiple addresses, if MSB bit is in the lower number address
183 ---------------------------------------------------------------------------- */
184 uint8_t LTC6947_write_field(uint8_t cs, long field_data, uint8_t address, uint8_t MSB_loc, uint8_t numbits)
185 {
186  long current_content, desired_content, reg_val;
187  int LSB_loc, i, j, num_reg, bit_shift;
188  long temp_arr[32];
189 
190  for (i=0; i<32 ; i++) temp_arr[i]=0; // init temp_arr
191 
192 // read data in current address location and put in a bit array
193  num_reg=0;
194  current_content=0;
195  do
196  {
197  bit_shift=(MSB_loc+1)-(numbits-num_reg*8);
198  current_content=LTC6947_read(cs, (address+num_reg)) + (current_content<<8);
199 
200  num_reg++;
201  }
202  while ((bit_shift<0) && (num_reg<4));
203  for (i=0; i<(8*num_reg); i++)
204  {
205  temp_arr[i]=(current_content>>i) & 1;
206  }
207 
208 // exchange current bits with desired bits
209  LSB_loc = 8*(num_reg-1)+MSB_loc-numbits+1;
210  for (i=LSB_loc, j=0; i<=(MSB_loc+(num_reg-1)*8); i++, j++)
211  {
212  temp_arr[i] = (field_data>>j) &1;
213  } // end of for loop
214 
215 // reconstruct bits into an integer
216  desired_content = 0;
217  for (i=0; i<(8*num_reg); i++)
218  {
219  desired_content = desired_content | (temp_arr[i]<<i);
220  } // end of for loop
221 
222 // write new field value to part
223  for (i=0; i<num_reg; i++)
224  {
225  reg_val = (desired_content >> 8*(num_reg-1-i)) & 0xff;
226  LTC6947_write(cs, (address+i), reg_val);
227  } // end of for loop
228 } // end of LTC6947_write_field

573 /* -------------------------------------------------------------------------
574  FUNCTION: LTC6947_set_frf
575  Calculates the integer (N), fractional (NUM) and output divider (OD) SPI values
576  using self created 64bit math functions.
577 
578  Datasheet equations
579  fvco = fpfd*(N + F)
580  frf = fvco/O
581  fpfd = fref/R
582  F=NUM/2^18 = NUM/262144
583 
584  can be modified to the following equations
585  N = (int) (fvco/fpfd) = (int) frf*O*R/fref
586  NUM = (int) [262144 * (frf*O*R/fref -N)]
587 
588  where
589  N = ndiv, O= odiv, NUM=fracN_num in the code below
590 
591  Linduino One (Arduino Uno) is limited to 32 bit floats/double/long.
592  32 bit math functions will create rounding errors with the above equations,
593  tha can result in frequency errors.
594  Therefore, the following code uses self created 64bit functions for 64bit integer math.
595 
596  frf (33 bits) LTC6947-4 max frf/fvco = 6.4GHZ, which is 23 bit number (2^33 = 8.59G)
597  fref (23 bits) LTC6947 min fref = 10MHz, which is a 23 bit number (2^23 = 8.3M)
598  O (3 bits)
599  R (5 bits)
600 
601  step 1: create 64 bit frf and fref numbers
602 
603  step 2: calculate O (output divider)
604 
605  step 3: get current R-divider setting
606 
607  step 4: calculate frf*O*R
608  max bit count/resolution: 33b+3b+5b= 41b
609  step 5: calculate N(10b), using value from step 1
610  N = (int) frf*O*R/fref
611  max bit count/resolution: 41b-23b = 18b
612  step 6: calculate NUM (18b)
613  NUM = (int) [262144 * (frf*O*R/fref -N)]
614  = (int) [2^18 * (frf*O*R/fref -N)]
615  to get the correct value the above calculation are performed like this
616  = ((int)((frf*O*R/fref)<<23) - N<<23 + rounding) >> 5, where << & >> are bit shift left, right
617  * note << 23b was chosen, because step one use 41b, 41b +23b = 64b (max number for 64 bit math)
618  ** note >> 5 was chosen because 23-5 = 2^18, which is the same as multiplying by 262144
619  *** the rounding variable is equivalent to adding 0.5 to the NUM number. the integer operation always
620  rounds down, so adding a 0.5 to the number before rounding down ensures the closest value is
621  selected.
622  In this function the lower 5 bits of (frf*O*R/fref)<<23) - N<<23 (possible variable 0-31)
623  are check to see if they are >=16.
624  If they are then 16 is added to this number before performing the int and >>5 functions.
625 ---------------------------------------------------------------------------- */
626 void LTC6947_set_frf()
627 {
628  unsigned long frf_MHz, frf_Hz, fref_MHz, fref_Hz, odiv, rdiv, ndiv, fracN_num, frac_num_remainder;
629  unsigned long N64[2], NUM64[2], R64[2], O64[2], temp_long[2];
630  char buffer[100];
631  unsigned long frf[2], frf_act[2];
632  unsigned long fref[2];
633  unsigned long temp_math[2];
634  unsigned long frf_rdiv_odiv[2];
635  unsigned long frf_rdiv_odiv_int[2];
636  unsigned long roundup[2];
637 
638  /* step 1: create 64 bit frf and fref numbers
639  32 bit xxxx MHz number and 32 bit yyy yyy Hz number. */
640  frf_MHz=LTC6947_Frf_MHz;
641  frf_Hz=LTC6947_Frf_Hz;
642  HZto64(frf, frf_MHz, frf_Hz);
643 
644  fref_MHz=LTC6947_Fref_MHz;
645  fref_Hz=LTC6947_Fref_Hz;
646  HZto64(fref,fref_MHz,fref_Hz);
647 
648 // step 2: calculate O (output divider)
649  odiv=LTC6947_calc_odiv(frf);
650 
651 // step 3: get current R-divider setting
652  rdiv=get_LTC6947_SPI_FIELD(LTC6947_CS,LTC6947_RD); // reads selected field
653 
654 // step 3: calculate frf*O*R
655  frf_rdiv_odiv[0]=0;
656  frf_rdiv_odiv[1]=odiv*rdiv;
657  mul64(frf_rdiv_odiv, frf); // odiv*rdiv*frf
658  frf_rdiv_odiv_int[0]=frf_rdiv_odiv[0]; // copy odiv*rdiv*frf to another variable
659  frf_rdiv_odiv_int[1]=frf_rdiv_odiv[1];
660 
661 // step 4: calculate N(10b), using value from step 3; N = (int) frf*O*R/fref
662  temp_math[0]=fref[0]; // copy fref to temp variable for math operation
663  temp_math[1]=fref[1];
664  div64(frf_rdiv_odiv_int, temp_math); // frf_rdiv_odiv_int= [(frf*odiv*rdiv)]/fref --> int(fvco/fpfd)
665  ndiv=frf_rdiv_odiv_int[1];
666 
667 // step 5: calculate NUM (18b), NUM = (int) [262144 * (frf*O*R/fref -N)]
668 // SEE notes in function header for step 5
669 // ((int)((frf*O*R/fref)<<23) - N<<23 + rounding) >> 5, where << & >> are bit shift left, right
670  shl64by(frf_rdiv_odiv,23);
671  div64(frf_rdiv_odiv, fref); // frf_rdiv_odiv --> [(frf*odiv*rdiv)<<23]/fref

673  shl64by(frf_rdiv_odiv_int,23); // frf_rdiv_odiv_int --> [(int)((frf*odiv*rdiv)/fref)] <<23
674 
675  sub64(frf_rdiv_odiv,frf_rdiv_odiv_int);
676 
677 // NUM is a 18 bit number in the spi register
678 // At this point in the program it is represented by 23 bits
679 // if the lower 5 bits are >=16 (16/31),
680 // then add 16 to the frac_num_remainder. After the bits are
681 // shifted down 5 bits this will be close to added 0.5 bits.
682  frac_num_remainder=(frf_rdiv_odiv[1] & 31L);
683  if (frac_num_remainder >= 16L)
684  {
685  roundup[0]=0L;
686  roundup[1]=16L;
687  add64(frf_rdiv_odiv,roundup);
688  }
689 
690 // shift NUM to get desired 18 bit number for SPI map.
691  shr64by(frf_rdiv_odiv,5);
692  fracN_num=frf_rdiv_odiv[1];
693 
694 // program part and print out results to screen
695  set_LTC6947_SPI_FIELD(LTC6947_CS,LTC6947_OD,odiv); // programs output divider
696  Serial.print("OD = ");
697  Serial.println(odiv);
698 
699  Serial.print("RD = ");
700  Serial.println(rdiv);

703  Serial.print(F("ND = "));
704  Serial.println(ndiv);
705 
706  if (fracN_num > 0L)
707  {
708  set_LTC6947_SPI_FIELD(LTC6947_CS,LTC6947_INTN,0L); // programs fractional mode
709  set_LTC6947_SPI_FIELD(LTC6947_CS,LTC6947_NUM,fracN_num); // programs NUM
710  if ( (ndiv<35) || (ndiv>1019) ) Serial.print(F("N DIV set to invalid setting - REFER TO DATASHEET\n"));
711  }
712  else
713  {
714  set_LTC6947_SPI_FIELD(LTC6947_CS,LTC6947_INTN,1L); // programs INTN
715  Serial.print(F("When NUM=0, changes to Integer Mode (NUM can not be 0)\n"));
716  if ( (ndiv<32) || (ndiv>1023) ) Serial.print(F("N DIV set to invalid setting - REFER TO DATASHEET\n"));
717  }
718  Serial.print(F("NUM = "));
719  Serial.println(fracN_num);
720 
721 
722 ///// calculate actual fout here and print out
723  frf_act[0]=fref[0];
724  frf_act[1]=fref[1];
725  N64[0]=0;
726  NUM64[0]=0;
727  R64[0]=0;
728  O64[0]=0;
729  N64[1]=ndiv;
730  NUM64[1]=fracN_num;
731  R64[1]=rdiv;
732  O64[1]=odiv;

742  Serial.print(F("Actual Output Frequency (closest available) is "));
743  Serial.print(frf_act[1]);

749  temp_math[1]=frf_act[1];
