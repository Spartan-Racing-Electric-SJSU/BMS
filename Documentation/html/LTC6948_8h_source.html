289 /* ------------------------------------------------------------------------- */
290 //! FUNCTION: LTC6948_set_frf
291 //! Calculates the integer (N), fractional (NUM) and output divider (OD) SPI values
292 //! using self created 64bit math functions.
293 //!
294 //! Datasheet equations:
295 //! - fvco = fpfd*(N + F)
296 //! - frf = fvco/O
297 //! - fpfd = fref/R
298 //! - F=NUM/2^18 = NUM/262144
299 //!
300 //! can be modified to the following equations:
301 //! - N = (int) (fvco/fpfd) = (int) frf*O*R/fref
302 //! - NUM = (int) [262144 * (frf*O*R/fref -N)]
303 //!
304 //! where:
305 //! - N = ndiv, O= odiv, NUM=fracN_num in the code below
306 //!
307 //! Linduino One (Arduino Uno) is limited to 32 bit floats/double/long.
308 //! 32 bit math functions will create rounding errors with the above equations,
309 //! that can result in frequency errors.
310 //! Therefore, the following code uses self created 64bit functions for 64bit integer math.
311 //!
312 //! - frf (33 bits) LTC6948-4 max frf/fvco = 6.4GHZ, which is 33 bit number (2^33 = 8.59G)
313 //! - fref (23 bits) LTC6948 min fref = 10MHz, which is a 23 bit number (2^23 = 8.3M)
314 //! - O (3 bits)
315 //! - R (5 bits)
316 //!
317 //! step 1: create 64 bit frf and fref numbers
318 //!
319 //! step 2: calculate O (output divider)
320 //!
321 //! step 3: get current R-divider setting
322 //!
323 //! step 4: calculate frf*O*R
324 //! max bit count/resolution: 33b+3b+5b= 41b
325 //!
326 //! step 5: calculate N(10b), using value from step 1
327 //! N = (int) frf*O*R/fref
328 //! max bit count/resolution: 41b-23b = 18b
329 //!
330 //! step 6: calculate NUM (18b)
331 //! - NUM = (int) [262144 * (frf*O*R/fref -N)]
332 //! - NUM = (int) [2^18 * (frf*O*R/fref -N)]
333 //!
334 //! to get the correct value the above calculation are performed like this
335 //! - NUM =((int)((frf*O*R/fref)<<23) - N<<23 + rounding) >> 5, where << & >> are bit shift left, right
336 //! - note 1: << 23b was chosen, because step one use 41b, 41b +23b = 64b (max number for 64 bit math)
337 //! - note 2: >> 5 was chosen because 23-5 = 2^18, which is the same as multiplying by 262144
338 //! - note 3: the rounding variable is equivalent to adding 0.5 to the NUM number. the integer operation always
339 //! rounds down, so adding a 0.5 to the number before rounding down ensures the closest value is selected.
340 //! In this function the lower 5 bits of (frf*O*R/fref)<<23) - N<<23 (possible variable 0-31)
341 //! are check to see if they are >=16.
342 //! If they are then 16 is added to this number before performing the int and >>5 functions.
343 //! @return void
344 void LTC6948_set_frf(char part_version[] //!< LTC6948 part number LTC6948-x
345  );
