83 void
84 LT_2978FaultLog::read(uint8_t address)
85 {
86  uint16_t size = sizeof(struct LT_2978FaultLog::FaultLogLtc2978);
87  uint8_t *data = (uint8_t *) malloc(size);
88  if (data == 0)
89  Serial.print(F("bad malloc."));
90  data[0] = 0x00;
91 
92 
93  // Copy to RAM
94  pmbus_->smbus()->sendByte(address, MFR_FAULT_LOG_RESTORE);
95  // MFR_FAULT_LOG_RESTORE requires a delay (see datasheet), typical delay is 2ms, multiplying by 10 for safety
96  delay(20);
97  // Read block data with log
98  pmbus_->smbus()->readBlock(address, MFR_FAULT_LOG, data, 255);
99 
100  struct LT_2978FaultLog::FaultLogLtc2978 *log = (struct LT_2978FaultLog::FaultLogLtc2978 *)data;
101 
102  log->firstValidByte = 47;
103 
104  log->lastValidByte = 237;
105 
106  uint8_t cycle_start = log->preamble.position_last;
107 
108  log->loops = (LT_2978FaultLog::FaultLogReadLoopLtc2978 *) (log->telemetryData - 39 + cycle_start);
109 
110  faultLog2978 = log;
111 }
