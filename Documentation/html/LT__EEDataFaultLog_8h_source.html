68  void getNvmBlock(uint8_t address, uint16_t offset, uint16_t numWords, uint8_t command, uint8_t *data)
69  {
70  bool pecMatch = true;
71 
72  // Tell the device to get ready and set the index at the beginning of EEPROM
73  while ((pmbus_->smbus()->readByte(address, 0xEF) & 0x40) != 0x40); // Wait until not busy
74  pmbus_->smbus()->writeByte(address, 0xBD, 0x2B);
75  while ((pmbus_->smbus()->readByte(address, 0xEF) & 0x40) != 0x40); // Wait until not busy
76  pmbus_->smbus()->writeByte(address, 0xBD, 0x91);
77  while ((pmbus_->smbus()->readByte(address, 0xEF) & 0x40) != 0x40); // Wait until not busy
78  pmbus_->smbus()->writeByte(address, 0xBD, 0xE4);
79  // Read the ID
80  while ((pmbus_->smbus()->readByte(address, 0xEF) & 0x40) != 0x40); // Wait until not busy
81  pmbus_->smbus()->readWord(address, 0xBF);
82  // Read the size and ingore it because we know it will not change
83  while ((pmbus_->smbus()->readByte(address, 0xEF) & 0x40) != 0x40); // Wait until not busy
84  pmbus_->smbus()->readWord(address, 0xBF);
85  // Consume words we need to ignore
86  for (uint16_t i = 0; i < offset; i++)
87  {
88  while ((pmbus_->smbus()->readByte(address, 0xEF) & 0x40) != 0x40); // Wait until not busy
89  pmbus_->smbus()->readWord(address, 0xBF);
90  }
91  // Consume words of the fault log and check the CRC
92  int pos = 0;
93  pmbus_->smbus()->pecClear();
94  for (uint16_t i = 0; i < numWords; i++)
95  {
96  while ((pmbus_->smbus()->readByte(address, 0xEF) & 0x40) != 0x40); // Wait until not busy
97  uint16_t w = pmbus_->smbus()->readWord(address, 0xBF);
98  // Toss CRC every 32 bytes
99  if ((i + 1) % 16 == 0)
100  {
101  data[pos] = 0xFF & w; // If this is not obvious, consider the endianess. This is the 32nd byte.
102  pmbus_->smbus()->pecAdd(data[pos]);
103  pos++;
104  uint8_t calcPec = pmbus_->smbus()->pecGet();
105  uint8_t devPec = 0xFF & (w >> 8); // This depends on endianess as well.
106  pmbus_->smbus()->pecClear();
107  if (calcPec != devPec)
108  pecMatch = false;
109  }
110  else // And endianess applies here too.
111  {
112  data[pos] = 0xFF & w;
113  pmbus_->smbus()->pecAdd(data[pos]);
114  pos++;
115  data[pos] = 0xFF & (w >> 8);
116  pmbus_->smbus()->pecAdd(data[pos]);
117  pos++;
118  }
119  }
120  // It is possible to have a partial good/bad CRC. The device will not load any Fault Log data
121  // to RAM if there is ANY CRC mismatch. Then the read Fault Log will return all zero data or
122  // random data depending on revision of silicon. This code mimics the most recent silicon
123  // and returns all zeros if there is a CRC mismatch.
124  if (!pecMatch)
125  memset (data, 0x00, numWords*2);
126 
127  pmbus_->smbus()->writeByte(address, 0xBD, 0x00);
128  while ((pmbus_->smbus()->readByte(address, 0xEF) & 0x40) != 0x40); // Wait until not busy
129  }
