133 // Read a 16-bit word of data from register specified by "command"
134 int8_t LT_I2CBus::readWordData(uint8_t address, uint8_t command, uint16_t *value)
135 {
136  int8_t ret = 1;
137 
138  LT_Wire.beginTransmission(address);
139  LT_Wire.write(command);
140  ret = LT_Wire.endTransmission(false);
141  LT_Wire.beginTransmission(address);
142  uint8_t tempHolder[2];
143  LT_Wire.requestFrom(address, tempHolder, (uint16_t)2);
144  *value = tempHolder[0] << 8;

147  return ret;

189 // Write a block of data, starting at register specified by "command" and ending at (command + length - 1)
190 int8_t LT_I2CBus::writeBlockData(uint8_t address, uint8_t command, uint16_t length, uint8_t *values)
191 {
192  uint8_t i = length;
193  int8_t ret = 1;
194 
195  LT_Wire.beginTransmission(address);
196  LT_Wire.write(command);
197  do
198  {
199  i--;
200  }
201  while (LT_Wire.write(values[length - 1 - i]) == 1 && i > 0);
202 
203  ret = LT_Wire.endTransmission(!inGroupProtocol_);
204 
205  return ret;
206 }
