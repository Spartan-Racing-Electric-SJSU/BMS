300 // LT I2C functions to emulate Linux SMBus functions for E-Z porting. There does not appear to be
301 // a single source of the "correct" way to implement these functions.
302 
303 ///////////////////////////////////////////////////////////////////////////////////////////////////////////////
304 //// From http://git.kernel.org/cgit/linux/kernel/git/torvalds/linux.git/tree/Documentation/i2c/dev-interface
305 ///////////////////////////////////////////////////////////////////////////////////////////////////////////////
306 //// You can do plain I2C transactions by using read(2) and write(2) calls.
307 //// You do not need to pass the address byte; instead, set it through
308 //// ioctl I2C_SLAVE before you try to access the device.
309 ////
310 //// You can do SMBus level transactions (see documentation file smbus-protocol
311 //// for details) through the following functions:
312 //// __s32 i2c_smbus_write_quick(int file, __u8 value);
313 //// __s32 i2c_smbus_read_byte(int file);
314 //// __s32 i2c_smbus_write_byte(int file, __u8 value);
315 //// __s32 i2c_smbus_read_byte_data(int file, __u8 command);
316 //// __s32 i2c_smbus_write_byte_data(int file, __u8 command, __u8 value);
317 //// __s32 i2c_smbus_read_word_data(int file, __u8 command);
318 //// __s32 i2c_smbus_write_word_data(int file, __u8 command, __u16 value);
319 //// __s32 i2c_smbus_process_call(int file, __u8 command, __u16 value);
320 //// __s32 i2c_smbus_read_block_data(int file, __u8 command, __u8 *values);
321 //// __s32 i2c_smbus_write_block_data(int file, __u8 command, __u8 length,
322 //// // __u8 *values);
323 //// All these transactions return -1 on failure; you can read errno to see
324 //// what happened. The 'write' transactions return 0 on success; the
325 //// 'read' transactions return the read value, except for read_block, which
326 //// returns the number of values read. The block buffers need not be longer
327 //// than 32 bytes.
328 ///////////////////////////////////////////////////////////////////////////////////////////////////////////////
