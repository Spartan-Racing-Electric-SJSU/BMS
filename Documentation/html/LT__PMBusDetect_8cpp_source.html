96  LT_PMBusDevice *device;

114 
115  // May be more than required. Can add code to trim based on deviceCnt_

117  devices_ = (LT_PMBusDevice **) calloc(strlen((char *)addresses) + 1, sizeof(LT_PMBusDevice *));
118 
119  // Check each device type one by one. Trading generality and composability

121  // and then call detect on the correct device first time.
122  for (i = 0; i < strlen((char *)addresses); i++)
123  {
124  if ((device = LT_PMBusDeviceLTC3880::detect(pmbus_, addresses[i])) != NULL)

151  devices_[deviceCnt_++] = device;
152 
153  }
154 
155  // Get all the rails, while merging duplicates.
156  for (i = 0; i < deviceCnt_; i++)
157  {
158  LT_PMBusRail **new_rail;
159  void *m;
160  bool merged;
161 
162  new_rail = devices_[i]->getRails();
163 
164  // Loop if device has rail support, and if there are rails to process.
165  while (new_rail != NULL && *new_rail != NULL)
166  {
167  // See if we are already in the list.
168  merged = false;
169  for (j = 0; j < railCnt_; j++)
170  {
171  if (rails_[j]->isMultiphase() && rails_[j]->getAddress() == (*new_rail)->getAddress())
172  {
173  rails_[j]->merge(*new_rail);
174  delete (*new_rail);
175  merged = true;
176  break;
177  }
178  }
179  if (!merged)
180  {
181  if (rails_ == NULL)
182  rails_ = (LT_PMBusRail **) malloc(sizeof(LT_PMBusRail *));
183  else
184  {
185  m = realloc(rails_, (railCnt_ + 1) * sizeof(LT_PMBusRail *));
186  if (m == NULL)
187  free(m);
188  else
189  rails_ = (LT_PMBusRail **) m;
190  }
191  if (rails_ != NULL)
192  rails_[railCnt_++] = *new_rail;
193  }
194 
195  new_rail++;
196  }
197  }
198

200  rails_[railCnt_] = NULL;
