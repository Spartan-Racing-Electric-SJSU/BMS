109  return LTC3882;

111  return LTC3882_1;
112  case 0x4300:
113  return LTC3883;
114  case 0x4600:
115  return LTC3886;
116  case 0x47A0:
117  return LTM4675;
118  case 0x4400:
119  return LTM4676;
120  case 0x4480:
121  return LTM4676;
122  case 0x47E0:
123  return LTM4676A;
124  case 0x47B0:
125  return LTM4677;
126  case 0x0210:
127  return LTC2974;
128  case 0x0220:
129  return LTC2975;
130  case 0x0130:
131  return LTC2977;
132  case 0x8030:
133  return LTC2980;
134  case 0x8040:
135  return LTC2980;

322  }
