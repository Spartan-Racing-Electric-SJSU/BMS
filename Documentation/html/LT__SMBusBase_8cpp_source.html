348 void LT_SMBusBase::writeBlock(uint8_t address, uint8_t command,
349  uint8_t *block, uint16_t block_size)
350 {
351  if (pec_enabled_)
352  {
353  uint16_t pos = 0;
354 
355  pecClear();
356  pecAdd(address << 1);
357  pecAdd(command);
358  pecAdd(block_size);
359 
360  while (pos < block_size)
361  pecAdd(block[pos++]);
362  uint8_t pec = pecGet();
363 
364  uint8_t *data_with_pec = (uint8_t *) malloc(block_size + 2);
365  data_with_pec[0] = block_size;
366  memcpy(data_with_pec + 1, block, block_size);
367  data_with_pec[block_size + 1] = pec;
368 
369  if (i2cbus_->writeBlockData(address, command, block_size + 2, data_with_pec))
370  Serial.print(F("Write Block With Pec: fail.\n"));
371  free(data_with_pec);
372  }
373  else
374  {
375  uint8_t *buffer = (uint8_t *)malloc(block_size + 1);
376  buffer[0] = block_size;
377  memcpy(buffer + 1, block, block_size);
378  if (i2cbus_->writeBlockData(address, command, block_size + 1, buffer))
379  Serial.print(F("Write Block: fail.\n"));
380  free(buffer);
381  }

399 
400  uint8_t *buffer = (uint8_t *)malloc(block_out_size + 1);
401  buffer[0] = block_out_size;
402  memcpy(buffer + 1, block_out, block_out_size);
403 
404  i2cbus_->startGroupProtocol();
405  if (i2cbus_->writeBlockData(address, command, block_out_size + 1, buffer))
406  Serial.print(F("Write/Read Block w/PEC: write fail\n"));
407  free(buffer);
408 
409 
410

412 
413  i2cbus_->endGroupProtocol();

416  Serial.print(F("Write/Read Block w/PEC: read fail.\n"));

420  }

422 
423  for (pos = 0; pos<buffer[0] + 1u; pos++)
424  pecAdd(buffer[pos]);
425  if (pecGet() != buffer[buffer[0]+1])
