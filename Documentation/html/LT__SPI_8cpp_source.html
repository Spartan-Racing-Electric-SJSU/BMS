18 1 0 sample falling setup rising
19 1 1 sample rising setup rising

99 // Reads and sends a word
100 // Return 0 if successful, 1 if failed
101 void spi_transfer_word(uint8_t cs_pin, uint16_t tx, uint16_t *rx)
102 {
103  union
104  {
105  uint8_t b[2];
106  uint16_t w;
107  } data_tx;
108 
109  union
110  {
111  uint8_t b[2];
112  uint16_t w;
113  } data_rx;
114 
115  data_tx.w = tx;
116 
117  output_low(cs_pin); //! 1) Pull CS low
118 
119  data_rx.b[1] = SPI.transfer(data_tx.b[1]); //! 2) Read MSB and send MSB
120  data_rx.b[0] = SPI.transfer(data_tx.b[0]); //! 3) Read LSB and send LSB
121 
122  *rx = data_rx.w;
123 
124  output_high(cs_pin); //! 4) Pull CS high
125 }

127 // Reads and sends a byte array
128 void spi_transfer_block(uint8_t cs_pin, uint8_t *tx, uint8_t *rx, uint8_t length)
129 {
130  int8_t i;
131 
132  output_low(cs_pin); //! 1) Pull CS low
133 
134  for (i=(length-1); i >= 0; i--)
135  rx[i] = SPI.transfer(tx[i]); //! 2) Read and send byte array
136 
137  output_high(cs_pin); //! 3) Pull CS high
138 }

162 void spi_enable(uint8_t spi_clock_divider) // Configures SCK frequency. Use constant defined in header file.

167  SPI.begin();

177 // Write a data byte using the SPI hardware
178 void spi_write(int8_t data) // Byte to be written to SPI port
179 {
180  SPDR = data; //! 1) Start the SPI transfer
181  while (!(SPSR & _BV(SPIF))); //! 2) Wait until transfer complete
182 }

184 // Read and write a data byte using the SPI hardware
185 // Returns the data byte read
186 int8_t spi_read(int8_t data) //!The data byte to be written
187 {
188  SPDR = data; //! 1) Start the SPI transfer
189  while (!(SPSR & _BV(SPIF))); //! 2) Wait until transfer complete
190  return SPDR; //! 3) Return the data read
191 }
