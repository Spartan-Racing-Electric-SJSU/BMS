175 //! Initializes Pin 9 and 10 for 16-bit PWM
176 void init_16_bit_PWM()
177 {
178  pinMode(DAC_A, OUTPUT); // Set Pin 9 as output
179  pinMode(DAC_B, OUTPUT); // Set Pin 10 as output
180 
181 
182  // Enable timer1 for PWM, Phase Correct mode for pin 9 and 10.
183  // Use IRC1 as top, no prescalar (~122Hz),
184  // Negative Edge Triggered
185  TCCR1A = _BV(COM1A1) | _BV(COM1B1) | _BV(WGM11);
186  TCCR1B = _BV(WGM13) | _BV(CS10);
187  TCCR1C = 0x0;
188  ICR1 = 0xFFFF; // Set Timer for 16-bit
189 
190  // The following code should be uncommented if Fast PWM Mode
191  // is desired
192  /*
193  // Enable timer1 for Fast PWM mode for pin 9 and 10.
194  // Use IRC1 as top, no prescalar (~244Hz),
195  // Negative Edge Triggered
196  TCCR1A = _BV(COM1A1) | _BV(COM1B1) | _BV(WGM11);
197  TCCR1B = _BV(WGM13)| _BV(WGM12) | _BV(CS10);
198  TCCR1C = 0x0;
199  ICR1 = 0xFFFF; // Set Timer for 16-bit
200  */
201 
202  // Set to 0% duty cycle
203  OCR1A = 0x0;
204  OCR1B = 0x0;
205 }
