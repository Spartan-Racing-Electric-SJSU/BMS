76 // Read the id string from the EEPROM, then parse the
77 // product name, demo board name, and demo board option
78 // from the id string into the global demo_board variable.
79 // Returns the number of characters read from the information string.
80 uint8_t read_quikeval_id_string(char *buffer)
81 {
82  uint8_t i, j; // Iteration variables
83  uint8_t comma_position[8] = {0}; // Contains the position of the commas in the buffer
84  uint8_t comma_count = 0; // The number of commas found in the buffer
85  uint8_t buffer_count; // The number of characters read
86  int8_t option; // Temporary demo board option
87  // read the id string from the demo board EEPROM
88  // starting EEPROM address=0, terminator=0x0a, max buffer length=52 bytes
89  // Enable I2C
90  pinMode(QUIKEVAL_MUX_MODE_PIN, OUTPUT); // Configure MUX to disconnect I2C pins on QuikEval connector. Remember EEPROM has its own I2C pins on the the connector.
91  int8_t QUIKEVAL_MUX_MODE_PIN_state;
92  QUIKEVAL_MUX_MODE_PIN_state = digitalRead(QUIKEVAL_MUX_MODE_PIN);
93  digitalWrite(QUIKEVAL_MUX_MODE_PIN, LOW);
94  buffer_count = eeprom_read_buffer_with_terminator(EEPROM_I2C_ADDRESS, buffer, 0, QUIKEVAL_ID_TERMINATOR, QUIKEVAL_ID_SIZE+2);
95  digitalWrite(QUIKEVAL_MUX_MODE_PIN, QUIKEVAL_MUX_MODE_PIN_state); // Restore the QUIKEVAL_MUX_MODE_PIN (8) to its previous state. (But, it will be left as output.)
96  if (buffer_count == 0) return(0); // quit if no data read
97  // find comma positions
98  for (i = 0; i < buffer_count; i++)
99  {
100  if (buffer[i] == ',') comma_position[comma_count++]=i;
101  }
102 
103  if (comma_position[6] < comma_position[5])// comma_position[6]=strlen(buffer); // some demo boards are missing the last comma
104  {
105  for (i = buffer_count - 2; i > comma_position[5]; i--)
106  {
107  if (buffer[i] != '-')
108  {
109  comma_position[6] = i+1;
110  break;
111  }
112  }
113  }
114 
115 
116 
117  // get product name. All characters before the 1st comma.
118  for (i = 0; i < comma_position[0]; i++)
119  {
120  demo_board.product_name[i]=buffer[i];
121  }
122  demo_board.product_name[i]='\0'; // add terminator
123  // Get demo board name between the 5th and 6th comma
124  j = 0;
125 
126  for (i = comma_position[5]+1; i < comma_position[6]; i++)
127  {
128  demo_board.name[j++]=buffer[i];
129  }
130  demo_board.name[j]='\0'; // add terminator
131  // get demo board option from the demo board name
132  for (i = 0; i < strlen(demo_board.name); i++)
133  {
134  if (demo_board.name[i] == '-') option = demo_board.name[i+1];
135  }
136  if ((option >= 0x41) && (option <= 0x5A)) demo_board.option = option;
137  // final demo board name is 6 characters without the option
138  demo_board.name[6]='\0';
139  return(buffer_count);
140 }

178 // Determine if the EEPROM is ready for communication by writing
179 // the address+!write byte and looking for an acknowledge. This is
180 // repeated every 1ms until an acknowledge occurs, or a timeout occurs.
181 // If a timeout occurs, an I2C stop is generated.
182 // Returns 0 if an acknowledge is generated and 1 if not.
183 int8_t eeprom_poll(uint8_t i2c_address)
184 {
185  uint8_t timer_count;
186  int8_t ack;
187  i2c_start(); // I2C start
188  ack = i2c_write(i2c_address | I2C_WRITE_BIT); // I2C address + !write
189  if (ack != 0)
190  {
191  // acknowledge not received, enter timer loop
192  for (timer_count = 0; timer_count < EEPROM_TIMEOUT; timer_count++)
193  {
194  i2c_repeated_start(); // I2C repeated start
195  ack = i2c_write(i2c_address | I2C_WRITE_BIT); // I2C address + !write
196  if (ack == 0)
197  break; // If the EEPROM acknowledges, jump out of delay loop
198  else
199  delay(1);
200  }
201  if (ack) i2c_stop(); // Generate a stop if no acknowledge
202  }
203  return(ack);
204 }

215 // Write the data byte to the EEPROM with i2c_address starting at EEPROM address.
216 // Returns the total number of bytes written
217 uint8_t eeprom_write_byte(uint8_t i2c_address, char data, uint16_t address)
218 {
219  uint8_t byte_count = 0;
220  if (address < EEPROM_DATA_SIZE) // Make sure the address is in bounds
221  {
222  if (!eeprom_poll(i2c_address)) // Check if EEPROM is ready
223  {
224  if (EEPROM_DATA_SIZE > 0x100)
225  i2c_write(address>>8); // Send upper byte of address if size > 256 bytes
226  i2c_write(address); // Send lower byte of address
227  i2c_write(data); // Send byte to EEPROM
228  i2c_stop(); // I2C stop
229  byte_count = 1;
230  }
231  }
232  return(byte_count);
233 }

247 // Write the buffer to the EEPROM with i2c_address starting at EEPROM address in blocks of EEPROM_PAGE_SIZE bytes.
248 // Returns the total number of byte written
249 uint8_t eeprom_write_buffer(uint8_t i2c_address, char *buffer, uint16_t address)
250 {
251  uint8_t i, j;
252  uint8_t index = 0;
253  uint8_t page_count;
254  uint8_t byte_count;
255  page_count=((strlen(buffer)-1)/EEPROM_PAGE_SIZE)+1; // Calculate page count
256  if (address < EEPROM_DATA_SIZE) // Make sure the address is in bounds
257  {
258  for (i = 0; i < page_count; i++)
259  {
260  if (!eeprom_poll(i2c_address)) // Check if EEPROM is ready
261  {
262  if (EEPROM_DATA_SIZE > 0x100)
263  i2c_write(address>>8); // Send upper byte of address if size > 256 bytes
264  i2c_write(address); // Send lower byte of address
265  byte_count = strlen(buffer)-index; // Calculate the remaining byte count
266  if (byte_count > EEPROM_PAGE_SIZE) // Limit to EEPROM_PAGE_SIZE
267  byte_count = EEPROM_PAGE_SIZE;
268  for (j = 0; j < byte_count; j++)
269  {
270  if (address >= EEPROM_DATA_SIZE) break; // Quit if the address is at the end
271  i2c_write(buffer[index++]); // Write byte to EEPROM
272  address++; // Increment the EEPROM address
273  }
274  i2c_stop(); // The stop bit starts the write process
275  }
276  else
277  break;
278  }
279  }
280  return(index); // Return the number of bytes written
281 }

317 // Read data bytes from the EEPROM starting at address until number bytes read equals count. A null terminator is
318 // added to the end of the buffer.
319 // Returns the number of bytes read.
320 uint8_t eeprom_read_buffer(uint8_t i2c_address, char *buffer, uint16_t address, uint8_t count)
321 {
322  uint8_t i = 0;
323  uint8_t data;
324  *buffer='\0'; // Initialize buffer with null
325  if (!eeprom_poll(i2c_address)) // Check if the EEPROM is ready
326  {
327  if (EEPROM_DATA_SIZE > 0x100)
328  i2c_write(address>>8); // Send upper byte of address if size > 256 bytes
329  i2c_write(address); // Send lower byte of address
330  i2c_repeated_start(); // I2C repeated start
331  i2c_write(i2c_address | I2C_READ_BIT); // I2C address + read
332  for (i = 0; i < count-1; i++) // Read count-1 bytes with ACK
333  {
334  data = i2c_read(WITH_ACK); // Receive 1 EEPROM data byte with ACK
335  *buffer++=data; // Place data byte in buffer and increment pointer
336  }
337  data = i2c_read(WITH_NACK); // Read last byte from EEPROM with NACK
338  *buffer++=data;
339  i2c_stop(); // I2C stop
340  *buffer='\0'; // Place null terminator at end of buffer
341  return(i+1);
342  }
343  else
344  return(0);
345 }

377 // Write the 2 byte integer data to the EEPROM starting at address. Use the eeprom_write_byte
378 // routine to avoid keeping track of page boundaries with the eeprom_write_buffer routine.
379 // Returns the total number of bytes written.
380 uint8_t eeprom_write_int16(uint8_t i2c_address, int16_t write_data, uint16_t address)
381 {
382  union
383  {
384  int16_t a;
385  int8_t b[2];
386  } data;
387  uint8_t byte_count;
388  data.a = write_data; // get the data
389  byte_count = eeprom_write_byte(i2c_address, data.b[0], address++); // write the LSB to EEPROM
390  if (eeprom_write_poll(i2c_address)) return(0); // poll EEPROM until complete
391  byte_count+=eeprom_write_byte(i2c_address, data.b[1], address); // write the MSB to EEPROM
392  if (eeprom_write_poll(i2c_address)) return(1); // poll EEPROM until complete
393  return(byte_count);
394 }
