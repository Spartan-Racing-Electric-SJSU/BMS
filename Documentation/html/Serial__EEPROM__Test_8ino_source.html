148 void loop()
149 {
150  unsigned char user_command;
151  char data;
152  byte byte_count;
153  if (Serial.available())
154  {
155  user_command = read_int();
156  Serial.println(user_command);
157  switch (user_command)
158  {
159  case 0:
160  // buffer read
161  byte_count = eeprom_read_buffer(i2c_address, &ui_buffer[0], eeprom_address, read_count);
162  if (byte_count != 0)
163  {
164  Serial.print("Data Received: ");
165  Serial.println(ui_buffer);
166  }
167  else
168  Serial.println("Error: No Data Received");
169  break;
170  case 1:
171  // buffer read with terminator
172  byte_count = eeprom_read_buffer_with_terminator(i2c_address, &ui_buffer[0], eeprom_address, terminator, read_count);
173  if (byte_count != 0)
174  {
175  Serial.print("Data Received: ");
176  Serial.println(ui_buffer);
177  }
178  else
179  Serial.println("Error: No Data Received");
180  break;
181  case 2:
182  // byte read
183  byte_count = eeprom_read_byte(i2c_address, &data, eeprom_address);
184  Serial.println("test");
185  if (byte_count != 0)
186  {
187  Serial.print("Byte Received: 0x");
188  Serial.print(data, HEX);
189  Serial.print("(");
190  Serial.print(char(ui_buffer[0]));
191  Serial.println(")");
192  }
193  else
194  Serial.println("Error: No Data Received");
195  break;
196  case 3:
197  // buffer write
198  Serial.print("Enter string to Write to EEPROM: ");
199  read_string();
200  byte_count = eeprom_write_buffer(i2c_address, &ui_buffer[0], eeprom_address);
201  if (byte_count == strlen(ui_buffer))
202  {
203  Serial.println(ui_buffer);
204  }
205  else
206  Serial.println("Data Write Error");
207  break;
208  case 4:
209  // byte write
210  Serial.print("Enter Data Byte to Write to EEPROM: ");
211  data = read_int();
212  byte_count = eeprom_write_byte(i2c_address, data, eeprom_address);
213  if (byte_count != 0)
214  {
215  Serial.print("0x");
216  Serial.println(data, HEX);
217  }
218  else
219  Serial.println("Data Write Error");
220  break;
221  case 5:
222  // I2C Address
223  Serial.print("Enter I2C Address: ");
224  i2c_address = read_int();
225  Serial.println(i2c_address, HEX);
226  break;
227  case 6:
228  // EEPROM Address
229  Serial.print("Enter EEPROM Address: ");
230  eeprom_address = read_int();
231  if (eeprom_address >= EEPROM_DATA_SIZE) eeprom_address = EEPROM_DATA_SIZE-1;
232  Serial.println(eeprom_address, HEX);
233  break;
234  case 7:
235  // Terminator
236  Serial.print("Enter Terminator: ");
237  terminator = read_int();
238  Serial.println(terminator, HEX);
239  break;
240  case 8:
241  // Read Count
242  Serial.print("Enter Read Count: ");
243  read_count = read_int();
244  Serial.println(read_count);
245  break;
246  case 9:
247  // Write Integer
248  Serial.print("Enter Integer To Write (-32,768 to 32,767): ");
249  int_data = read_int();
250  byte_count = eeprom_write_int16(i2c_address, int_data, eeprom_address);
251  if (byte_count != 0)
252  {
253  Serial.println(int_data);
254  }
255  else
256  Serial.println("Data Write Error");
257  break;
258  case 10:
259  // Read Integer from EEPROM
260  byte_count = eeprom_read_int16(i2c_address, &int_data, eeprom_address);
261  if (byte_count != 0)
262  {
263  Serial.print("Integer Read From EEPROM:");
264  Serial.println(int_data);
265  }
266  else
267  Serial.println("Data Read Error");
268  break;
269  case 11:
270  // Write Long
271  Serial.print("Enter Long To Write (-2,147,483,648 to 2,147,483,647): ");
272  long_data = read_int();
273  byte_count = eeprom_write_int32(i2c_address, long_data, eeprom_address);
274  if (byte_count != 0)
275  {
276  Serial.println(long_data);
277  }
278  else
279  Serial.println("Data Write Error");
280  break;
281  case 12:
282  // Read Long from EEPROM
283  byte_count = eeprom_read_int32(i2c_address, &long_data, eeprom_address);
284  if (byte_count != 0)
285  {
286  Serial.print("Long Read From EEPROM:");
287  Serial.println(long_data);
288  }
289  else
290  Serial.println("Data Read Error");
291  break;
292  case 13:
293  // Write Float
294  Serial.print("Enter Float To Write: ");
295  float_data = read_float();
296  byte_count = eeprom_write_float(i2c_address, float_data, eeprom_address);
297  if (byte_count != 0)
298  {
299  Serial.println(float_data, 6);
300  }
301  else
302  Serial.println("Data Write Error");
303  break;
304  case 14:
305  // Read Float from EEPROM
306  byte_count = eeprom_read_float(i2c_address, &float_data, eeprom_address);
307  if (byte_count != 0)
308  {
309  Serial.print("Float Read From EEPROM: ");
310  Serial.println(float_data, 6);
311  }
312  else
313  Serial.println("Data Read Error");
314  break;
315  case 15:
316  // Clear Calibration Settings That are stored on the Demo boards EEPROM
317  Serial.println("Do You Want to Clear Calibration on the Demo Board?");
318  Serial.println("1-Clear Calibration\n");
319  Serial.println("0-Keep Calibration\n");
320  Serial.print("Enter a Command: ");
321  user_command = read_int();
322 
323  if (user_command == 1)
324  {
325  // Clear ALL
326  for (int i = 0; i < 0x7FF; i++)
327  eeprom_write_byte(EEPROM_I2C_ADDRESS, 0x0FF, (EEPROM_CAL_STATUS_ADDRESS + i));
328  Serial.print("Calibration Cleared\n");
329  }
330  else
331  {
332  Serial.print("Calibration NOT Cleared\n");
333  }
334  }
335  print_prompt();
336  }
