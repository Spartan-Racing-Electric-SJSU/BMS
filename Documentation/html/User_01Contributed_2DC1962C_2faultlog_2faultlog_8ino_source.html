108 //! Repeats Linduino loop
109 //! @return void
110 void loop()
111 {
112  uint8_t user_command;
113  uint8_t *addresses = NULL;
114  struct LT_3880FaultLog::FaultLogLtc3880 *ltc3880_fault_log;
115  struct LT_2977FaultLog::FaultLogLtc2977 *ltc2977_fault_log;
116  struct LT_2974FaultLog::FaultLogLtc2974 *ltc2974_fault_log;
117  uint16_t timeout;
118 
119  if (Serial.available()) //! Checks for user input
120  {
121  while ((user_command = read_int()) == 0); //! Reads the user command
122  if (user_command != 'm')
123  Serial.println(user_command);
124 
125  switch (user_command) //! Prints the appropriate submenu
126  {
127  case 1:
128  Serial.println();
129  if (faultLog3880->hasFaultLog(ltc3880_i2c_address))
130  {
131  faultLog3880->read(ltc3880_i2c_address);
132  faultLog3880->print(&Serial);
133  faultLog3880->release();
134  }
135  else
136  Serial.println(F("No LTC3880 Fault Log"));
137 
138  Serial.println();
139  if (faultLog2974->hasFaultLog(ltc2974_i2c_address))
140  {
141  faultLog2974->read(ltc2974_i2c_address);
142  faultLog2974->print(&Serial);
143  faultLog2974->release();
144  }
145  else
146  Serial.println(F("No LTC2974 Fault Log"));
147 
148  Serial.println();
149  if (faultLog2977->hasFaultLog(ltc2977_i2c_address))
150  {
151  faultLog2977->read(ltc2977_i2c_address);
152  faultLog2977->print(&Serial);
153  faultLog2977->release();
154  }
155  else
156  Serial.println(F("No LTC2977 Fault Log"));
157 
158  break;
159  case 2:
160  faultLog3880->clearFaultLog(ltc3880_i2c_address);
161  delay(2);
162  smbus->waitForAck(ltc3880_i2c_address, 0x00);
163  pmbus->waitForNotBusy(ltc3880_i2c_address);
164 
165  timeout = 4096;
166  while (timeout-- > 0)
167  {
168  if (0 == pmbus->readMfrStatusByte(ltc3880_i2c_address) & ~(1 << 3))
169  break;
170  }
171 
172  faultLog2974->clearFaultLog(ltc2974_i2c_address);
173  delay(2);
174  pmbus->waitForNotBusy(ltc2974_i2c_address);
175  faultLog2977->clearFaultLog(ltc2977_i2c_address);
176  delay(2);
177  pmbus->waitForNotBusy(ltc2977_i2c_address);
178  break;
179  case 3:
180  pmbus->clearAllFaults(ltc3880_i2c_address);
181  pmbus->clearAllFaults(ltc2974_i2c_address);
182  pmbus->clearAllFaults(ltc2977_i2c_address);
183  break;
184  case 4:
185  addresses = smbus->probe(0);
186  while (*addresses != 0)
187  {
188  Serial.print(F("ADDR 0x"));
189  Serial.println(*addresses++, HEX);
190  }
191  break;
192  case 5:
193  pmbus->startGroupProtocol();
194  pmbus->reset(ltc3880_i2c_address);
195  pmbus->restoreFromNvm(ltc2974_i2c_address);
196  pmbus->restoreFromNvm(ltc2977_i2c_address);
197  pmbus->executeGroupProtocol();
198  break;
199  case 6:
200  smbus->sendByte(ltc3880_i2c_address, MFR_FAULT_LOG_STORE);
201  delay(2);
202  smbus->waitForAck(ltc3880_i2c_address, 0x00);
203  pmbus->waitForNotBusy(ltc3880_i2c_address);
204 
205  timeout = 4096;
206  while (timeout-- > 0)
207  {
208  if (0 != pmbus->readMfrStatusByte(ltc3880_i2c_address) & ~(1 << 3))
209  break;
210  }
211 
212  smbus->sendByte(ltc2974_i2c_address, MFR_FAULT_LOG_STORE);
213  delay(2);
214  pmbus->waitForNotBusy(ltc2974_i2c_address);
215 
216  smbus->sendByte(ltc2977_i2c_address, MFR_FAULT_LOG_STORE);
217  delay(2);
218  pmbus->waitForNotBusy(ltc2977_i2c_address);
219 
220  break;
221  default:
222  Serial.println(F("Incorrect Option"));
223  break;
224  }
225  print_prompt();
226  }
227 }
