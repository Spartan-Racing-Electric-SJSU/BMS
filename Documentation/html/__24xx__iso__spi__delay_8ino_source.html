137 //! Repeats Linduino loop
138 void loop()
139 {
140  uint8_t user_command; // The user input command
141  uint8_t ack = 0;
142  if (demo_board_connected)
143  {
144  if (Serial.available()) // Check for user input
145  {
146  user_command = read_int(); // Read the user command
147  if (user_command == 'm');
148  else
149  Serial.println(user_command);
150  delay(50); // Allow the print to finish
151  switch (user_command)
152  {
153  case 1:
154  ack |= menu_1_read_single_ended();
155  break;
156  case 2:
157  ack |= menu_2_read_differential();
158  break;
159  default:
160  Serial.println("Incorrect Option");
161  break;
162  }
163  if (ack)
164  Serial.println(F("***** SPI ERROR *****"));
165  Serial.println(F("*****************************************************************"));
166  print_prompt();
167  }
168  }
169 }

199 //! read from ADC single-ended
200 //! @return 0 if successful, 1 if failure
201 //! @return 0 when m is entered into menu, 1 if timeout for EOC
202 uint8_t menu_1_read_single_ended()
203 {
204  uint8_t adc_command; // The LTC24XX command byte
205  int16_t user_command; // The user input command
206  int32_t adc_code=0; // The LTC24XX code
207  float adc_voltage; // The LTC24XX voltage
208 
209  while (1)
210  {
211  Serial.print(F("*************************\n\n")); // Display single-ended menu
212  Serial.print(F("0-CH0 8-CH8\n"));
213  Serial.print(F("1-CH1 9-CH9\n"));
214  Serial.print(F("2-CH2 10-CH10\n"));
215  Serial.print(F("3-CH3 11-CH11\n"));
216  Serial.print(F("4-CH4 12-CH12\n"));
217  Serial.print(F("5-CH5 13-CH13\n"));
218  Serial.print(F("6-CH6 14-CH14\n"));
219  Serial.print(F("7-CH7 15-CH15\n"));
220  Serial.print(F("16-ALL\n"));
221  Serial.print(F("m-Main Menu\n"));
222  Serial.print(F("\nEnter a Command: "));
223  user_command = read_int(); // Read the single command
224  if (user_command == 'm')
225  return(0);
226  else
227  Serial.println(user_command);
228  Serial.println();
229 
230  if (user_command == 16)
231  {
232  Serial.print(F("ALL\n"));
233  adc_command = BUILD_COMMAND_SINGLE_ENDED[0]; // Build ADC command for channel 0
234  delay(delay_EOC); // Wait for EOC
235  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code); // Throws out last reading
236 
237  for (int8_t x = 0; x <= 15; x++)
238  {
239  adc_command = BUILD_COMMAND_SINGLE_ENDED[(x + 1) % 16]; // Read all channels in single-ended mode
240  delay(delay_EOC); // Wait for EOC
241  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code);
242  adc_voltage = LTC24XX_diff_code_to_voltage(adc_code, reference_voltage);
243  Serial.print(F(" ****"));
244  Serial.print(F("CH"));
245  Serial.print(x);
246  Serial.print(F(": "));
247  Serial.print(adc_voltage, 4);
248  Serial.print(F("V\n"));
249  }
250  }
251  else if (user_command<16 && user_command >=0) // Read selected channel
252  {
253  adc_command = BUILD_COMMAND_SINGLE_ENDED[user_command];
254  Serial.print(F("\nADC Command: h"));
255  Serial.println(adc_command, HEX);
256  delay(delay_EOC); // Wait for EOC
257  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code); // Throws out last reading
258  delay(delay_EOC); // wait for EOC
259  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code);
260  Serial.print(F("Received Code: 0x"));
261  Serial.println(adc_code, HEX);
262  adc_voltage = LTC24XX_diff_code_to_voltage(adc_code, reference_voltage);
263  Serial.print(F(" ****"));
264  Serial.print(F("CH"));
265  Serial.print(user_command);
266  Serial.print(F(": "));
267  Serial.print(adc_voltage, 4);
268  Serial.print(F("V\n"));
269  }
270  }
271 }

273 //! Read channels in differential mode
274 //! @return 0 when m is entered into menu, 1 if timeout for EOC
275 uint8_t menu_2_read_differential()
276 {
277  int8_t y; // Offset into differential channel array to select polarity
278  uint8_t adc_command; // The LTC24XX command byte
279  int16_t user_command; // The user input command
280  int32_t adc_code=1; // The LTC24XX code
281  float adc_voltage; // The LTC24XX voltage
282 
283  while (1)
284  {
285  // Display differential menu
286  Serial.print(F("\n*************************\n\n"));
287  Serial.print(F("0-0P-1N 8-1P-0N\n"));
288  Serial.print(F("1-2P-3N 9-3P-2N\n"));
289  Serial.print(F("2-4P-5N 10-5P-4N\n"));
290  Serial.print(F("3-6P-7N 11-7P-6N\n"));
291  Serial.print(F("4-8P-9N 12-9P-8N\n"));
292  Serial.print(F("5-10P-11N 13-11P-10N\n"));
293  Serial.print(F("6-12P_13N 14-13P-12N\n"));
294  Serial.print(F("7-14P-15N 15-15P-14N\n"));
295  Serial.print(F("16-ALL Even_P-Odd_N\n"));
296  Serial.print(F("17-ALL Odd_P-Even_N\n"));
297  Serial.print(F("m-Main Menu\n"));
298  Serial.print(F("\nEnter a Command: "));
299  user_command = read_int();
300  if (user_command == 'm')
301  return(0);
302  Serial.println(user_command);
303  Serial.println();
304 
305  if ((user_command == 16) || (user_command == 17))
306  {
307  if (user_command == 16)
308  {
309  Serial.print(F("ALL Even_P-Odd_N\n")); // Cycles through options 0-7
310  y = 0;
311  }
312  if (user_command == 17)
313  {
314  Serial.print(F("ALL Odd_P-Even_N\n")); // Cycles through options 8-15
315  y = 8;
316  }
317  adc_command = BUILD_COMMAND_DIFF[y]; // Set up first channel
318  delay(delay_EOC); // Wait for EOC
319  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code); // Throws out last reading
320  for (int8_t x = 0; x <= 7; x++) // Read all channels. All even channels are positive and odd channels are negative
321  {
322  adc_command = BUILD_COMMAND_DIFF[((x+1) % 8) + y];
323  delay(delay_EOC); // Wait for EOC
324  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code);
325  adc_voltage = LTC24XX_diff_code_to_voltage(adc_code, reference_voltage);
326  Serial.println();
327  Serial.print(F("**** "));
328  print_user_command(x + y);
329  Serial.print(F(": "));
330  Serial.print(adc_voltage, 4);
331  Serial.print(F("V\n"));
332  }
333  }
334  else // Read selected channels
335  {
336  // Reads and displays a selected channel
337  adc_command = BUILD_COMMAND_DIFF[user_command];
338  Serial.print(F("ADC Command: 0b"));
339  Serial.println(adc_command, BIN);
340  delay(delay_EOC); // Wait for EOC
341  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code); // Throws out last reading
342  delay(delay_EOC); // Wait for EOC
343  LTC24XX_SPI_16bit_command_32bit_data(LTC24XX_CS, adc_command, 0x00, &adc_code);
344  Serial.print(F("Received Code: 0x"));
345  Serial.println(adc_code, HEX);
346  adc_voltage = LTC24XX_diff_code_to_voltage(adc_code, reference_voltage);
347  Serial.println();
348  Serial.print(F("**** "));
349  Serial.print(F("CH"));
350  print_user_command(user_command);
351  Serial.print(F(": "));
352  Serial.print(adc_voltage, 4);
353  Serial.print(F("V"));
354  Serial.println();
355  }
356  }
357 }
