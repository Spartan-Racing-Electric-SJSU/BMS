97 //! Initialize Linduino
98 //! @return void
99 void setup()
100 {
101  uint32_t speed = 500000;
102 
103  Serial.begin(115200); //! Initialize the serial port to the PC
104 
105  if (doprint) Serial.println(F("Detecting Addresses..."));
106  detector->detect();
107  device = (devices = detector->getDevices());
108  while (*device != NULL)
109  {
110  if (doprint) Serial.print(F(" Type: "));
111  if (doprint) Serial.print((*device)->getType());
112  if (doprint) Serial.print(F(" Address: "));
113  if (doprint) Serial.print((*device)->getAddress(), HEX);
114 
115  if (doprint) Serial.print(F(", Speed: "));
116  if (doprint) Serial.print((*device)->getMaxSpeed(), DEC);
117  if (doprint) Serial.println();
118 
119  // Switch the device to Pec
120  pmbusNoPec->enablePec((*device)->getAddress());
121  (*device)->changePMBus(pmbusPec);
122 
123  // Speed test to find fastest speed the bus can operate at.
124  if (speed > (*device)->getMaxSpeed())
125  speed = (*device)->getMaxSpeed();
126 
127  device++;
128  }
129 
130  if (doprint) Serial.print("Changing bus speed to ");
131  if (doprint) Serial.println(speed, DEC);
132  // This will change the speed of the I2C bus and enable clock stretching if > 100kHz
133  // Only affects Pec becuase the device was switched to Pec.
134  device = devices;
135  while ((*device) != 0)
136  {
137  (*device)->setSpeed(speed);
138  device++;
139  }
140  // Change the NoPec version to keep out world consistent.
141  pmbusNoPec->smbus()->i2cbus()->changeSpeed(speed);
142 
143  Serial.println(F("Index,DeviceId,Channel,MfrSpecialId,Time,DeviceAddress,DevicePage,TelemetryCommand,TelemetryType,ValueLength,Value,FloatValue"));
144 
145 }
