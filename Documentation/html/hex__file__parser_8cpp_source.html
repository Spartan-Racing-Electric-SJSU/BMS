88 /*
89  * Parse a list of hex file bytes returning a list of ltc record bytes
90  *
91  * in_data: Function to get one line of data to parse.
92  * return: One byte of data
93  *
94  * Notes: CRC is ignored. No error processing; original data must be correct.
95  * Copies char input data so that it can be const, and frees when done.
96  * It is best to use const input so that memory usage is only temporary.
97  */
98 uint8_t parse_hex(uint8_t (*get_data)(void))
99 {
100  uint8_t start_code;
101  uint16_t byte_count;
102  uint16_t record_type;
103 // uint32_t address;
104 // uint16_t crc;
105  uint16_t i;
106  char data[5];
107 
108  if (parse_data_position == parse_data_length)
109  {
110  // Get colon
111  start_code = 0x00;
112  while (start_code != ':')
113  start_code = get_data();
114 
115  data[0] = get_data();
116  data[1] = get_data();
117  data[2] = '\0';
118  byte_count = httoi(data);
119 #if (DEBUG_PARSE_DATA)
120  Serial.print(F("Cnt: "));
121  Serial.println(byte_count, HEX);
122 #endif
123  if (byte_count > PARSE_DATA_LEN)
124  Serial.println(F("Parse_data size too samll"));
125 
126 // Serial.print(F("data size ")); Serial.println(byte_count, DEC);
127 
128  data[0] = get_data();
129  data[1] = get_data();
130  data[2] = get_data();
131  data[3] = get_data();
132  data[4] = '\0';
133 #if (DEBUG_PARSE_DATA)
134  int address = httoi(data);
135  Serial.print(F("Addr: "));
136  Serial.println(address, HEX);
137 #endif
138 
139  data[0] = get_data();
140  data[1] = get_data();
141  data[2] = '\0';
142  record_type = httoi(data);
143 #if (DEBUG_PARSE_DATA)
144  Serial.print(F("Rec: "));
145  Serial.println(record_type, HEX);
146 #endif
147  if (record_type == 0)
148  {
149  for (i = 0; i < byte_count; i++)
150  {
151  data[0] = get_data();
152  data[1] = get_data();
153  data[2] = '\0';
154  parse_data[i] = httoi(data);
155 #if (DEBUG_PARSE_DATA)
156  Serial.print(F("Parse: "));
157  Serial.print(data[0], HEX);
158  Serial.print(F(" "));
159  Serial.print(data[1], HEX);
160  Serial.print(F(" "));
161  Serial.println(parse_data[i], HEX);
162 #endif
163  }
164 
165  data[0] = get_data();
166  data[1] = get_data();
167  data[2] = '\0';
168 #if (DEBUG_PARSE_DATA)
169  int crc = httoi(data);
170  Serial.print(F("Crc: "));
171  Serial.println(crc, HEX);
172 #endif
173  parse_data_position = 0;
174  parse_data_length = byte_count;
175  }
176  else if (record_type == 1) // Make termination record
177  {
178  parse_data[0] = 4;
179  parse_data[1] = 0;
180  parse_data[2] = 0x22;
181  parse_data[3] = 0;
182  parse_data_position = 0;
183  parse_data_length = 4;
184  }
185  }
186 #if (DEBUG_PARSE_DATA)
187  Serial.print(F("Byte "));
188  Serial.println(parse_data[parse_data_position], HEX);
189 #endif
190  return parse_data[parse_data_position++];
191 }

193 uint16_t parse_records(uint8_t *in_data, uint16_t in_length, tRecordHeaderLengthAndType **out_records)
194 {
195  uint16_t in_position;
196  uint16_t out_position;
197  tRecordHeaderLengthAndType *record;
198 
199  in_position = 0;
200  out_position = 0;
201 
202  while (1)
203  {
204  record = (tRecordHeaderLengthAndType *) &in_data[in_position];
205  *(out_records+out_position) = record;
206  in_position += (*out_records[out_position]).Length;
207  out_position += 1;
208 
209  if (in_position >= in_length)
210  break;
211  }
212 
213  return out_position;
214 }

216 pRecordHeaderLengthAndType parse_record(uint8_t (*get_data)(void))
217 {
218  uint32_t header;
219  uint16_t size;
220  uint16_t pos;
221  pRecordHeaderLengthAndType record;
222 
223  header = (uint32_t)get_data() << 0 | (uint32_t)get_data() << 8 | (uint32_t)get_data() << 16 | (uint32_t)get_data() << 24;
224 
225  record = (pRecordHeaderLengthAndType) &header;
226  size = record->Length;
227 
228  if (size > getMaxRecordSize())
229  {
230  Serial.println(F("record_data size too samll"));
231  Serial.print(F("requires "));
232  Serial.print(size, DEC);
233  Serial.print(F(" has "));
234  Serial.println(getMaxRecordSize());
235  Serial.print(F("Record Type 0x"));
236  Serial.println(record->RecordType, HEX);
237  }
238 
239 // Serial.print(F("Record size ")); Serial.println(size, DEC);
240 
241  uint8_t *record_data = getRecordData();
242 
243  record_data[0] = (header >> 0) & 0xFF;
244  record_data[1] = (header >> 8) & 0xFF;
245  record_data[2] = (header >> 16) & 0xFF;
246  record_data[3] = (header >> 24) & 0xFF;
247 
248  if (size <= getMaxRecordSize())
249  {
250  for (pos = 0; pos < size - 4; pos++)
251  record_data[pos + 4] = get_data();
252  }
253  record = (pRecordHeaderLengthAndType) record_data;
254 
255  return record;
256 }
