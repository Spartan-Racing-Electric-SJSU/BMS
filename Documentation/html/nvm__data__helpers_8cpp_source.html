75 // This function replaced the older one above. It reads through the whole linked list
76 // and writes the nodes word by word. It also sets the 'nvramListTopVerify' pointer to
77 // the root node of the linked list it just wrote. This allows the verify function
78 // to know what to verify.
79 uint8_t writeNvmData(t_RECORD_NVM_DATA *pRecord)
80 {
81  uint8_t allGood = 1; // Eventually use this for a timeout
82  int8_t busy;
83  int16_t count;
84 
85  nvram_somethingToVerify = 1;
86 
87  for (uint16_t i = 0; i < nWords; i++)
88  {
89  if (allGood)
90  {
91  if (pRecord->detailedRecordHeader.UsePec)
92  smbusPec__->writeWord((uint8_t) pRecord->detailedRecordHeader.DeviceAddress,
93  pRecord->detailedRecordHeader.CommandCode,
94  words[i]);
95  else
96  smbusNoPec__->writeWord((uint8_t) pRecord->detailedRecordHeader.DeviceAddress,
97  pRecord->detailedRecordHeader.CommandCode,
98  words[i]);
99 
100  count = 0;
101  do
102  {
103  if (pRecord->detailedRecordHeader.UsePec)
104  busy = smbusPec__->readByte((uint8_t) pRecord->detailedRecordHeader.DeviceAddress, 0xef);
105  else
106  busy = smbusNoPec__->readByte((uint8_t) pRecord->detailedRecordHeader.DeviceAddress, 0xef);
107 
108  busy = (busy & 0x40)==0;
109  }
110  while (busy && (count++ < 4096));
111  if (count == 4097)
112  {
113  Serial.println(F("NVM Write Timeout"));
114  allGood = 0;
115  }
116  }
117  }
118  return allGood;
119 }
