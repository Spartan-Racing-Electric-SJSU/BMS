116  Serial.println(F("Please wait for programming EEPROM..."));

203 //! Repeats Linduino loop
204 //! @return void
205 void loop()
206 {
207  uint8_t user_command;
208  uint8_t res;
209  uint8_t model[7];
210  uint8_t revision[10];
211  uint8_t *addresses = NULL;
212 
213  if (Serial.available()) //! Checks for user input
214  {
215  user_command = read_int(); //! Reads the user command
216  if (user_command != 'm')
217  Serial.println(user_command);
218 
219  switch (user_command) //! Prints the appropriate submenu
220  {
221  case 1:
222  program_nvm();
223  break;
224  case 2:
225  verify_nvm();
226  // Lock the NVM in case the verify failed.
227  pmbus->smbus()->writeByte(0x5b, MFR_EE_UNLOCK, 0x00);
228  break;
229  case 3:
230  restore_nvm();
231  break;
232  case 4:
233  program_nvm();
234  reset_all_devices();
235  break;
236  case 5:
237  pmbus->clearFaultsGlobal();
238  break;
239  case 6:
240  pmbus->enablePec(ltc3880_i2c_address);
241  pmbus->enablePec(ltc2974_i2c_address);
242  pmbus->enablePec(ltc2977_i2c_address);
243  smbus = smbusPec;
244  pmbus->smbus(smbus);
245  pec = true;
246  break;
247  case 7:
248  pmbus->disablePec(ltc3880_i2c_address);
249  pmbus->disablePec(ltc2974_i2c_address);
250  pmbus->disablePec(ltc2977_i2c_address);
251  smbus = smbusNoPec;
252  pmbus->smbus(smbus);
253  pec = false;
254  break;
255  case 8:
256  addresses = smbus->probe(0);
257  while (*addresses != 0)
258  {
259  Serial.print(F("ADDR 0x"));
260  Serial.println(*addresses++, HEX);
261  }
262  break;
263  case 9:
264  reset_all_devices();
265  break;
266  default:
267  Serial.println(F("Incorrect Option"));
268  break;
269  }
270  print_prompt();
271  }
272 }
