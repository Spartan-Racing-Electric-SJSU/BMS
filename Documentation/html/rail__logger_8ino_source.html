104 //! Initialize Linduino
105 //! @return void
106 void setup()
107 {
108  uint32_t speed = 500000;
109 
110  Serial.begin(115200); //! Initialize the serial port to the PC
111 
112  if (doprint) Serial.println(F("Detecting Addresses..."));
113  detector->detect();
114  device = (devices = detector->getDevices());
115  while (*device != NULL)
116  {
117  if (doprint) Serial.print(F(" Device: "));
118  if (doprint) Serial.print((*device)->getType());
119  if (doprint) Serial.print(F(" Address: "));
120  if (doprint) Serial.print((*device)->getAddress(), HEX);
121 
122  if (doprint) Serial.print(F(", Speed: "));
123  if (doprint) Serial.print((*device)->getMaxSpeed(), DEC);
124  if (doprint) Serial.println();
125 
126  // Switch the device to Pec
127  pmbusNoPec->enablePec((*device)->getAddress());
128  (*device)->changePMBus(pmbusPec);
129 
130  // Speed test to find fastest speed the bus can operate at.
131  if (speed > (*device)->getMaxSpeed())
132  speed = (*device)->getMaxSpeed();
133 
134  device++;
135  }
136 
137  rail = (rails = detector->getRails());
138  while (*rail != NULL)
139  {
140  if (doprint) Serial.print(F(" Rail: "));
141  if (doprint) Serial.print(F(" Address: "));
142  if (doprint) Serial.print((*rail)->getAddress(), HEX);
143  if (doprint) Serial.print(F(" Pages: "));
144  if (doprint) Serial.print((*rail)->getNoPages(), DEC);
145  if (doprint) Serial.println();
146  (*rail)->changePMBus(pmbusPec);
147  rail++;
148  }
149 
150  if (doprint) Serial.print("Changing bus speed to ");
151  if (doprint) Serial.println(speed, DEC);
152  // This will change the speed of the I2C bus and enable clock stretching if > 100kHz
153  // Only affects Pec becuase the device was switched to Pec.
154  device = devices;
155  while ((*device) != 0)
156  {
157  (*device)->setSpeed(speed);
158  device++;
159  }
160  // Change the NoPec version to keep out world consistent.
161  pmbusNoPec->smbus()->i2cbus()->changeSpeed(speed);
162 
163  Serial.println(F("Id,Time(ms),Address,NoPages,Multiphase,Type,Value"));
164 
165 }

190 //! Repeats Linduino loop
191 //! @return void
192 void loop()
193 {
194  float f;
195  uint16_t status;
196 
197  // Telemetry
198  rail = rails;
199  while ((*rail) != NULL)
200  {
201  if ((*rail)->hasCapability(HAS_STATUS_WORD))
202  {
203  status = (*rail)->readStatusWord();
204  print_info(*rail);
205  Serial.print(F("SWORD,0x"));
206  Serial.println(status, HEX);
207  }
208 
209  if ((*rail)->hasCapability(HAS_VIN))
210  {
211  f = (*rail)->readVin(true);
212  print_info(*rail);
213  Serial.print(F("VIN,"));
214  Serial.println(f, DEC);
215  }
216 
217  if ((*rail)->hasCapability(HAS_VOUT))
218  {
219  f = (*rail)->readVout(true);
220  print_info(*rail);
221  Serial.print(F("VOUT,"));
222  Serial.println(f, DEC);
223  }
224 
225  if ((*rail)->hasCapability(HAS_IIN))
226  {
227  f = (*rail)->readIin(true);
228  print_info(*rail);
229  Serial.print(F("IIN,"));
230  Serial.println(f, DEC);
231  }
232 
233  if ((*rail)->hasCapability(HAS_IOUT))
234  {
235  f = (*rail)->readIout(true);
236  print_info(*rail);
237  Serial.print(F("Iout,"));
238  Serial.println(f, DEC);
239  }
240 
241  if ((*rail)->hasCapability(HAS_PIN))
242  {
243  f = (*rail)->readPin(true);
244  print_info(*rail);
245  Serial.print(F("PIN,"));
246  Serial.println(f, DEC);
247  }
248 
249  if ((*rail)->hasCapability(HAS_POUT))
250  {
251  f = (*rail)->readPout(true);
252  print_info(*rail);
253  Serial.print(F("POUT,"));
254  Serial.println(f, DEC);
255  }
256 
257  if ((*rail)->hasCapability(HAS_TEMP))
258  {
259  f = (*rail)->readInternalTemperature(true);
260  print_info(*rail);
261  Serial.print(F("TEMP,"));
262  Serial.println(f, DEC);
263  }
264 
265  f = (*rail)->readEfficiency(true);
266  if (f > 20.0) // Weeds out no load rails or unmeasuremable rails.
267  {
268  print_info(*rail);
269  Serial.print(F("EFF,"));
270  Serial.println(f, DEC);
271  }
272 
273  if ((*rail)->isMultiphase())
274  {
275  f = (*rail)->readPhaseBalance(true);
276  print_info(*rail);
277  Serial.print(F("IBAL,"));
278  Serial.println(f, DEC);
279  }
280 
281  rail++;
282  }
283 
284  delay(POLL_DELAY);
