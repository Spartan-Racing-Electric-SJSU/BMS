97 void write_custom_table(uint8_t chip_select, struct table_coeffs coefficients[64], uint16_t start_address, uint8_t table_length)
98 {
99  int8_t i;
100  uint32_t coeff;
101 
102  output_low(chip_select);
103 
104  SPI.transfer(WRITE_TO_RAM);
105  SPI.transfer(highByte(start_address));
106  SPI.transfer(lowByte(start_address));
107 
108  for (i=0; i< table_length; i++)
109  {
110  coeff = coefficients[i].measurement;
111  SPI.transfer((uint8_t)(coeff >> 16));
112  SPI.transfer((uint8_t)(coeff >> 8));
113  SPI.transfer((uint8_t)coeff);
114 
115  coeff = coefficients[i].temperature;
116  SPI.transfer((uint8_t)(coeff >> 16));
117  SPI.transfer((uint8_t)(coeff >> 8));
118  SPI.transfer((uint8_t)coeff);
119  }
120  output_high(chip_select);
121 }

150 void eeprom_transfer(uint8_t chip_select, uint8_t eeprom_read_or_write)
151 // Read from or write to the EEPROM.
152 // To read from the EEPROM, pass READ_FROM_EEPROM into eeprom_read_or_write.
153 // To write to the EEPROM, pass WRITE_TO_EEPROM into eeprom_read_or_write.
154 {
155  uint8_t eeprom_status;
156  Serial.println("** EEPROM transfer started ** ");
157 
158  // Set EEPROM key
159  transfer_four_bytes(chip_select, WRITE_TO_RAM, EEPROM_START_ADDRESS, EEPROM_KEY);
160 
161  // Set EEPROM read/write
162  transfer_byte(chip_select, WRITE_TO_RAM, COMMAND_STATUS_REGISTER, eeprom_read_or_write);
163 
164  // Wait for read/write to finish
165  wait_for_process_to_finish(chip_select);
166 
167  // Check for success
168  eeprom_status = transfer_byte(chip_select, READ_FROM_RAM, EEPROM_STATUS_REGISTER, 0);
169  if (eeprom_status == 0)
170  {
171  Serial.println("** EEPROM transfer succeeded ** ");
172  }
173  else
174  {
175  Serial.print(F("** EEPROM transfer had a problem. Status byte ="));
176  Serial.println(eeprom_status);
177  }
178 }

216 void get_result(uint8_t chip_select, uint8_t channel_number, uint8_t channel_output)
217 {
218  uint32_t raw_data;
219  uint8_t fault_data;
220  uint16_t start_address = get_start_address(CONVERSION_RESULT_MEMORY_BASE, channel_number);
221  uint32_t raw_conversion_result;
222 
223  raw_data = transfer_four_bytes(chip_select, READ_FROM_RAM, start_address, 0);
224 
225  Serial.print(F("\nChannel "));
226  Serial.println(channel_number);
227 
228  // 24 LSB's are conversion result
229  raw_conversion_result = raw_data & 0xFFFFFF;
230  print_conversion_result(raw_conversion_result, channel_output);
231 
232  // If you're interested in the raw voltage or resistance, use the following
233  if (channel_output != VOLTAGE)
234  {
235  read_voltage_or_resistance_results(chip_select, channel_number);
236  }
237 
238  // 8 MSB's show the fault data
239  fault_data = raw_data >> 24;
240  print_fault_data(fault_data);
241 }

244 void print_conversion_result(uint32_t raw_conversion_result, uint8_t channel_output)
245 {
246  int32_t signed_data = raw_conversion_result;
247  float scaled_result;
248 
249  // Convert the 24 LSB's into a signed 32-bit integer
250  if (signed_data & 0x800000)
251  signed_data = signed_data | 0xFF000000;
252 
253  // Translate and print result
254  if (channel_output == TEMPERATURE)
255  {
256  scaled_result = float(signed_data) / 1024;
257  Serial.print(F(" Temperature = "));
258  Serial.println(scaled_result);
259  }
260  else if (channel_output == VOLTAGE)
261  {
262  scaled_result = float(signed_data) / 2097152;
263  Serial.print(F(" Direct ADC reading in V = "));
264  Serial.println(scaled_result);
265  }

283 // Translate the fault byte into usable fault data and print it out
284 void print_fault_data(uint8_t fault_byte)
285 {
286  //
287  Serial.print(F(" FAULT DATA = "));
288  Serial.println(fault_byte, BIN);
289 
290  if (fault_byte & SENSOR_HARD_FAILURE)
291  Serial.println(F(" - SENSOR HARD FALURE"));
292  if (fault_byte & ADC_HARD_FAILURE)
293  Serial.println(F(" - ADC_HARD_FAILURE"));
294  if (fault_byte & CJ_HARD_FAILURE)
295  Serial.println(F(" - CJ_HARD_FAILURE"));
296  if (fault_byte & CJ_SOFT_FAILURE)
297  Serial.println(F(" - CJ_SOFT_FAILURE"));
298  if (fault_byte & SENSOR_ABOVE)
299  Serial.println(F(" - SENSOR_ABOVE"));
300  if (fault_byte & SENSOR_BELOW)
301  Serial.println(F(" - SENSOR_BELOW"));
302  if (fault_byte & ADC_RANGE_ERROR)
303  Serial.println(F(" - ADC_RANGE_ERROR"));
304  if (!(fault_byte & VALID))
305  Serial.println(F("INVALID READING !!!!!!"));
306  if (fault_byte == 0b11111111)
307  Serial.println(F("CONFIGURATION ERROR !!!!!!"));
308 }

311 // SPI RAM data transfer
312 // *********************
313 // To write to the RAM, set ram_read_or_write = WRITE_TO_RAM.
314 // To read from the RAM, set ram_read_or_write = READ_FROM_RAM.
315 // input_data is the data to send into the RAM. If you are reading from the part, set input_data = 0.

317 uint32_t transfer_four_bytes(uint8_t chip_select, uint8_t ram_read_or_write, uint16_t start_address, uint32_t input_data)
318 {
319  uint32_t output_data;
320  uint8_t tx[7], rx[7];
321 
322  tx[6] = ram_read_or_write;
323  tx[5] = highByte(start_address);
324  tx[4] = lowByte(start_address);
325  tx[3] = (uint8_t)(input_data >> 24);
326  tx[2] = (uint8_t)(input_data >> 16);
327  tx[1] = (uint8_t)(input_data >> 8);
328  tx[0] = (uint8_t) input_data;
329 
330  spi_transfer_block(chip_select, tx, rx, 7);
331 
332  output_data = (uint32_t) rx[3] << 24 |
333  (uint32_t) rx[2] << 16 |
334  (uint32_t) rx[1] << 8 |
335  (uint32_t) rx[0];
336 
337  return output_data;
338 }

341 uint8_t transfer_byte(uint8_t chip_select, uint8_t ram_read_or_write, uint16_t start_address, uint8_t input_data)
342 {
343  uint8_t tx[4], rx[4];
344 
345  tx[3] = ram_read_or_write;
346  tx[2] = (uint8_t)(start_address >> 8);
347  tx[1] = (uint8_t)start_address;
348  tx[0] = input_data;
349  spi_transfer_block(chip_select, tx, rx, 4);
350  return rx[0];
351 }
