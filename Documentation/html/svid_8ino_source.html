102 //! Repeats Linduino loop
103 //! @return void
104 void loop()
105 {
106  uint8_t user_command;
107  uint8_t model[7];
108  uint8_t *addresses = NULL;
109  float voltage;
110 
111  if (Serial.available()) //! Checks for user input
112  {
113  user_command = read_int(); //! Reads the user command
114  if (user_command != 'm')
115  Serial.println(user_command);
116  voltage = 0.85;
117  switch (user_command) //! Prints the appropriate submenu
118  {
119  case 0:
120  pmbus->setVout(ltc3880_i2c_address, voltage - 0.060);
121  break;
122  case 1:
123  pmbus->setVout(ltc3880_i2c_address, voltage - 0.045);
124  break;
125  case 2:
126  pmbus->setVout(ltc3880_i2c_address, voltage - 0.030);
127  break;
128  case 3:
129  pmbus->setVout(ltc3880_i2c_address, voltage - 0.015);
130  break;
131  case 4:
132  pmbus->setVout(ltc3880_i2c_address, voltage);
133  break;
134  case 5:
135  pmbus->enablePec(ltc3880_i2c_address);
136  pmbus->enablePec(ltc2974_i2c_address);
137  pmbus->enablePec(ltc2977_i2c_address);
138  smbus = smbusPec;
139  pmbus = pmbusPec;
140  break;
141  case 6:
142  pmbus->disablePec(ltc3880_i2c_address);
143  pmbus->disablePec(ltc2974_i2c_address);
144  pmbus->disablePec(ltc2977_i2c_address);
145  smbus = smbusNoPec;
146  pmbus = pmbusNoPec;
147  break;
148  case 7:
149  addresses = smbus->probe(0);
150  while (*addresses != 0)
151  {
152  Serial.print("ADDR 0x");
153  Serial.println(*addresses++, HEX);
154  }
155  break;
156  case 8 :
157  pmbus->resetGlobal();
158  break;
159  default:
160  Serial.println("Incorrect Option");
161  break;
162  }
163  print_prompt();
164  }
165 }
